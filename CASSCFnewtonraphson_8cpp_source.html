225 double CheMPS2::CASSCF::augmentedHessianNR(double * gradient, double * updateNorm){
226 
227  /* A good read to understand
228  (1) how the augmented Hessian arises from a rational function optimization
229  (2) where the parameter lambda in Eq. (22) of Yanai, IJQC 109, 2178-2190 (2009) comes from
230  (3) why the smallest algebraic eigenvalue + corresponding eigenvector should be retained for minimizations
231  Banerjee, Adams, Simons, Shepard, "Search for stationary points on surfaces",
232  J. Phys. Chem. 1985, volume 89, pages 52-57, doi:10.1021/j100247a015 */
233 
234  //Calculate the gradient
235  int x_linearlength = unitary->getNumVariablesX();
236  double gradNorm = calcGradient(gradient);
237 
238  //Calculate the Hessian
239  int aug_linlength = x_linearlength+1;
240  int size = aug_linlength * aug_linlength;
241  double * hessian = new double[size];
242  calcHessian(hessian, aug_linlength);
243 
244  //Augment the gradient into the Hessian matrix
245  for (int cnt=0; cnt<x_linearlength; cnt++){
246  hessian[cnt + x_linearlength*aug_linlength] = gradient[cnt];
247  hessian[x_linearlength + aug_linlength*cnt] = gradient[cnt];
248  }
249  hessian[x_linearlength + aug_linlength*x_linearlength] = 0.0;
250 
251  //Find its lowest eigenvalue and vector
252  double * work = new double[size];
253  double * eigen = new double[aug_linlength];
254  char jobz = 'V';
255  char uplo = 'U';
256  int info;
257  dsyev_(&jobz,&uplo,&aug_linlength,hessian,&aug_linlength,eigen,work,&size,&info);
258 
259  if (CheMPS2::DMRGSCF_debugPrint){
260  cout << " DMRGSCF::augmentedHessianNR : Lowest eigenvalue = " << eigen[0] << endl;
261  cout << " DMRGSCF::augmentedHessianNR : The last number of the eigenvector (which will be rescaled to one) = " << hessian[x_linearlength] << endl;
262  }
263  double scalar = 1.0/hessian[x_linearlength];
264  int inc = 1;
265  dscal_(&x_linearlength,&scalar,hessian,&inc);
266 
267  //Copy the update to the gradient vector --> needed for updates of the unitary and DIIS
268  dcopy_(&x_linearlength,hessian,&inc,gradient,&inc);
269  updateNorm[0] = 0.0;
270  for (int cnt=0; cnt<unitary->getNumVariablesX(); cnt++){ updateNorm[0] += gradient[cnt] * gradient[cnt]; }
271  updateNorm[0] = sqrt(updateNorm[0]);
272  cout << " DMRGSCF::augmentedHessianNR : Norm of the update = " << updateNorm[0] << endl;
273 
274  delete [] hessian;
275  delete [] eigen;
276  delete [] work;
277 
278  return gradNorm;
279 
280 }

282 double CheMPS2::CASSCF::calcGradient(double * gradient){
283 
284  for (int cnt=0; cnt<unitary->getNumVariablesX(); cnt++){
285  gradient[cnt] = 2*( Fmat( unitary->getFirstIndex(cnt), unitary->getSecondIndex(cnt) ) - Fmat( unitary->getSecondIndex(cnt), unitary->getFirstIndex(cnt) ));
286  }
287 
288  double gradNorm = 0.0;
289  for (int cnt=0; cnt<unitary->getNumVariablesX(); cnt++){ gradNorm += gradient[cnt] * gradient[cnt]; }
290  gradNorm = sqrt(gradNorm);
291  cout << " DMRGSCF::calcGradient : Norm of the gradient = " << gradNorm << endl;
292  return gradNorm;

296 void CheMPS2::CASSCF::calcHessian(double * hessian, const int rowjump){
297 
298  const int lindim = ( unitary->getNumVariablesX() * ( unitary->getNumVariablesX() + 1 ))/2;
299 
300  #pragma omp parallel for schedule(static)
301  for (int count=0; count<lindim; count++){
302 
303  int col = 1;
304  while ( (col*(col+1))/2 <= count ) col++;
305  col -= 1;
306  int row = count - (col*(col+1))/2;
307 
308  int p_index = unitary->getFirstIndex(row);
309  int q_index = unitary->getSecondIndex(row);
310  int r_index = unitary->getFirstIndex(col);
311  int s_index = unitary->getSecondIndex(col);
312  hessian[row + rowjump * col] = Wmat(p_index,q_index,r_index,s_index)
313  - Wmat(q_index,p_index,r_index,s_index)
314  - Wmat(p_index,q_index,s_index,r_index)
315  + Wmat(q_index,p_index,s_index,r_index);
316  hessian[col + rowjump * row] = hessian[row + rowjump * col];
317 
318  }

322 double CheMPS2::CASSCF::Wmat(const int index1, const int index2, const int index3, const int index4) const{
323 
324  const int irrep1 = HamOrig->getOrbitalIrrep(index1);
325  const int irrep2 = HamOrig->getOrbitalIrrep(index2);
326  if (irrep1 != irrep2){ return 0.0; } //From now on: irrep1 == irrep2
327 
328  const int irrep3 = HamOrig->getOrbitalIrrep(index3);
329  const int irrep4 = HamOrig->getOrbitalIrrep(index4);
330  if (irrep3 != irrep4){ return 0.0; } //From now on: irrep3 == irrep4
331 
332  double value = 0.0;
333 
334  if ( (irrep1 == irrep3) && (index2 == index3) ) { //irrep1 == irrep2 == irrep3 == irrep4
335  value = Fmat(index1,index4) + Fmat(index4,index1);
336  }
337 
338  if (index1 >= iHandler->getOrigNVIRTstart(irrep1)){ return value; }
339  if (index3 >= iHandler->getOrigNVIRTstart(irrep3)){ return value; } //index1 and index3 are now certainly not virtual!
340 
341  const int relIndex1 = index1 - iHandler->getOrigNOCCstart(irrep1);
342  const int relIndex2 = index2 - iHandler->getOrigNOCCstart(irrep2);
343  const int relIndex3 = index3 - iHandler->getOrigNOCCstart(irrep3);
344  const int relIndex4 = index4 - iHandler->getOrigNOCCstart(irrep4);
345 
346  if (index1 < iHandler->getOrigNDMRGstart(irrep1)){
347  if (index3 < iHandler->getOrigNDMRGstart(irrep3)){
348 
349  // (index1,index3) (occupied,occupied) --> (alpha,beta) can be (occupied,occupied) or (active,active)
350  if (index1==index3){
351 
352  // Part of one-body matrix elements if 1-RDM occ indices
353  value += 4 * ( TmatRotated(index2,index4) + QmatOCC(index2,index4) + QmatACT(index2, index4) );
354  value += 12 * VmatRotated->get(irrep2, irrep4, irrep1, irrep1, relIndex2, relIndex4, relIndex1, relIndex1)
355  - 4 * VmatRotated->get(irrep2, irrep1, irrep4, irrep1, relIndex2, relIndex1, relIndex4, relIndex1);
356  return value;
357 
358  } else { //index1 != index3
359 
360  value += 16 * VmatRotated->get(irrep2, irrep4, irrep1, irrep3, relIndex2, relIndex4, relIndex1, relIndex3)
361  - 4 * VmatRotated->get(irrep2, irrep1, irrep4, irrep3, relIndex2, relIndex1, relIndex4, relIndex3)
362  - 4 * VmatRotated->get(irrep2, irrep4, irrep3, irrep1, relIndex2, relIndex4, relIndex3, relIndex1);
363  return value;
364 
365  }
366 
367  } else {
368 
369  const int DMRGindex3 = iHandler->getDMRGcumulative(irrep3) + index3 - iHandler->getOrigNDMRGstart(irrep3);
370 
371  // (index1,index3) (occupied,active) --> (alpha,beta) can be (active,occupied) or (occupied,active)
372  for (int alpha_index=iHandler->getOrigNDMRGstart(irrep3); alpha_index<iHandler->getOrigNVIRTstart(irrep3); alpha_index++){
373  const int DMRGindexALPHA = iHandler->getDMRGcumulative(irrep3) + alpha_index - iHandler->getOrigNDMRGstart(irrep3);
374  const int relALPHA = alpha_index - iHandler->getOrigNOCCstart(irrep3);
375  value += 2 * DMRG1DM[ DMRGindex3 + nOrbDMRG * DMRGindexALPHA ]
376  * ( 4 * VmatRotated->get(irrep2, irrep4, irrep1, irrep3, relIndex2, relIndex4, relIndex1, relALPHA)
377  - VmatRotated->get(irrep2, irrep4, irrep3, irrep1, relIndex2, relIndex4, relALPHA, relIndex1)
378  - VmatRotated->get(irrep2, irrep1, irrep4, irrep3, relIndex2, relIndex1, relIndex4, relALPHA) );
379  }
380  return value;
381 
382  }
383  } else {
384 
385  const int DMRGindex1 = iHandler->getDMRGcumulative(irrep1) + index1 - iHandler->getOrigNDMRGstart(irrep1);
386 
387  if (index3 < iHandler->getOrigNDMRGstart(irrep3)){
388 
389  // (index1,index3) (active,occupied) --> (alpha,beta) can be (active,occupied) or (occupied,active)
390  for (int alpha_index=iHandler->getOrigNDMRGstart(irrep1); alpha_index<iHandler->getOrigNVIRTstart(irrep1); alpha_index++){
391  const int DMRGindexALPHA = iHandler->getDMRGcumulative(irrep1) + alpha_index - iHandler->getOrigNDMRGstart(irrep1);
392  const int relALPHA = alpha_index - iHandler->getOrigNOCCstart(irrep1);
393  value += 2 * DMRG1DM[ DMRGindex1 + nOrbDMRG * DMRGindexALPHA ]
394  * ( 4 * VmatRotated->get(irrep2, irrep4, irrep1, irrep3, relIndex2, relIndex4, relALPHA, relIndex3)
395  - VmatRotated->get(irrep2, irrep1, irrep4, irrep3, relIndex2, relALPHA, relIndex4, relIndex3)
396  - VmatRotated->get(irrep2, irrep4, irrep3, irrep1, relIndex2, relIndex4, relIndex3, relALPHA) );
397  }
398  return value;
399 
400  } else {
401 
402  const int DMRGindex3 = iHandler->getDMRGcumulative(irrep3) + index3 - iHandler->getOrigNDMRGstart(irrep3);
403 
404  // (index1,index3) (active,active) --> (alpha,beta) can be (occupied,occupied) or (active,active)
405  // Case1: (alpha,beta)==(occ,occ) --> alpha == beta
406  // Part of one-body matrix elements if 1-RDM active indices
407  value += 2 * DMRG1DM[ DMRGindex1 + nOrbDMRG * DMRGindex3 ] * ( TmatRotated(index2,index4) + QmatOCC(index2,index4) );
408 
409  // Case2: (alpha,beta)==(act,act)
410  const int productIrrep = SymmInfo.directProd(irrep1,irrep3);
411  for (int irrep_alpha=0; irrep_alpha<numberOfIrreps; irrep_alpha++){
412  int irrep_beta = SymmInfo.directProd(productIrrep,irrep_alpha);
413  for (int alpha_index=iHandler->getOrigNDMRGstart(irrep_alpha); alpha_index<iHandler->getOrigNVIRTstart(irrep_alpha); alpha_index++){
414  const int DMRGindexALPHA = iHandler->getDMRGcumulative(irrep_alpha) + alpha_index - iHandler->getOrigNDMRGstart(irrep_alpha);
415  const int relALPHA = alpha_index - iHandler->getOrigNOCCstart(irrep_alpha);
416  for (int beta_index=iHandler->getOrigNDMRGstart(irrep_beta); beta_index<iHandler->getOrigNVIRTstart(irrep_beta); beta_index++){
417  const int DMRGindexBETA = iHandler->getDMRGcumulative(irrep_beta) + beta_index - iHandler->getOrigNDMRGstart(irrep_beta);
418  const int relBETA = beta_index - iHandler->getOrigNOCCstart(irrep_beta);
419  const double TwoDMval1 = DMRG2DM[ DMRGindex3 + nOrbDMRG * ( DMRGindexALPHA + nOrbDMRG * ( DMRGindex1 + nOrbDMRG * DMRGindexBETA ) ) ];
420  const double TwoDMval2 = DMRG2DM[ DMRGindex3 + nOrbDMRG * ( DMRGindexALPHA + nOrbDMRG * ( DMRGindexBETA + nOrbDMRG * DMRGindex1 ) ) ];
421  const double TwoDMval3 = DMRG2DM[ DMRGindex3 + nOrbDMRG * ( DMRGindex1 + nOrbDMRG * ( DMRGindexBETA + nOrbDMRG * DMRGindexALPHA ) ) ];

461 double CheMPS2::CASSCF::FmatHelper(const int index1, const int index2) const{
462 
463  const int irrep1 = HamOrig->getOrbitalIrrep(index1);
464  const int irrep2 = HamOrig->getOrbitalIrrep(index2);
465  if (irrep1 != irrep2){ return 0.0; } //From now on: both irreps are the same.
466 
467  if (index1 >= iHandler->getOrigNVIRTstart(irrep1)){ return 0.0; } //index1 virtual: 1/2DM returns 0.0
468 
469  double value = 0.0;
470 
471  if (index1 < iHandler->getOrigNDMRGstart(irrep1)){ //index1 occupied
472 
473  value += 2 * ( TmatRotated(index2,index1) + QmatOCC(index2,index1) + QmatACT(index2,index1) );
474 
475  } else { //index1 active
476 
477  const int DMRGindex1 = iHandler->getDMRGcumulative(irrep1) + index1 - iHandler->getOrigNDMRGstart(irrep1);
478 
479  //1DM will only return non-zero if r_index also active, and corresponds to the same irrep
480  for (int r_index=iHandler->getOrigNDMRGstart(irrep1); r_index<iHandler->getOrigNVIRTstart(irrep1); r_index++){
481  const int DMRGindexR = iHandler->getDMRGcumulative(irrep1) + r_index - iHandler->getOrigNDMRGstart(irrep1);
482  value += DMRG1DM[ DMRGindex1 + nOrbDMRG * DMRGindexR ] * ( TmatRotated(index2,r_index) + QmatOCC(index2,r_index) );
483  }
484 
485  //All the summation indices are active
486  const int relIndex2 = index2 - iHandler->getOrigNOCCstart(irrep2);
487  for (int irrep_r=0; irrep_r<numberOfIrreps; irrep_r++){
488  const int productIrrep = SymmInfo.directProd(irrep1,irrep_r);
489  for (int irrep_s=0; irrep_s<numberOfIrreps; irrep_s++){
490  int irrep_t = SymmInfo.directProd(productIrrep,irrep_s);
491  for (int r_index=iHandler->getOrigNDMRGstart(irrep_r); r_index<iHandler->getOrigNVIRTstart(irrep_r); r_index++){
492  const int DMRGindexR = iHandler->getDMRGcumulative(irrep_r) + r_index - iHandler->getOrigNDMRGstart(irrep_r);
493  const int relR = r_index - iHandler->getOrigNOCCstart(irrep_r);
494  for (int s_index=iHandler->getOrigNDMRGstart(irrep_s); s_index<iHandler->getOrigNVIRTstart(irrep_s); s_index++){
495  const int DMRGindexS = iHandler->getDMRGcumulative(irrep_s) + s_index - iHandler->getOrigNDMRGstart(irrep_s);
496  const int relS = s_index - iHandler->getOrigNOCCstart(irrep_s);
497  for (int t_index=iHandler->getOrigNDMRGstart(irrep_t); t_index<iHandler->getOrigNVIRTstart(irrep_t); t_index++){
498  const int DMRGindexT = iHandler->getDMRGcumulative(irrep_t) + t_index - iHandler->getOrigNDMRGstart(irrep_t);
499  const int relT = t_index - iHandler->getOrigNOCCstart(irrep_t);
500  const double TwoDMvalue = DMRG2DM[ DMRGindex1 + nOrbDMRG * ( DMRGindexR + nOrbDMRG * ( DMRGindexS + nOrbDMRG * DMRGindexT ) ) ];
501  value += TwoDMvalue * VmatRotated->get(irrep2, irrep_r, irrep_s, irrep_t, relIndex2, relR, relS, relT);
502  }
503  }
504  }
505  }
506  }
