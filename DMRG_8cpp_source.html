156 double CheMPS2::DMRG::Solve(){
157 
158  bool change = (TotalMinEnergy<1e8) ? true : false; //1 sweep from right to left: fixed virtual dimensions
159 
160  double Energy = 0.0;
161 
162  for (int instruction=0; instruction < OptScheme->getNInstructions(); instruction++){
163 
164  int nIterations = 0;
165  double EnergyPrevious = Energy + 10 * OptScheme->getEconv(instruction); //Guarantees that there's always at least 1 left-right sweep
166 
167  while ( (fabs(Energy-EnergyPrevious) > OptScheme->getEconv(instruction) ) && ( nIterations < OptScheme->getMaxSweeps(instruction) )){
168 
169  struct timeval start, end;
170  EnergyPrevious = Energy;
171  gettimeofday(&start, NULL);
172  Energy = sweepleft(change, instruction);
173  gettimeofday(&end, NULL);
174  double elapsed = (end.tv_sec - start.tv_sec) + 1e-6 * (end.tv_usec - start.tv_usec);
175  cout << "*** Information on left sweep " << nIterations << " of instruction " << instruction << ":" << endl;
176  cout << "*** Elapsed wall time = " << elapsed << " seconds" << endl;
177  cout << "*** Minimum energy = " << LastMinEnergy << endl;
178  cout << "*** Maximum discarded weight = " << MaxDiscWeightLastSweep << endl;
179  if (!change) change = true; //rest of sweeps: variable virtual dimensions
180  gettimeofday(&start, NULL);
181  Energy = sweepright(change, instruction);
182  gettimeofday(&end, NULL);
183  elapsed = (end.tv_sec - start.tv_sec) + 1e-6 * (end.tv_usec - start.tv_usec);
184  cout << "*** Information on right sweep " << nIterations << " of instruction " << instruction << ":" << endl;
185  cout << "*** Elapsed wall time = " << elapsed << " seconds" << endl;
186  cout << "*** Minimum energy = " << LastMinEnergy << endl;
187  cout << "*** Maximum discarded weight = " << MaxDiscWeightLastSweep << endl;
188  if (CheMPS2::DMRG_storeMpsOnDisk){ saveMPS(MPSstoragename, MPS, denBK, false); }
189 
190  nIterations++;
191 
192  cout << "*** Energy difference with respect to previous leftright sweep = " << fabs(Energy-EnergyPrevious) << endl;
193  if (Exc_activated){ calcOverlapsWithLowerStates(); }
194 
195  }
196 
197  cout << "****************************************************************************" << endl;
198  cout << "*** Information on completed instruction " << instruction << ":" << endl;
199  cout << "*** The reduced virtual dimension DSU(2) = " << OptScheme->getD(instruction) << endl;
200  cout << "*** Minimum energy encountered during all instructions = " << TotalMinEnergy << endl;
201  cout << "*** Minimum energy encountered during the last sweep = " << LastMinEnergy << endl;
202  cout << "*** Maximum discarded weight during the last sweep = " << MaxDiscWeightLastSweep << endl;
203  cout << "****************************************************************************" << endl;
204 
205  }
206 
207  return TotalMinEnergy;
208 
209 }

211 double CheMPS2::DMRG::sweepleft(const bool change, const int instruction){
212 
213  double Energy = 0.0;
214  double NoiseLevel = OptScheme->getNoisePrefactor(instruction) * MaxDiscWeightLastSweep;
215  MaxDiscWeightLastSweep = 0.0;
216  LastMinEnergy = 1e8;
217 
218  for (int index = L-2; index>0; index--){
219  //Construct S
220  Sobject * denS = new Sobject(index,denBK->gIrrep(index),denBK->gIrrep(index+1),denBK);
221  denS->Join(MPS[index],MPS[index+1]);
222 
223  //Feed everything to the solver
224  Heff Solver(denBK, Prob);
225  double ** VeffTilde = NULL;
226  if (Exc_activated){
227  VeffTilde = new double*[nStates-1];
228  for (int cnt=0; cnt<nStates-1; cnt++){
229  VeffTilde[cnt] = new double[denS->gKappa2index(denS->gNKappa())];
230  calcVeffTilde(VeffTilde[cnt], denS, cnt);
231  }
232  }
233  Energy = Solver.SolveDAVIDSON(denS, Ltensors, Atensors, Btensors, Ctensors, Dtensors, S0tensors, S1tensors, F0tensors, F1tensors, Qtensors, Xtensors, nStates-1, VeffTilde);
234  if (Exc_activated){
235  //calcOverlapsWithLowerStatesDuringSweeps_debug(VeffTilde, denS);
236  for (int cnt=0; cnt<nStates-1; cnt++){ delete [] VeffTilde[cnt]; }
237  delete [] VeffTilde;
238  }
239  Energy += Prob->gEconst();
240  if (Energy<TotalMinEnergy){ TotalMinEnergy = Energy; }
241  if (Energy<LastMinEnergy){ LastMinEnergy = Energy; }
242 
243  //Decompose the S-object
244  if (NoiseLevel>0.0){ denS->addNoise(NoiseLevel); }
245  double discWeight = denS->Split(MPS[index],MPS[index+1],OptScheme->getD(instruction),false,change);
246  delete denS;
247  if (discWeight > MaxDiscWeightLastSweep){ MaxDiscWeightLastSweep = discWeight; }
248 
249  //Print info
250  cout << "Energy at sites (" << index << ", " << (index+1) << ") is " << Energy << endl;
251  if (CheMPS2::DMRG_printDiscardedWeight && change){ cout << " Info(DMRG) : Discarded weight in SVD decomp. (non-reduced) = " << discWeight << endl; }
252 
253  //Prepare for next step
254  updateMovingLeftSafe(index);
255 
256  }
257 
258  return Energy;
259
