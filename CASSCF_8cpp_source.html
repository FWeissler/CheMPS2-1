199 double CheMPS2::CASSCF::get2DMrotated(const int index1, const int index2, const int index3, const int index4) const{
200 
201  if ((index1<0) || (index1>=L)){ return 0.0; }
202  if ((index2<0) || (index2>=L)){ return 0.0; }
203  if ((index3<0) || (index3>=L)){ return 0.0; }
204  if ((index4<0) || (index4>=L)){ return 0.0; } //Within bounds now
205 
206  const int irrep1 = HamOrig->getOrbitalIrrep(index1);
207  const int irrep2 = HamOrig->getOrbitalIrrep(index2);
208  const int irrep3 = HamOrig->getOrbitalIrrep(index3);
209  const int irrep4 = HamOrig->getOrbitalIrrep(index4);
210 
211  if (SymmInfo.directProd(irrep1,irrep2) != SymmInfo.directProd(irrep3,irrep4)){ return 0.0; } //Matrix element is symmetry allowed
212 
213  int cnt1 = index1 - jumpsHamOrig[irrep1];
214  int cnt2 = index2 - jumpsHamOrig[irrep2];
215  int cnt3 = index3 - jumpsHamOrig[irrep3];
216  int cnt4 = index4 - jumpsHamOrig[irrep4];
217 
218  if (cnt1 >= Nocc[irrep1] + NDMRG[irrep1]){ return 0.0; }
219  if (cnt2 >= Nocc[irrep2] + NDMRG[irrep2]){ return 0.0; }
220  if (cnt3 >= Nocc[irrep3] + NDMRG[irrep3]){ return 0.0; }
221  if (cnt4 >= Nocc[irrep4] + NDMRG[irrep4]){ return 0.0; } //No virtual orbitals now
222 
223  const int numberOcc = ((cnt1 < Nocc[irrep1]) ? 1 : 0) + ((cnt2 < Nocc[irrep2]) ? 1 : 0) + ((cnt3 < Nocc[irrep3]) ? 1 : 0) + ((cnt4 < Nocc[irrep4]) ? 1 : 0);
224 
225  if ((numberOcc%2)!=0){ return 0.0; } //Now it is allowed --> need to find out its value.
226 
227  if (numberOcc==4){
228  if (index3==index4){
229  if ((index1==index2) && (index1==index3)){ return 2.0; } //all indices equal
230  else { return 0.0; }
231  }
232  return (((index1 == index3) && (index2 == index4)) ? 4.0 : 0.0 ) - (((index1 == index4) && (index2 == index3)) ? 2.0 : 0.0 );
233  }
234 
235  if (numberOcc==0){
236 
237  int DMRGindex1 = 0;
238  int DMRGindex2 = 0;
239  int DMRGindex3 = 0;
240  int DMRGindex4 = 0;
241 
242  for (int bla=0; bla<irrep1; bla++){ DMRGindex1 += NDMRG[bla]; }
243  for (int bla=0; bla<irrep2; bla++){ DMRGindex2 += NDMRG[bla]; }
244  for (int bla=0; bla<irrep3; bla++){ DMRGindex3 += NDMRG[bla]; }
245  for (int bla=0; bla<irrep4; bla++){ DMRGindex4 += NDMRG[bla]; }
246 
247  DMRGindex1 += cnt1 - Nocc[irrep1];
248  DMRGindex2 += cnt2 - Nocc[irrep2];
249  DMRGindex3 += cnt3 - Nocc[irrep3];
250  DMRGindex4 += cnt4 - Nocc[irrep4];
251 
252  return theDMRG2DM->getTwoDMA_HAM(DMRGindex1, DMRGindex2, DMRGindex3, DMRGindex4);
253 
254  }
255 
256  //Now two indices are occ, two are active
257  if (cnt1<Nocc[irrep1]){ // index1 in occ
258 
259  if (cnt3<Nocc[irrep3]){
260 
261  if (cnt1 != cnt3){
262  return 0.0;
263  } else {
264  int DMRGindex2 = 0;
265  int DMRGindex4 = 0;
266  for (int bla=0; bla<irrep2; bla++){ DMRGindex2 += NDMRG[bla]; }
267  for (int bla=0; bla<irrep4; bla++){ DMRGindex4 += NDMRG[bla]; }
268  DMRGindex2 += cnt2 - Nocc[irrep2];
269  DMRGindex4 += cnt4 - Nocc[irrep4];
270 
271  return 2 * DMRG1DM[DMRGindex2 + nOrbDMRG*DMRGindex4]; //2 * get1DMrotated(index2,index4);
272  }
273 
274  }
275  if (cnt4<Nocc[irrep4]){
276 
277  if (cnt1 != cnt4){
278  return 0.0;
279  } else {
280  int DMRGindex2 = 0;
281  int DMRGindex3 = 0;
282  for (int bla=0; bla<irrep2; bla++){ DMRGindex2 += NDMRG[bla]; }
283  for (int bla=0; bla<irrep3; bla++){ DMRGindex3 += NDMRG[bla]; }
284  DMRGindex2 += cnt2 - Nocc[irrep2];
285  DMRGindex3 += cnt3 - Nocc[irrep3];
286 
287  return - DMRG1DM[DMRGindex2 + nOrbDMRG*DMRGindex3]; // - get1DMrotated(index2,index3);
288  }
289 
290  }
291 
292  } else { // index1 in act
293 
294  if (cnt3>=Nocc[irrep3]){
295 
296  if (cnt2 != cnt4){
297  return 0.0;
298  } else {
299  int DMRGindex1 = 0;
300  int DMRGindex3 = 0;
301  for (int bla=0; bla<irrep1; bla++){ DMRGindex1 += NDMRG[bla]; }
302  for (int bla=0; bla<irrep3; bla++){ DMRGindex3 += NDMRG[bla]; }
303  DMRGindex1 += cnt1 - Nocc[irrep1];
304  DMRGindex3 += cnt3 - Nocc[irrep3];
305 
306  return 2 * DMRG1DM[DMRGindex1 + nOrbDMRG*DMRGindex3]; //2 * get1DMrotated(index1,index3);
307  }
308 
309  }
310  if (cnt4>=Nocc[irrep4]){
311 
312  if (cnt2 != cnt3){
313  return 0.0;
314  } else {
315  int DMRGindex1 = 0;
316  int DMRGindex4 = 0;
317  for (int bla=0; bla<irrep1; bla++){ DMRGindex1 += NDMRG[bla]; }
318  for (int bla=0; bla<irrep4; bla++){ DMRGindex4 += NDMRG[bla]; }
319  DMRGindex1 += cnt1 - Nocc[irrep1];
320  DMRGindex4 += cnt4 - Nocc[irrep4];
321 
322  return - DMRG1DM[DMRGindex1 + nOrbDMRG*DMRGindex4]; //- get1DMrotated(index1,index4);
323  }
324 
325  }
326 
327  }
328 
329  return 0.0; //Two occ, two active; but index1 and index2 are in the same space (occ/act) --> zero

333 double CheMPS2::CASSCF::get1DMrotated(const int index1, const int index2) const{
334 
335  if ((index1<0) || (index1>=L)){ return 0.0; }
336  if ((index2<0) || (index2>=L)){ return 0.0; } //Within bounds now.
337 
338  const int irrep = HamOrig->getOrbitalIrrep(index1);
339 
340  if (irrep != HamOrig->getOrbitalIrrep(index2)){ return 0.0; } //Of same irrep now.
341 
342  int cnt1 = index1 - jumpsHamOrig[irrep];
343  int cnt2 = index2 - jumpsHamOrig[irrep];
344 
345  if (cnt1 < Nocc[irrep]){ //If in occupied HF orbitals
346  if (cnt1==cnt2){ return 2.0; }
347  else{ return 0.0; }
348  }
349 
350  cnt1 -= Nocc[irrep];
351  cnt2 -= Nocc[irrep];
352 
353  if ((cnt1 >= 0) && (cnt1 < NDMRG[irrep]) && (cnt2 >= 0) && (cnt2 < NDMRG[irrep])){ //If both in DMRG orbitals
354  int jump = 0;
355  for (int bla=0; bla<irrep; bla++){ jump += NDMRG[bla]; }
356  int DMRGindex1 = jump + cnt1;
357  int DMRGindex2 = jump + cnt2;
358  return DMRG1DM[DMRGindex1 + nOrbDMRG*DMRGindex2];
359  }
360 
361  return 0.0; //All other cases : 0.0
362

404 void CheMPS2::CASSCF::updateUnitary(double * temp1, double * temp2){
405 
406  //Per irrep
407  for (int irrep=0; irrep<numberOfIrreps; irrep++){
408 
409  int linsize = OrbPerIrrep[irrep];
410  int size = linsize * linsize;
411 
412  if (linsize>1){ //linsize is op z'n minst 2 dus temp1, temp1+size, temp1+2*size,temp1+3*size zijn zeker ok
413 
414  //Construct the anti-symmetric x-matrix
415  double * xblock = temp1;
416  for (int cnt=0; cnt<size; cnt++){ xblock[cnt] = 0.0; }
417  for (int cntOcc=0; cntOcc<Nocc[irrep]; cntOcc++){
418  for (int cntDMRG=0; cntDMRG<NDMRG[irrep]; cntDMRG++){
419  xblock[ Nocc[irrep] + cntDMRG + linsize*cntOcc ] = xmatrix[irrep][0][ cntDMRG + NDMRG[irrep] * cntOcc];
420  xblock[ cntOcc + linsize*(Nocc[irrep] + cntDMRG) ] = - xblock[ Nocc[irrep] + cntDMRG + linsize*cntOcc ];
421  }
422  }
423  for (int cntDMRG=0; cntDMRG<NDMRG[irrep]; cntDMRG++){
424  for (int cntVirt=0; cntVirt<Nvirt[irrep]; cntVirt++){
425  xblock[ Nocc[irrep] + NDMRG[irrep] + cntVirt + linsize*(Nocc[irrep] + cntDMRG )] = xmatrix[irrep][1][ cntVirt + Nvirt[irrep] * cntDMRG ];
426  xblock[ Nocc[irrep] + cntDMRG + linsize*(Nocc[irrep] + NDMRG[irrep] + cntVirt )]
427  = - xblock[ Nocc[irrep] + NDMRG[irrep] + cntVirt + linsize*( Nocc[irrep] + cntDMRG ) ];
428  }
429  }
430  for (int cntOcc=0; cntOcc<Nocc[irrep]; cntOcc++){
431  for (int cntVirt=0; cntVirt<Nvirt[irrep]; cntVirt++){
432  xblock[ Nocc[irrep] + NDMRG[irrep] + cntVirt + linsize*cntOcc ] = xmatrix[irrep][2][ cntVirt + Nvirt[irrep] * cntOcc ];
433  xblock[ cntOcc + linsize*( Nocc[irrep] + NDMRG[irrep] + cntVirt ) ]
434  = - xblock[ Nocc[irrep] + NDMRG[irrep] + cntVirt + linsize*cntOcc ];
435  }
436  }
437 
438  //Calculate its eigenvalues and eigenvectors
439  double * Bmat = temp2;
440  char notr = 'N';
441  double alpha = 1.0;
442  double beta = 0.0; //SET !!!
443  dgemm_(&notr,&notr,&linsize,&linsize,&linsize,&alpha,xblock,&linsize,xblock,&linsize,&beta,Bmat,&linsize); //Bmat = xblock * xblock
444 
445  char uplo = 'U';
446  char jobz = 'V';
447  double * eigenval = temp1 + 2*size;
448  double * work = temp1 + size;
449  int info;
450  dsyev_(&jobz, &uplo, &linsize, Bmat, &linsize, eigenval, work, &size, &info); // xblock * xblock = Bmat * eigenval * Bmat^T
451 
452  dgemm_(&notr,&notr,&linsize,&linsize,&linsize,&alpha,xblock,&linsize,Bmat,&linsize,&beta,work,&linsize);
453  char trans = 'T';
454  double * work2 = temp2 + size;
455  dgemm_(&trans,&notr,&linsize,&linsize,&linsize,&alpha,Bmat,&linsize,work,&linsize,&beta,work2,&linsize); //work2 = Bmat^T * xblock * Bmat
456 
457  if (CheMPS2::CASSCF_debugPrint){
458  cout << "lambdas of irrep block " << irrep << " : " << endl;
459  for (int cnt=0; cnt<linsize/2; cnt++){
460  cout << " block = [ " << work2[2*cnt + linsize*2*cnt] << " , " << work2[2*cnt + linsize*(2*cnt+1)] << " ] " << endl;
461  cout << " [ " << work2[2*cnt+1 + linsize*2*cnt] << " , " << work2[2*cnt+1 + linsize*(2*cnt+1)] << " ] " << endl;
462  }
463  }
464 
465  for (int cnt=0; cnt<linsize/2; cnt++){
466  eigenval[cnt] = 0.5*( work2[2*cnt + linsize*(2*cnt+1)] - work2[2*cnt+1 + linsize*(2*cnt)] );
467  work2[2*cnt + linsize*(2*cnt+1)] -= eigenval[cnt];
468  work2[2*cnt+1 + linsize*(2*cnt)] += eigenval[cnt];
469  }
470 
471  if (CheMPS2::CASSCF_debugPrint){
472  double TwoNormResidual = 0.0;
473  for (int cnt=0; cnt<size; cnt++){ TwoNormResidual += work2[cnt] * work2[cnt]; }
474  TwoNormResidual = sqrt(TwoNormResidual);
475  cout << "TwoNormResidual of irrep block " << irrep << " = " << TwoNormResidual << endl;
476  }
477 
478  //Calculate exp(x)
479  for (int cnt=0; cnt<size; cnt++){ work2[cnt] = 0.0; }
480  for (int cnt=0; cnt<linsize/2; cnt++){
481  double cosine = cos(eigenval[cnt]);
482  double sine = sin(eigenval[cnt]);
483  work2[2*cnt + linsize*(2*cnt )] = cosine;
484  work2[2*cnt+1 + linsize*(2*cnt+1)] = cosine;
485  work2[2*cnt + linsize*(2*cnt+1)] = sine;
486  work2[2*cnt+1 + linsize*(2*cnt )] = - sine;
487  }
488  for (int cnt=2*(linsize/2); cnt<linsize; cnt++){
489  work2[cnt*(linsize + 1)] = 1.0;
490  }
491  dgemm_(&notr,&notr,&linsize,&linsize,&linsize,&alpha,Bmat,&linsize,work2,&linsize,&beta,work,&linsize);
492  dgemm_(&notr,&trans,&linsize,&linsize,&linsize,&alpha,work,&linsize,Bmat,&linsize,&beta,work2,&linsize); //work2 = exp(xblock)

650 void CheMPS2::CASSCF::setupStart(int * NoccIn, int * NDMRGIn, int * NvirtIn){
651 
652  setupStartCalled = true;
653 
654  //Copy the partitioning
655  Nocc = new int[numberOfIrreps];
656  NDMRG = new int[numberOfIrreps];
657  Nvirt = new int[numberOfIrreps];
658  for (int cnt=0; cnt<numberOfIrreps; cnt++){
659 
660  if ((NoccIn[cnt]<0) || (NDMRGIn[cnt]<0) || (NvirtIn[cnt]<0)){ cout << "One of the entries for the setup was smaller than zero." << endl; }
661  if (NoccIn[cnt] + NDMRGIn[cnt] + NvirtIn[cnt] != OrbPerIrrep[cnt]){ cout << "The sum of the partition for the setup doesn't match the number of orbitals." << endl; }
662 
663  Nocc[cnt] = NoccIn[cnt];
664  NDMRG[cnt] = NDMRGIn[cnt];
665  Nvirt[cnt] = NvirtIn[cnt];
666 
667  }
668 
669  //Number of orbitals for DMRG, their irrep, and their number in terms of the original Hamiltonian
670  nOrbDMRG = 0;
671  for (int cnt=0; cnt<numberOfIrreps; cnt++){ nOrbDMRG += NDMRG[cnt]; }
672  irrepOfEachDMRGOrbital = new int[nOrbDMRG];
673  listDMRG = new int[nOrbDMRG];
674  int passed = 0;
675  int passed2 = 0;
676  for (int cnt=0; cnt<numberOfIrreps; cnt++){
677  for (int cnt2=0; cnt2<NDMRG[cnt]; cnt2++){
678  irrepOfEachDMRGOrbital[passed+cnt2] = cnt;
679  listDMRG[passed+cnt2] = passed2 + Nocc[cnt] + cnt2;
680  }
681  passed += NDMRG[cnt];
682  passed2 += OrbPerIrrep[cnt];
683  }
684 
685  //How many and which orbitals are condensed now.
686  nCondensed = 0;
687  for (int cnt=0; cnt<numberOfIrreps; cnt++){ nCondensed += Nocc[cnt]; }

689  passed = 0;
690  int irrep = 0;
691  int counter = 0;
692  for (int cnt=0; cnt<L; cnt++){
693  if (cnt-passed < Nocc[irrep]){
694  listCondensed[counter] = cnt;
695  counter++;
696  }

711  if (case_==1){ size = NDMRG[irrep] * Nvirt[irrep]; }

722  int size = OrbPerIrrep[irrep] * OrbPerIrrep[irrep];

724  for (int cnt=0; cnt<size; cnt++){ unitary[irrep][cnt] = 0.0; }
