72 void CheMPS2::Correlations::FillOneRDMSpinDensSpinflip(){
73 
74  //OneRDM
75  const double prefactor = 1.0 / ( Prob->gN() - 1.0 );
76  for (int row=0; row<L; row++){
77  for (int col=row; col<L; col++){
78  OneRDM[row + L*col] = 0.0;
79  for (int third=0; third<L; third++){ OneRDM[row + L*col] += the2DM->getTwoDMA_DMRG(row,third,col,third); }
80  OneRDM[row + L*col] *= prefactor;
81  OneRDM[col + L*row] = OneRDM[row + L*col];
82  }
83  }
84 
85  //Spin
86  for (int row=0; row<L; row++){
87  for (int col=0; col<L; col++){
88  Cspin[row + L*col] = the2DM->getTwoDMB_DMRG(row,col,row,col);
89  }
90  Cspin[row + L*row] += OneRDM[row + L*row];
91  }
92 
93  //Density
94  for (int row=0; row<L; row++){
95  for (int col=0; col<L; col++){
96  Cdens[row + L*col] = the2DM->getTwoDMA_DMRG(row,col,row,col) - OneRDM[row + L*row] * OneRDM[col + L*col];
97  }
98  Cdens[row + L*row] += OneRDM[row + L*row];
99  }
100 
101  //Spin-flip
102  for (int row=0; row<L; row++){
103  for (int col=0; col<L; col++){
104  Cspinflip[row + L*col] = 0.5 * ( the2DM->getTwoDMB_DMRG(row,col,col,row) - the2DM->getTwoDMA_DMRG(row,col,col,row) );
105  }
106  Cspinflip[row + L*row] += OneRDM[row + L*row];
107  }
108 
109  //Singlet diradical: partial fill
110  for (int row=0; row<L; row++){
111  for (int col=0; col<L; col++){
112  Cdirad[row + L*col] = - 0.5 * ( OneRDM[row + L*row] - the2DM->getTwoDMA_DMRG(row,row,row,row) )
113  * ( OneRDM[col + L*col] - the2DM->getTwoDMA_DMRG(col,col,col,col) );
114  }
115  }
116 
117 }

179 double CheMPS2::Correlations::SingleOrbitalEntropy_DMRG(const int index) const{
180 
181  const double val4 = 0.5 * the2DM->getTwoDMA_DMRG(index,index,index,index);
182  const double val23 = 0.5 * ( OneRDM[index + L*index] - the2DM->getTwoDMA_DMRG(index,index,index,index) );
183  const double val1 = 1.0 - val4 - 2*val23;
184 
185  if (val1 < 0.0){ cerr << " Correlations::SingleOrbitalEntropy : Warning : val1 for orbital " << index << " = " << val1 << endl; }
186  if (val23 < 0.0){ cerr << " Correlations::SingleOrbitalEntropy : Warning : val23 for orbital " << index << " = " << val23 << endl; }
187  if (val4 < 0.0){ cerr << " Correlations::SingleOrbitalEntropy : Warning : val4 for orbital " << index << " = " << val4 << endl; }
188 
189  double entropy = 0.0;
190  if (val1 > CheMPS2::CORRELATIONS_discardEig){ entropy -= val1 * log(val1 ); }
191  if (val23 > CheMPS2::CORRELATIONS_discardEig){ entropy -= 2 * val23 * log(val23); }
192  if (val4 > CheMPS2::CORRELATIONS_discardEig){ entropy -= val4 * log(val4 ); }
193  return entropy;
194 
195 }

220 void CheMPS2::Correlations::FillSite(TensorT * denT, TensorGYZ ** Gtensors, TensorGYZ ** Ytensors, TensorGYZ ** Ztensors, TensorK ** Ktensors, TensorM ** Mtensors){
221 
222  const int theindex = denT->gIndex();
223  const int MAXDIM = max(denBK->gMaxDimAtBound(theindex), denBK->gMaxDimAtBound(theindex+1));
224  double * workmem = new double[MAXDIM*MAXDIM];
225  int lindimRDM = 16;
226  double * RDM = new double[lindimRDM*lindimRDM];
227  int lwork = 3*lindimRDM - 1;
228  double * work = new double[lwork];
229  double * eigs = new double[lindimRDM];
230 
231  const double prefactorSpin = 1.0/(Prob->gTwoS() + 1.0);
232  const double sqrt_one_half = sqrt(0.5);
233 
234  for (int previousindex=0; previousindex<theindex; previousindex++){
235 
236  const bool equalIrreps = (denBK->gIrrep(previousindex) == denBK->gIrrep(theindex)) ? true : false;
237  const double diag1 = diagram3(denT, Gtensors[previousindex], workmem) * prefactorSpin * 0.5 * sqrt_one_half;
238  const double diag2 = 0.125 * ( the2DM->getTwoDMB_DMRG(previousindex,theindex,theindex,previousindex)
239  - the2DM->getTwoDMA_DMRG(previousindex,theindex,theindex,previousindex) );
240 
241  const double val1 = diagram1(denT, Ytensors[previousindex], workmem) * prefactorSpin; //1x1 block N=0, Sz=0
242  const double val2 = diagram2(denT, Ztensors[previousindex], workmem) * prefactorSpin; //1x1 block N=4, Sz=0
243  const double val3 = diag1 + diag2; //1x1 block N=2, Sz=2*sigma
244  const double val4 = diagram1(denT, Gtensors[previousindex], workmem) * prefactorSpin * sqrt_one_half; //2x2 block N=1, alpha_LL
245  const double val5 = diagram3(denT, Ytensors[previousindex], workmem) * prefactorSpin * 0.5; //2x2 block N=1, alpha_RR
246  const double val6 = (equalIrreps) ? ( diagram4(denT, Ktensors[previousindex], workmem) * prefactorSpin * 0.5 ) : 0.0 ; //2x2 block N=1, alpha_LR
247  const double val7 = diagram2(denT, Gtensors[previousindex], workmem) * prefactorSpin * sqrt_one_half; //2x2 block N=3, alpha_LL
248  const double val8 = diagram3(denT, Ztensors[previousindex], workmem) * prefactorSpin * 0.5; //2x2 block N=3, alpha_RR
249  const double val9 = (equalIrreps) ? ( diagram5(denT, Mtensors[previousindex], workmem) * prefactorSpin * 0.5 ) : 0.0 ; //2x2 block N=3, alpha_LR
250 
251  //4x4 block N=2, Sz=0
252  const double alpha = diagram2(denT, Ytensors[previousindex], workmem) * prefactorSpin;
253  const double gamma = diagram1(denT, Ztensors[previousindex], workmem) * prefactorSpin;
254  const double beta = diag1 - diag2;
255  const double lambda = 2*diag2;
256  const double delta = (equalIrreps) ? ( - diagram5(denT, Ktensors[previousindex], workmem) * prefactorSpin * 0.5 ) : 0.0;
257  const double epsilon = (equalIrreps) ? ( diagram4(denT, Mtensors[previousindex], workmem) * prefactorSpin * 0.5 ) : 0.0;
258  const double kappa = 0.5 * the2DM->getTwoDMA_DMRG(previousindex,previousindex,theindex,theindex);
259 
260  /*
261 
262  [ val1 ]
263  [ val4 val6 ]
264  [ val6 val5 ]
265  [ val4 val6 ]
266  [ val6 val5 ]
267  [ val3 ]
268  [ alpha delta -delta kappa ]
269  [ delta beta lambda epsilon ]
270  [ -delta lambda beta -epsilon ]
271  [ kappa epsilon -epsilon gamma ]
272  [ val3 ]
273  [ val7 val9 ]
274  [ val9 val8 ]
275  [ val7 val9 ]
276  [ val9 val8 ]
277  [ val2 ]
278 
279  */
280 
281  for (int cnt=0; cnt<lindimRDM*lindimRDM; cnt++){ RDM[cnt] = 0.0; }
282  RDM[0 + lindimRDM * 0 ] = val1;
283  RDM[15 + lindimRDM * 15] = val2;
284  RDM[5 + lindimRDM * 5 ] = RDM[10 + lindimRDM * 10] = val3;
285  RDM[1 + lindimRDM * 1 ] = RDM[3 + lindimRDM * 3 ] = val4;
286  RDM[2 + lindimRDM * 2 ] = RDM[4 + lindimRDM * 4 ] = val5;
287  RDM[1 + lindimRDM * 2 ] = RDM[2 + lindimRDM * 1 ] = RDM[3 + lindimRDM * 4 ] = RDM[4 + lindimRDM * 3 ] = val6;
288  RDM[11 + lindimRDM * 11] = RDM[13 + lindimRDM * 13] = val7;
289  RDM[12 + lindimRDM * 12] = RDM[14 + lindimRDM * 14] = val8;
290  RDM[11 + lindimRDM * 12] = RDM[12 + lindimRDM * 11] = RDM[13 + lindimRDM*14] = RDM[14 + lindimRDM*13] = val9;
291  RDM[6 + lindimRDM * 6 ] = alpha;
292  RDM[7 + lindimRDM * 7 ] = RDM[8 + lindimRDM * 8 ] = beta;
293  RDM[9 + lindimRDM * 9 ] = gamma;
294  RDM[6 + lindimRDM * 7 ] = RDM[7 + lindimRDM * 6 ] = delta;
295  RDM[6 + lindimRDM * 8 ] = RDM[8 + lindimRDM * 6 ] = - delta;
296  RDM[7 + lindimRDM * 9 ] = RDM[9 + lindimRDM * 7 ] = epsilon;
297  RDM[8 + lindimRDM * 9 ] = RDM[9 + lindimRDM * 8 ] = - epsilon;
298  RDM[6 + lindimRDM * 9 ] = RDM[9 + lindimRDM * 6 ] = kappa;
299  RDM[7 + lindimRDM * 8 ] = RDM[8 + lindimRDM * 7 ] = lambda;
300 
301  if (CheMPS2::CORRELATIONS_debugPrint){
302  const double RDM_1orb_prev_4 = 0.5 * the2DM->getTwoDMA_DMRG(previousindex,previousindex,previousindex,previousindex);
303  const double RDM_1orb_prev_23 = 0.5 * ( OneRDM[previousindex + L*previousindex]
304  - the2DM->getTwoDMA_DMRG(previousindex,previousindex,previousindex,previousindex) );
305  const double RDM_1orb_prev_1 = 1.0 - RDM_1orb_prev_4 - 2*RDM_1orb_prev_23;
306 
307  const double RDM_1orb_curr_4 = 0.5 * the2DM->getTwoDMA_DMRG(theindex,theindex,theindex,theindex);
308  const double RDM_1orb_curr_23 = 0.5 * ( OneRDM[theindex + L*theindex] - the2DM->getTwoDMA_DMRG(theindex,theindex,theindex,theindex) );
309  const double RDM_1orb_curr_1 = 1.0 - RDM_1orb_curr_4 - 2*RDM_1orb_curr_23;
310 
311  cout << " Correlations::FillSite : Looking at DMRG sites (" << previousindex << "," << theindex << ")." << endl;
312 
313  //Check 1 : full trace
314  double fulltrace = 0.0;
315  for (int cnt=0; cnt<lindimRDM; cnt++){ fulltrace += RDM[ cnt * ( 1 + lindimRDM ) ]; }
316  cout << " 1 - trace(2-orb RDM) = " << 1.0 - fulltrace << endl;
317 
318  //Check 2 : trace over orb previousindex
319  double getal1 = RDM[0 + lindimRDM * 0 ] + RDM[1 + lindimRDM * 1 ] + RDM[3 + lindimRDM * 3 ] + RDM[9 + lindimRDM * 9 ] - RDM_1orb_curr_1;
320  double getal2 = RDM[2 + lindimRDM * 2 ] + RDM[5 + lindimRDM * 5 ] + RDM[8 + lindimRDM * 8 ] + RDM[12 + lindimRDM * 12] - RDM_1orb_curr_23;
321  double getal3 = RDM[4 + lindimRDM * 4 ] + RDM[7 + lindimRDM * 7 ] + RDM[10 + lindimRDM * 10] + RDM[14 + lindimRDM * 14] - RDM_1orb_curr_23;
322  double getal4 = RDM[6 + lindimRDM * 6 ] + RDM[11 + lindimRDM * 11] + RDM[13 + lindimRDM * 13] + RDM[15 + lindimRDM * 15] - RDM_1orb_curr_4;
323  double RMS = sqrt(getal1*getal1 + getal2*getal2 + getal3*getal3 + getal4*getal4);
324  cout << " 2-norm difference of one-orb RDM of the CURRENT site via 2DM and via trace(2-orb RDM) = " << RMS << endl;
325 
326  //Check 3 : trace over orb currentindex
327  getal1 = RDM[0 + lindimRDM * 0 ] + RDM[2 + lindimRDM * 2 ] + RDM[4 + lindimRDM * 4 ] + RDM[6 + lindimRDM * 6 ] - RDM_1orb_prev_1;
328  getal2 = RDM[1 + lindimRDM * 1 ] + RDM[5 + lindimRDM * 5 ] + RDM[7 + lindimRDM * 7 ] + RDM[11 + lindimRDM * 11] - RDM_1orb_prev_23;
329  getal3 = RDM[3 + lindimRDM * 3 ] + RDM[8 + lindimRDM * 8 ] + RDM[10 + lindimRDM * 10] + RDM[13 + lindimRDM * 13] - RDM_1orb_prev_23;
330  getal4 = RDM[9 + lindimRDM * 9 ] + RDM[12 + lindimRDM * 12] + RDM[14 + lindimRDM * 14] + RDM[15 + lindimRDM * 15] - RDM_1orb_prev_4;
331  RMS = sqrt(getal1*getal1 + getal2*getal2 + getal3*getal3 + getal4*getal4);
332  cout << " 2-norm difference of one-orb RDM of the PREVIOUS site via 2DM and via trace(2-orb RDM) = " << RMS << endl;
333  }
334 
335  char jobz = 'N'; //eigenvalues only
336  char uplo = 'U';
337  int info;
338  dsyev_(&jobz, &uplo, &lindimRDM, RDM, &lindimRDM, eigs, work, &lwork, &info);
339 
340  double entropy = 0.0;
341  for (int cnt=0; cnt<lindimRDM; cnt++){
342  if (eigs[cnt] > CheMPS2::CORRELATIONS_discardEig){ //With discardEig = 1e-100, the discarded contributions are smaller than 2.4e-98
343  entropy -= eigs[cnt] * log(eigs[cnt]);
344  }
345  if (eigs[cnt] < 0.0){
346  cerr << " Correlations::FillSite : Warning : Eigenvalue RDM orbitals (" << previousindex << "," << theindex << ") = " << eigs[cnt] << endl;
347  }
348  }
349 
350  const double thisMutInfo = 0.5 * ( SingleOrbitalEntropy_DMRG(previousindex) + SingleOrbitalEntropy_DMRG(theindex) - entropy );
351  MutInfo[previousindex + L * theindex] = MutInfo[theindex + L * previousindex] = thisMutInfo;
352  Cdirad[previousindex + L * theindex] += 2 * beta;
353  Cdirad[theindex + L * previousindex] += 2 * beta;
354 
355  }
356 
357  delete [] eigs;
358  delete [] work;
359  delete [] RDM;
360  delete [] workmem;
361 
362 }

386

388  int inc = 1;
389  total += (TwoSR + 1.0) * ddot_(&length, workmem, &inc, blockT, &inc);

391  }

393  }
394  }

436 double CheMPS2::Correlations::diagram3(TensorT * denT, TensorGYZ * denG, double * workmem) const{ //checked
