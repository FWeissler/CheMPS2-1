33 CheMPS2::SyBookkeeper::SyBookkeeper(const Problem * Probin, const int Din) : Irreps(Probin->gSy()){
34 
35  Prob = Probin;
36 
37  //Set the min and max particle number and spin
38  Nmin = new int[gL()+1];
39  Nmax = new int[gL()+1];
40  TwoSmin = new int*[gL()+1];
41  TwoSmax = new int*[gL()+1];
42  for (int bound=0; bound<=gL(); bound++){
43  Nmin[bound] = max( max(0, gN()+2*(bound - gL())) , bound - gL() + (gN() + gTwoS())/2);
44  Nmax[bound] = min( min(2*bound, gN() ), bound + (gN() - gTwoS())/2);
45  TwoSmin[bound] = new int[Nmax[bound]-Nmin[bound]+1];
46  TwoSmax[bound] = new int[Nmax[bound]-Nmin[bound]+1];
47  for (int N=Nmin[bound]; N<=Nmax[bound]; N++){
48  TwoSmin[bound][N-Nmin[bound]] = max(N%2, gTwoS() - (gL() - bound - abs(gN() - N - gL() + bound)));
49  TwoSmax[bound][N-Nmin[bound]] = min( bound - abs(bound - N), gTwoS() + (gL() - bound - abs(gN() - N - gL() + bound)));
50  }
51  }
52 
53  //FCIdim & CurrentDim memory allocation
54  FCIdim = new int***[gL()+1];
55  CurrentDim = new int***[gL()+1];
56  for (int bound=0; bound<=gL(); bound++){
57  FCIdim[bound] = new int**[gNmax(bound)-gNmin(bound)+1];
58  CurrentDim[bound] = new int**[gNmax(bound)-gNmin(bound)+1];
59  for (int N=gNmin(bound); N<=gNmax(bound); N++){
60  FCIdim[bound][N-gNmin(bound)] = new int*[(gTwoSmax(bound,N)-gTwoSmin(bound,N))/2+1];
61  CurrentDim[bound][N-gNmin(bound)] = new int*[(gTwoSmax(bound,N)-gTwoSmin(bound,N))/2+1];
62  for (int TwoS=gTwoSmin(bound,N); TwoS<=gTwoSmax(bound,N); TwoS+=2){
63  FCIdim[bound][N-gNmin(bound)][(TwoS-gTwoSmin(bound,N))/2] = new int[getNumberOfIrreps()];
64  CurrentDim[bound][N-gNmin(bound)][(TwoS-gTwoSmin(bound,N))/2] = new int[getNumberOfIrreps()];
65  }
66  }
67  }
68 
69  //Fill the FCI dims & copy it to the Current dims
70  fillFCIdim();
71 
72  //Scale the Current dims
73  ScaleCurrentDim(Din);
74 
75  if (CheMPS2::SYBK_debugPrint) print();
76  assert( IsPossible() );
77 
78 }

138 void CheMPS2::SyBookkeeper::fillFCIdim(){
139 
140  //First fill FCIdim from left
141  for (int Icnt=0; Icnt<getNumberOfIrreps(); Icnt++) FCIdim[0][0][0][Icnt] = 0;
142  FCIdim[0][0][0][0] = 1;
143 
144  for (int bound=1; bound<=gL(); bound++){
145  for (int N=gNmin(bound); N<=gNmax(bound); N++){
146  for (int TwoS=gTwoSmin(bound,N); TwoS<=gTwoSmax(bound,N); TwoS+=2){
147  for (int Icnt=0; Icnt<getNumberOfIrreps(); Icnt++){
148  FCIdim[bound][N-gNmin(bound)][(TwoS-gTwoSmin(bound,N))/2][Icnt] = min( CheMPS2::SYBK_dimensionCutoff, gFCIdim(bound-1,N,TwoS,Icnt) + gFCIdim(bound-1,N-2,TwoS,Icnt) + gFCIdim(bound-1,N-1,TwoS+1,directProd(Icnt,gIrrep(bound-1))) + gFCIdim(bound-1,N-1,TwoS-1,directProd(Icnt,gIrrep(bound-1))) );
149  }
150  }
151  }
152  }
153 
154  //Then allocate FCIdimRight
155  int **** FCIdimRight = new int***[gL()+1];
156  for (int bound=0; bound<=gL(); bound++){
157  FCIdimRight[bound] = new int**[gNmax(bound)-gNmin(bound)+1];
158  for (int N=gNmin(bound); N<=gNmax(bound); N++){
159  FCIdimRight[bound][N-gNmin(bound)] = new int*[(gTwoSmax(bound,N)-gTwoSmin(bound,N))/2+1];
160  for (int TwoS=gTwoSmin(bound,N); TwoS<=gTwoSmax(bound,N); TwoS+=2){
161  FCIdimRight[bound][N-gNmin(bound)][(TwoS-gTwoSmin(bound,N))/2] = new int[getNumberOfIrreps()];
162  }
163  }
164  }
165 
166  //Then calculate FCI dim from right -->FCIdimRight
167  for (int Icnt=0; Icnt<getNumberOfIrreps(); Icnt++) FCIdimRight[gL()][0][0][Icnt] = 0;
168  FCIdimRight[gL()][0][0][gIrrep()] = 1;
169 
170  for (int bound=gL()-1; bound>=0; bound--){
171  for (int N=gNmin(bound); N<=gNmax(bound); N++){
172  for (int TwoS=gTwoSmin(bound,N); TwoS<=gTwoSmax(bound,N); TwoS+=2){
173  for (int Icnt=0; Icnt<getNumberOfIrreps(); Icnt++){
174  FCIdimRight[bound][N-gNmin(bound)][(TwoS-gTwoSmin(bound,N))/2][Icnt] = min( CheMPS2::SYBK_dimensionCutoff, gDimPrivate(FCIdimRight,bound+1,N,TwoS,Icnt) + gDimPrivate(FCIdimRight,bound+1,N+2,TwoS,Icnt) + gDimPrivate(FCIdimRight,bound+1,N+1,TwoS+1,directProd(Icnt,gIrrep(bound))) + gDimPrivate(FCIdimRight,bound+1,N+1,TwoS-1,directProd(Icnt,gIrrep(bound))) );
175  }
176  }
177  }
178  }
179 
180  //Then take min from FCIdim and FCIdimRight and store in FCIdim; and copy it to CurrentDim
181  for (int bound=0; bound<=gL(); bound++){
182  for (int N=gNmin(bound); N<=gNmax(bound); N++){
183  for (int TwoS=gTwoSmin(bound,N); TwoS<=gTwoSmax(bound,N); TwoS+=2){
184  for (int Icnt=0; Icnt<getNumberOfIrreps(); Icnt++){
185  FCIdim[bound][N-gNmin(bound)][(TwoS-gTwoSmin(bound,N))/2][Icnt] = min( gFCIdim(bound,N,TwoS,Icnt), gDimPrivate(FCIdimRight,bound,N,TwoS,Icnt) );
186  CurrentDim[bound][N-gNmin(bound)][(TwoS-gTwoSmin(bound,N))/2][Icnt] = gFCIdim(bound,N,TwoS,Icnt);
187  }
188  }
189  }
190  }
191 
192  //Deallocate FCIdimRight
193  for (int bound=0; bound<=gL(); bound++){
194  for (int N=gNmin(bound); N<=gNmax(bound); N++){
195  for (int TwoS=gTwoSmin(bound,N); TwoS<=gTwoSmax(bound,N); TwoS+=2){
196  delete [] FCIdimRight[bound][N-gNmin(bound)][(TwoS-gTwoSmin(bound,N))/2];
197  }
198  delete [] FCIdimRight[bound][N-gNmin(bound)];
199  }
200  delete [] FCIdimRight[bound];
201  }
202  delete [] FCIdimRight;
203

206 void CheMPS2::SyBookkeeper::ScaleCurrentDim(const int virtualD){
207 
208  for (int bound=1; bound<gL(); bound++){
209  int totaldim = 0;
210  for (int N=gNmin(bound); N<=gNmax(bound); N++){
211  for (int TwoS=gTwoSmin(bound,N); TwoS<=gTwoSmax(bound,N); TwoS+=2){
212  for (int Icnt=0; Icnt<getNumberOfIrreps(); Icnt++){
213  totaldim += gCurrentDim(bound,N,TwoS,Icnt);
214  }
215  }
216  }
217 
218  if (totaldim > virtualD){
219  double factor = (1.0 * virtualD) / totaldim;
220  for (int N=gNmin(bound); N<=gNmax(bound); N++){
221  for (int TwoS=gTwoSmin(bound,N); TwoS<=gTwoSmax(bound,N); TwoS+=2){
222  for (int Icnt=0; Icnt<getNumberOfIrreps(); Icnt++){
223  CurrentDim[bound][N-gNmin(bound)][(TwoS-gTwoSmin(bound,N))/2][Icnt] = (ceil( factor * gCurrentDim(bound,N,TwoS,Icnt)) + 0.1);
224  }
225  }
226  }
227  }
228 
229  if (CheMPS2::SYBK_debugPrint){
230  cout << "Bound = " << bound << endl;
231  cout << " Totaldim (FCI) = " << totaldim << endl;
232  totaldim = 0;
233  for (int N=gNmin(bound); N<=gNmax(bound); N++){
234  for (int TwoS=gTwoSmin(bound,N); TwoS<=gTwoSmax(bound,N); TwoS+=2){
235  for (int Icnt=0; Icnt<getNumberOfIrreps(); Icnt++){
236  totaldim += gCurrentDim(bound,N,TwoS,Icnt);
237  }
238  }
239  }
240  cout << " Totaldim (rescaled) = " << totaldim << endl;
241  }
242  }
243 
244  if (CheMPS2::SYBK_debugPrint){
245  for (int bound=0; bound<=gL(); bound++){
246  for (int N=gNmin(bound); N<=gNmax(bound); N++){
247  for (int TwoS=gTwoSmin(bound,N); TwoS<=gTwoSmax(bound,N); TwoS+=2){
248  for (int Icnt=0; Icnt<getNumberOfIrreps(); Icnt++){
249  if ((gFCIdim(bound,N,TwoS,Icnt)!=0) && (gCurrentDim(bound,N,TwoS,Icnt)==0)){
250  cout << "Error at print: " << endl;
251  cout << " gFCIdim(" << bound << "," << N << "," << TwoS << "," << Icnt << ") = " << gFCIdim(bound,N,TwoS,Icnt) << endl;
252  cout << " gCurrentDim(" << bound << "," << N << "," << TwoS << "," << Icnt << ") = " << gCurrentDim(bound,N,TwoS,Icnt) << endl;
253  }
254  }
255  }
256  }
257  }

262 int CheMPS2::SyBookkeeper::gDimPrivate(int **** storage, const int bound, const int N, const int TwoS, const int Icnt) const{
263 
264  if ((bound<0) || (bound>gL())) return 0;
265  if ((N>gNmax(bound)) || (N<gNmin(bound))) return 0;
266  if ((TwoS%2) != (gTwoSmin(bound,N)%2)) return 0;
267  if ((TwoS < gTwoSmin(bound,N)) || (TwoS > gTwoSmax(bound,N))) return 0;
268  if ((Icnt<0) || (Icnt>getNumberOfIrreps())) return 0;
269  return storage[bound][N-gNmin(bound)][(TwoS-gTwoSmin(bound,N))/2][Icnt];
270
