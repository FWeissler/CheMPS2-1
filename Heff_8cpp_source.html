214 double CheMPS2::Heff::SolveDAVIDSON(Sobject * denS, TensorL *** Ltensors, TensorA **** Atensors, TensorB **** Btensors, TensorC **** Ctensors, TensorD **** Dtensors, TensorS0 **** S0tensors, TensorS1 **** S1tensors, TensorF0 **** F0tensors, TensorF1 **** F1tensors, TensorQ *** Qtensors, TensorX ** Xtensors, int nLower, double ** VeffTilde) const{
215 
216  //From people.inf.ethz.ch/arbenz/ewp/Lnotes/chapter11.pdf : algorithm 11.1, with instead of line (16), equation (11.3).
217  const int DAVIDSON_NUM_VEC = CheMPS2::HEFF_DAVIDSON_NUM_VEC;
218  const int DAVIDSON_NUM_VEC_KEEP = CheMPS2::HEFF_DAVIDSON_NUM_VEC_KEEP;
219 
220  //Convert mem of Sobject to symmetric conventions
221  denS->prog2symm();
222 
223  int length_vec = denS->gKappa2index(denS->gNKappa());
224  int num_vec = 0;
225  double ** vecs = new double*[DAVIDSON_NUM_VEC];
226  double ** Hvecs = new double*[DAVIDSON_NUM_VEC];
227  int num_allocated = 0;
228 
229  double * mxM = new double[DAVIDSON_NUM_VEC * DAVIDSON_NUM_VEC];
230  double * mxM_eigs = new double[DAVIDSON_NUM_VEC];
231  double * mxM_vecs = new double[DAVIDSON_NUM_VEC * DAVIDSON_NUM_VEC];
232  int mxM_lwork = 3*DAVIDSON_NUM_VEC-1;
233  double * mxM_work = new double[mxM_lwork];
234 
235  double rtol = CheMPS2::HEFF_DAVIDSON_RTOL_BASE * sqrt(length_vec);
236  double rnorm = 10*rtol;
237 
238  double * t_vec = new double[length_vec];
239  double * u_vec = new double[length_vec];
240  double * work_vec = new double[length_vec];
241  int inc1 = 1;
242  dcopy_(&length_vec,denS->gStorage(),&inc1,t_vec,&inc1); //starting vector for Davidson is the current state of the Sobject in symmetric conventions.
243 
244  //Checking whether the S-object contains anything
245  double Sobjectnorm = 0.0;
246  for (int cnt=0; cnt<length_vec; cnt++){ Sobjectnorm += t_vec[cnt]*t_vec[cnt]; }
247  Sobjectnorm = sqrt(Sobjectnorm);
248  if (Sobjectnorm==0.0){
249  for (int cnt=0; cnt<length_vec; cnt++){ t_vec[cnt] = ((double) rand())/RAND_MAX; }
250  if (CheMPS2::HEFF_debugPrint){
251  cout << "WARNING AT HEFF : S-object with zero norm was replaced with random vector." << endl;
252  cout << " Note that DMRG::checkSobject() allows to validate the S-object." << endl;
253  cout<< " This should be a rare (but possible) warning." << endl;
254  }
255  }
256  //End checking whether the S-object contains anything
257 
258  double * HeffDiag = new double[length_vec];
259  fillHeffDiag(HeffDiag, denS, Ctensors, Dtensors, F0tensors, F1tensors, Xtensors, nLower, VeffTilde);
260 
261  double * Reortho_Lowdin = NULL;
262  double * Reortho_Overlap_eigs = NULL;
263  double * Reortho_Overlap = NULL;
264  double * Reortho_Eigenvecs = NULL;
265  bool Reortho_Allocated = false;
266 
267  int nIterations = 0;
268 
269  while (rnorm > rtol){
270 
271  //1. Orthogonalize the new t_vec w.r.t. the old basis
272  for (int cnt=0; cnt<num_vec; cnt++){
273  double min_overlap = - ddot_(&length_vec,t_vec,&inc1,vecs[cnt],&inc1);
274  daxpy_(&length_vec,&min_overlap,vecs[cnt],&inc1,t_vec,&inc1);
275  }
276 
277  //2. Normalize the t_vec
278  char norm = 'F';
279  double alpha = 1.0/dlange_(&norm,&length_vec,&inc1,t_vec,&length_vec,t_vec); //work not referenced as Frobenius norm
280  dscal_(&length_vec,&alpha,t_vec,&inc1);
281 
282  //3. T_vec becomes part of vecs
283  if (num_vec<num_allocated){
284  double * temp = vecs[num_vec];
285  vecs[num_vec] = t_vec;
286  t_vec = temp;
287  } else {
288  vecs[num_allocated] = t_vec;
289  Hvecs[num_allocated] = new double[length_vec];
290  t_vec = new double[length_vec];
291  num_allocated++;
292  }
293  makeHeff(vecs[num_vec], Hvecs[num_vec], denS, Ltensors, Atensors, Btensors, Ctensors, Dtensors, S0tensors, S1tensors, F0tensors, F1tensors, Qtensors, Xtensors, nLower, VeffTilde);
294  nIterations++;
295 
296  //4. mxM contains the Hamiltonian in the basis "vecs"
297  for (int cnt=0; cnt<num_vec; cnt++){
298  mxM[cnt + DAVIDSON_NUM_VEC * num_vec] = ddot_(&length_vec,vecs[num_vec],&inc1,Hvecs[cnt],&inc1);
299  mxM[num_vec + DAVIDSON_NUM_VEC * cnt] = mxM[cnt + DAVIDSON_NUM_VEC * num_vec];
300  }
301  mxM[num_vec + DAVIDSON_NUM_VEC * num_vec] = ddot_(&length_vec,vecs[num_vec],&inc1,Hvecs[num_vec],&inc1);
302 
303  //5. When t-vec was added to vecs, the number of vecs was actually increased by one. For convenience (doing 4.), only now the number is incremented.
304  num_vec++;
305 
306  //6. Calculate the eigenvalues and vectors of mxM
307  char jobz = 'V';
308  char uplo = 'U';
309  int info;
310  for (int cnt1=0; cnt1<num_vec; cnt1++){
311  for (int cnt2=0; cnt2<num_vec; cnt2++){
312  mxM_vecs[cnt1 + DAVIDSON_NUM_VEC * cnt2] = mxM[cnt1 + DAVIDSON_NUM_VEC * cnt2];
313  }
314  }
315  int lda = DAVIDSON_NUM_VEC;
316  dsyev_(&jobz,&uplo,&num_vec,mxM_vecs,&lda,mxM_eigs,mxM_work,&mxM_lwork,&info); //ascending order of eigs
317 
318  //7. Calculate u and r. r is stored in t_vec, u in u_vec.
319  for (int cnt=0; cnt<length_vec; cnt++){
320  t_vec[cnt] = 0.0;
321  u_vec[cnt] = 0.0;
322  }
323  for (int cnt=0; cnt<num_vec; cnt++){
324  double alpha = mxM_vecs[cnt]; //eigenvector with lowest eigenvalue, hence mxM_vecs[cnt + DAVIDSON_NUM_VEC * 0]
325  daxpy_(&length_vec,&alpha,Hvecs[cnt],&inc1,t_vec,&inc1);
326  daxpy_(&length_vec,&alpha, vecs[cnt],&inc1,u_vec,&inc1);
327  }
328  alpha = -mxM_eigs[0];
329  daxpy_(&length_vec,&alpha,u_vec,&inc1,t_vec,&inc1);
330 
331  //8. Calculate the norm of r
332  rnorm = dlange_(&norm,&length_vec,&inc1,t_vec,&length_vec,t_vec);
333 
334  //9. In case convergence is not yet reached: prepare for the following iteration
335  if (rnorm > rtol){
336 
337  //9a. Calculate the new t_vec based on the residual of the lowest eigenvalue, to add to the vecs.
338  for (int cnt=0; cnt<length_vec; cnt++){
339  if (fabs(HeffDiag[cnt] - mxM_eigs[0])> CheMPS2::HEFF_DAVIDSON_PRECOND_CUTOFF ){
340  work_vec[cnt] = u_vec[cnt]/(HeffDiag[cnt] - mxM_eigs[0]); // work_vec = K^(-1) u_vec
341  } else {
342  work_vec[cnt] = u_vec[cnt]/ CheMPS2::HEFF_DAVIDSON_PRECOND_CUTOFF ;
343  if (CheMPS2::HEFF_debugPrint) cout << "|(HeffDiag[" << cnt << "] - mxM_eigs[0])| = " << fabs(HeffDiag[cnt] - mxM_eigs[0]) << endl;
344  }
345  }
346  alpha = - ddot_(&length_vec,work_vec,&inc1,t_vec,&inc1)/ddot_(&length_vec,work_vec,&inc1,u_vec,&inc1); // alpha = - (u^T K^(-1) r) / (u^T K^(-1) u)
347  daxpy_(&length_vec,&alpha,u_vec,&inc1,t_vec,&inc1); // t_vec = r - (u^T K^(-1) r) / (u^T K^(-1) u) u
348  for (int cnt=0; cnt<length_vec; cnt++){
349  if (fabs(HeffDiag[cnt] - mxM_eigs[0])> CheMPS2::HEFF_DAVIDSON_PRECOND_CUTOFF ){
350  t_vec[cnt] = - t_vec[cnt]/(HeffDiag[cnt] - mxM_eigs[0]); //t_vec = - K^(-1) (r - (u^T K^(-1) r) / (u^T K^(-1) u) u)
351  } else {
352  t_vec[cnt] = - t_vec[cnt]/ CheMPS2::HEFF_DAVIDSON_PRECOND_CUTOFF ;
353  }
354  }
355 
356  // 9b. When the maximum number of vectors is reached: construct the one with lowest eigenvalue & restart
357  if (num_vec == DAVIDSON_NUM_VEC){
358 
359  if (DAVIDSON_NUM_VEC_KEEP<=1){
360 
361  alpha = 1.0/dlange_(&norm,&length_vec,&inc1,u_vec,&length_vec,u_vec); //work not referenced as Frobenius norm
362  dscal_(&length_vec,&alpha,u_vec,&inc1);
363  dcopy_(&length_vec,u_vec,&inc1,vecs[0],&inc1);
364  makeHeff(vecs[0], Hvecs[0], denS, Ltensors, Atensors, Btensors, Ctensors, Dtensors, S0tensors, S1tensors, F0tensors, F1tensors, Qtensors, Xtensors, nLower, VeffTilde);
365  nIterations++;
