53 
54  if (movingRight){

161 void CheMPS2::TensorX::addTermQLRight(const int ikappa, TensorT * denT, TensorL ** Lprev, TensorQ * Qprev, double * workmemRR, double * workmemLR, double * workmemLL){

164  int dimTot = dimR * dimR;
165  for (int cnt=0; cnt<dimTot; cnt++){ workmemRR[cnt] = 0.0; }
166 
167  for (int geval=0; geval<4; geval++){
168  int NLup,TwoSLup,ILup,NLdown,TwoSLdown,ILdown;
169  switch(geval){
170  case 0:
171  NLup = sectorN1[ikappa]-1;
172  TwoSLup = sectorTwoS1[ikappa]-1;
173  ILup = denBK->directProd( sectorI1[ikappa] , denBK->gIrrep(index-1) );
174  NLdown = sectorN1[ikappa];
175  TwoSLdown = sectorTwoS1[ikappa];
176  ILdown = sectorI1[ikappa];
177  break;
178  case 1:
179  NLup = sectorN1[ikappa]-1;
180  TwoSLup = sectorTwoS1[ikappa]+1;
181  ILup = denBK->directProd( sectorI1[ikappa] , denBK->gIrrep(index-1) );
182  NLdown = sectorN1[ikappa];
183  TwoSLdown = sectorTwoS1[ikappa];
184  ILdown = sectorI1[ikappa];
185  break;
186  case 2:
187  NLup = sectorN1[ikappa]-2;
188  TwoSLup = sectorTwoS1[ikappa];
189  ILup = sectorI1[ikappa];
190  NLdown = sectorN1[ikappa]-1;
191  TwoSLdown = sectorTwoS1[ikappa]-1;
192  ILdown = denBK->directProd( sectorI1[ikappa] , denBK->gIrrep(index-1) );
193  break;
194  case 3:
195  NLup = sectorN1[ikappa]-2;
196  TwoSLup = sectorTwoS1[ikappa];
197  ILup = sectorI1[ikappa];
198  NLdown = sectorN1[ikappa]-1;
199  TwoSLdown = sectorTwoS1[ikappa]+1;
200  ILdown = denBK->directProd( sectorI1[ikappa] , denBK->gIrrep(index-1) );
201  break;
202  }
203  int dimLup = denBK->gCurrentDim(index-1, NLup, TwoSLup, ILup);
204  int dimLdown = denBK->gCurrentDim(index-1, NLdown, TwoSLdown, ILdown);
205 
206  if ((dimLup>0) && (dimLdown>0)){
207  double * BlockTup = denT->gStorage(NLup, TwoSLup, ILup, sectorN1[ikappa], sectorTwoS1[ikappa], sectorI1[ikappa]);
208  double * BlockTdown = denT->gStorage(NLdown, TwoSLdown, ILdown, sectorN1[ikappa], sectorTwoS1[ikappa], sectorI1[ikappa]);
209  double * BlockQ = Qprev->gStorage(NLup, TwoSLup, ILup, NLdown, TwoSLdown, ILdown);
210 
211  double factor;
212  double * ptr;
213  if (geval<2){
214 
215  factor = 1.0;
216  ptr = BlockQ;
217 
218  } else {
219 
220  int fase = ((((sectorTwoS1[ikappa]+1-TwoSLdown)/2)%2)!=0)?-1:1;
221  factor = fase * sqrt((TwoSLdown + 1.0)/(sectorTwoS1[ikappa] + 1.0));
222 
223  int dimLupdown = dimLup * dimLdown;
224  int inc = 1;
225  ptr = workmemLL;
226  dcopy_(&dimLupdown,BlockQ,&inc,ptr,&inc);
227 
228  for (int loca=0; loca<index-1; loca++){
229  if (denBK->gIrrep(index-1) == denBK->gIrrep(loca)){
230  double alpha = Prob->gMxElement(loca, index-1, index-1, index-1);
231  double * BlockL = Lprev[index-2-loca]->gStorage(NLup, TwoSLup, ILup, NLdown, TwoSLdown, ILdown);
232  daxpy_(&dimLupdown,&alpha,BlockL,&inc,ptr,&inc);
233  }
234  }
235 
236  }
237 
238  //factor * Tup^T * L --> mem2 //set
239  char trans = 'T';
240  char notr = 'N';
241  double beta = 0.0;
242  dgemm_(&trans,&notr,&dimR,&dimLdown,&dimLup,&factor,BlockTup,&dimLup,ptr,&dimLup,&beta,workmemLR,&dimR);
243 
244  //mem2 * Tdown --> mem //add
245  factor = 1.0;
246  beta = 1.0;
247  dgemm_(&notr,&notr,&dimR,&dimR,&dimLdown,&factor,workmemLR,&dimR,BlockTdown,&dimLdown,&beta,workmemRR,&dimR);
248 
249  }
250  }
251  //mem + mem^T --> storage
252  for (int irow = 0; irow<dimR; irow++){
253  for (int icol = irow; icol<dimR; icol++){
254  workmemRR[irow + dimR * icol] += workmemRR[icol + dimR * irow];
255  workmemRR[icol + dimR * irow] = workmemRR[irow + dimR * icol];
256  }
257  }
258  int inc = 1;
259  double alpha = 1.0;
260  daxpy_(&dimTot,&alpha,workmemRR,&inc,storage + kappa2index[ikappa],&inc);
261
