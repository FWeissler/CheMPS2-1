33 CheMPS2::FCI::FCI(Hamiltonian * Ham, const unsigned int theNel_up, const unsigned int theNel_down, const int TargetIrrep_in, const double maxMemWorkMB_in, const int FCIverbose_in){
34 
35  // Copy the basic information
36  FCIverbose = FCIverbose_in;
37  maxMemWorkMB = maxMemWorkMB_in;
38  L = Ham->getL();
39  assert( theNel_up <= L );
40  assert( theNel_down <= L );
41  assert( maxMemWorkMB > 0.0 );
42  Nel_up = theNel_up;
43  Nel_down = theNel_down;
44 
45  // Construct the irrep product table and the list with the orbitals irreps
46  CheMPS2::Irreps myIrreps( Ham->getNGroup() );
47  NumIrreps = myIrreps.getNumberOfIrreps();
48  TargetIrrep = TargetIrrep_in;
49  orb2irrep = new int[ L ];
50  for (unsigned int orb = 0; orb < L; orb++){ orb2irrep[ orb ] = Ham->getOrbitalIrrep( orb ); }
51 
52  /* Copy the Hamiltonian over:
53  G_ij = T_ij - 0.5 \sum_k <ik|kj> and ERI_{ijkl} = <ij|kl>
54  <ij|kl> is the electron repulsion integral, int dr1 dr2 i(r1) j(r1) k(r2) l(r2) / |r1-r2| */
55  Econstant = Ham->getEconst();
56  Gmat = new double[ L * L ];
57  ERI = new double[ L * L * L * L ];
58  for (unsigned int orb1 = 0; orb1 < L; orb1++){
59  for (unsigned int orb2 = 0; orb2 < L; orb2++){
60  double tempvar = 0.0;
61  for (unsigned int orb3 = 0; orb3 < L; orb3++){
62  tempvar += Ham->getVmat( orb1, orb3, orb3, orb2 );
63  for (unsigned int orb4 = 0; orb4 < L; orb4++){
64  // CheMPS2::Hamiltonian uses physics notation ; ERI chemists notation.
65  ERI[ orb1 + L * ( orb2 + L * ( orb3 + L * orb4 ) ) ] = Ham->getVmat( orb1 , orb3 , orb2 , orb4 );
66  }
67  }
68  Gmat[ orb1 + L * orb2 ] = Ham->getTmat( orb1 , orb2 ) - 0.5 * tempvar;
69  }
70  }
71 
72  // Set all other internal variables
73  StartupCountersVsBitstrings();
74  StartupLookupTables();
75  StartupIrrepCenter();
76 
77 }

134  // Can you represent the alpha and beta Slater determinants as unsigned integers?
135  assert( L <= CHAR_BIT * sizeof(unsigned int) );
136 
137  // Variable which is only needed here: 2^L
138  unsigned int TwoPowL = 1;
139  for (unsigned int orb = 0; orb < L; orb++){ TwoPowL *= 2; }
140 
141  // Create the required arrays to perform the conversions between counters and bitstrings
142  numPerIrrep_up = new unsigned int[ NumIrreps ];
143  numPerIrrep_down = new unsigned int[ NumIrreps ];
144  str2cnt_up = new int*[ NumIrreps ];
145  str2cnt_down = new int*[ NumIrreps ];

147  cnt2str_down = new unsigned int*[ NumIrreps ];
148 
149  for (unsigned int irrep = 0; irrep < NumIrreps; irrep++){
150  numPerIrrep_up [ irrep ] = 0;
151  numPerIrrep_down[ irrep ] = 0;
152  str2cnt_up [ irrep ] = new int[ TwoPowL ];
153  str2cnt_down[ irrep ] = new int[ TwoPowL ];
154  }
155 
156  int * bits = new int[ L ]; // Temporary helper array
157 
158  // Loop over all allowed bit strings in the spinless fermion Fock space
159  for (unsigned int bitstring = 0; bitstring < TwoPowL; bitstring++){
160 
161  // Find the number of particles and the irrep which correspond to each basis vector
162  str2bits( L , bitstring , bits );
163  unsigned int Nparticles = 0;
164  int Irrep = 0;
165  for (unsigned int orb=0; orb<L; orb++){
166  if ( bits[orb] ){
167  Nparticles++;
168  Irrep = getIrrepProduct( Irrep , getOrb2Irrep( orb ) );
169  }
170  }
171 
172  // If allowed: set the corresponding str2cnt to the correct counter and keep track of the number of allowed vectors
173  for ( unsigned int irr = 0; irr < NumIrreps; irr++ ){
174  str2cnt_up [ irr ][ bitstring ] = -1;
175  str2cnt_down[ irr ][ bitstring ] = -1;
176  }
177  if ( Nparticles == Nel_up ){
178  str2cnt_up[ Irrep ][ bitstring ] = numPerIrrep_up[ Irrep ];
179  numPerIrrep_up[ Irrep ]++;
180  }
181  if ( Nparticles == Nel_down ){
182  str2cnt_down[ Irrep ][ bitstring ] = numPerIrrep_down[ Irrep ];
183  numPerIrrep_down[ Irrep ]++;
184  }
185 
186  }
187 
188  // Fill the reverse info array: cnt2str
189  for ( unsigned int irrep = 0; irrep < NumIrreps; irrep++ ){
190 
191  if ( FCIverbose>1 ){

193  << numPerIrrep_down[ irrep ] << " beta Slater determinants." << endl;
194  }
195

197  cnt2str_down[ irrep ] = new unsigned int[ numPerIrrep_down[ irrep ] ];
198  for (unsigned int bitstring = 0; bitstring < TwoPowL; bitstring++){
199  if ( str2cnt_up [ irrep ][ bitstring ] != -1 ){ cnt2str_up [ irrep ][ str2cnt_up [ irrep ][ bitstring ] ] = bitstring; }
200  if ( str2cnt_down[ irrep ][ bitstring ] != -1 ){ cnt2str_down[ irrep ][ str2cnt_down[ irrep ][ bitstring ] ] = bitstring; }
201  }
202 
203  }
204 
205  delete [] bits; // Delete temporary helper array

225  lookup_irrep_alpha[ irrep ] = new int[ L * L * numPerIrrep_up[ irrep ] ];

231  for ( unsigned int j = 0; j < L; j++){
232  // Check for the sign. If no check for sign, you multiply with sign 0 and everything should be OK...
233  lookup_cnt_alpha [irrep][ i + L * ( j + L * cnt_new_alpha ) ] = 0;
234  lookup_irrep_alpha[irrep][ i + L * ( j + L * cnt_new_alpha ) ] = 0;

240 
241  int phase_creator = 1;
242  for ( unsigned int creator = 0; creator < L; creator++ ){
243  if ( bits[ creator ] ){
244  bits[ creator ] = 0;
245 
246  int phase_annihilator = 1;
247  for ( unsigned int annihilator = 0; annihilator < L; annihilator++ ){
248  if ( !(bits[ annihilator ]) ){
249  bits[ annihilator ] = 1;
250 
251  const int irrep_old = getIrrepProduct( irrep , getIrrepProduct( getOrb2Irrep( creator ) , getOrb2Irrep( annihilator ) ) );
252  const int cnt_old = str2cnt_up[ irrep_old ][ bits2str( L , bits ) ];
253  const int phase = phase_creator * phase_annihilator;
254 
255  lookup_cnt_alpha [irrep][ creator + L * ( annihilator + L * cnt_new_alpha ) ] = cnt_old;
256  lookup_irrep_alpha[irrep][ creator + L * ( annihilator + L * cnt_new_alpha ) ] = irrep_old;
257  lookup_sign_alpha [irrep][ creator + L * ( annihilator + L * cnt_new_alpha ) ] = phase;
258 
259  bits[ annihilator ] = 0;
260  } else {
261  phase_annihilator *= -1;
262  }
263  }
264 
265  bits[ creator ] = 1;
266  phase_creator *= -1;
267  }
268  }
269  }
270 
271  lookup_cnt_beta [ irrep ] = new int[ L * L * numPerIrrep_down[ irrep ] ];
272  lookup_irrep_beta[ irrep ] = new int[ L * L * numPerIrrep_down[ irrep ] ];
273  lookup_sign_beta [ irrep ] = new int[ L * L * numPerIrrep_down[ irrep ] ];
274

276 
277  for ( unsigned int i = 0; i < L; i++ ){
278  for ( unsigned int j = 0; j < L; j++ ){
279  // Check for the sign. If no check for sign, you multiply with sign and everything should be OK...

290  if ( bits[ creator ] ){
291  bits[ creator ] = 0;
292

295  if ( !(bits[ annihilator ]) ){
296  bits[ annihilator ] = 1;
297
