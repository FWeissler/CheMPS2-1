48 long long CheMPS2::FourIndex::calcNumberOfUniqueElements(const bool allocate){
49 
50  //The object size: see text above storage in Fourindex.h
51  long long theTotalSize = 0;
52 
53  for (int Icenter=0; Icenter<SymmInfo.getNumberOfIrreps(); Icenter++){
54  if (allocate){ storage[Icenter] = new long long***[SymmInfo.getNumberOfIrreps()]; }
55  for (int I_i=0; I_i<SymmInfo.getNumberOfIrreps(); I_i++){
56  int I_j = Irreps::directProd(Icenter,I_i);
57  if ((Isizes[I_i]>0)&&(Isizes[I_j]>0)){
58  if (allocate){ storage[Icenter][I_i] = new long long**[SymmInfo.getNumberOfIrreps()]; }
59  for (int I_k=I_i; I_k<SymmInfo.getNumberOfIrreps(); I_k++){
60  int I_l = Irreps::directProd(Icenter,I_k);
61  if ((Isizes[I_k]>0)&&(Isizes[I_l]>0)){
62  if ((I_i <= I_j) && (I_j <= I_l)){
63  if (Icenter == 0){ // I_i = I_j and I_k = I_l
64  if (I_i == I_k){
65  if (allocate){ storage[Icenter][I_i][I_k] = new long long*[Isizes[I_i]*(Isizes[I_i]+1)/2]; }
66  for (int i=0; i<Isizes[I_i]; i++){
67  for (int k=i; k<Isizes[I_k]; k++){
68  if (allocate){
69  storage[Icenter][I_i][I_k][i + k*(k+1)/2] = new long long[Isizes[I_j]-i];
70  for (int j=i; j<Isizes[I_j]; j++){
71  storage[Icenter][I_i][I_k][i + k*(k+1)/2][j-i] = theTotalSize;
72  theTotalSize += Isizes[I_l] - ((i==j) ? k : j);
73  }
74  } else {
75  delete [] storage[Icenter][I_i][I_k][i + k*(k+1)/2];
76  }
77  }
78  }
79  if (!allocate){ delete [] storage[Icenter][I_i][I_k]; }
80  } else { // I_i < I_k
81  if (allocate){ storage[Icenter][I_i][I_k] = new long long*[Isizes[I_i]*Isizes[I_k]]; }
82  for (int i=0; i<Isizes[I_i]; i++){
83  for (int k=0; k<Isizes[I_k]; k++){
84  if (allocate){
85  storage[Icenter][I_i][I_k][i + k*Isizes[I_i]] = new long long[Isizes[I_j]-i];
86  for (int j=i; j<Isizes[I_j]; j++){
87  storage[Icenter][I_i][I_k][i+k*Isizes[I_i]][j-i] = theTotalSize;
88  theTotalSize += Isizes[I_l] - ((i==j) ? k : 0 );
89  }
90  } else {
91  delete [] storage[Icenter][I_i][I_k][i + k*Isizes[I_i]];
92  }
93  }
94  }
95  if (!allocate){ delete [] storage[Icenter][I_i][I_k]; }
96  }
97  } else { //Icenter !=0 ; I_i < I_j and I_k != I_l
98  if (I_i == I_k){
99  if (allocate){ storage[Icenter][I_i][I_k] = new long long*[Isizes[I_i]*(Isizes[I_i]+1)/2]; }
100  for (int i=0; i<Isizes[I_i]; i++){
101  for (int k=i; k<Isizes[I_k]; k++){
102  if (allocate){
103  storage[Icenter][I_i][I_k][i + k*(k+1)/2] = new long long[Isizes[I_j]];
104  for (int j=0; j<Isizes[I_j]; j++){
105  storage[Icenter][I_i][I_k][i + k*(k+1)/2][j] = theTotalSize;
106  theTotalSize += Isizes[I_l] - j;
107  }
108  } else {
109  delete [] storage[Icenter][I_i][I_k][i + k*(k+1)/2];
110  }
111  }
112  }
113  if (!allocate){ delete [] storage[Icenter][I_i][I_k]; }
114  } else { // I_i < I_k
115  if (allocate){ storage[Icenter][I_i][I_k] = new long long*[Isizes[I_i]*Isizes[I_k]]; }
116  for (int i=0; i<Isizes[I_i]; i++){
117  for (int k=0; k<Isizes[I_k]; k++){
118  if (allocate){
119  storage[Icenter][I_i][I_k][i + k*Isizes[I_i]] = new long long[Isizes[I_j]];
120  for (int j=0; j<Isizes[I_j]; j++){
121  storage[Icenter][I_i][I_k][i + k*Isizes[I_i]][j] = theTotalSize;
122  theTotalSize += Isizes[I_l];
123  }
124  } else {
125  delete [] storage[Icenter][I_i][I_k][i + k*Isizes[I_i]];
126  }
127  }
128  }
129  if (!allocate){ delete [] storage[Icenter][I_i][I_k]; }
130  }
131  }
132  }
133  }
134  }
135  if (!allocate){ delete [] storage[Icenter][I_i]; }
136  }
137  }
138  if (!allocate){ delete [] storage[Icenter]; }
139  }
140 
141  return theTotalSize;

172 long long CheMPS2::FourIndex::getPointer(const int irrep_i, const int irrep_j, const int irrep_k, const int irrep_l, const int i, const int j, const int k, const int l) const {
173 
174  if (Irreps::directProd(irrep_i,irrep_j)==Irreps::directProd(irrep_k,irrep_l)){
175 
176  if ((irrep_i <= irrep_j) && (irrep_i <= irrep_k) && (irrep_j <= irrep_l)){ // (ijkl irrep ordering)
177  return getPtrIrrepOrderOK(irrep_i,irrep_j,irrep_k,irrep_l,i,j,k,l);
178  }
179 
180  if ((irrep_j <= irrep_i) && (irrep_j <= irrep_l) && (irrep_i <= irrep_k)){ // (jilk irrep ordering)
181  return getPtrIrrepOrderOK(irrep_j,irrep_i,irrep_l,irrep_k,j,i,l,k);
182  }
183 
184  if ((irrep_k <= irrep_j) && (irrep_k <= irrep_i) && (irrep_j <= irrep_l)){ // (kjil irrep ordering)
185  return getPtrIrrepOrderOK(irrep_k,irrep_j,irrep_i,irrep_l,k,j,i,l);
186  }
187 
188  if ((irrep_j <= irrep_k) && (irrep_j <= irrep_l) && (irrep_k <= irrep_i)){ // (jkli irrep ordering)
189  return getPtrIrrepOrderOK(irrep_j,irrep_k,irrep_l,irrep_i,j,k,l,i);
190  }
191 
192  if ((irrep_i <= irrep_l) && (irrep_i <= irrep_k) && (irrep_l <= irrep_j)){ // (ilkj irrep ordering)
193  return getPtrIrrepOrderOK(irrep_i,irrep_l,irrep_k,irrep_j,i,l,k,j);
194  }
195 
196  if ((irrep_l <= irrep_i) && (irrep_l <= irrep_j) && (irrep_i <= irrep_k)){ // (lijk irrep ordering)
197  return getPtrIrrepOrderOK(irrep_l,irrep_i,irrep_j,irrep_k,l,i,j,k);
198  }
199 
200  if ((irrep_k <= irrep_l) && (irrep_k <= irrep_i) && (irrep_l <= irrep_j)){ // (klij irrep ordering)
201  return getPtrIrrepOrderOK(irrep_k,irrep_l,irrep_i,irrep_j,k,l,i,j);
202  }
203 
204  if ((irrep_l <= irrep_k) && (irrep_l <= irrep_j) && (irrep_k <= irrep_i)){ // (lkji irrep ordering)
205  return getPtrIrrepOrderOK(irrep_l,irrep_k,irrep_j,irrep_i,l,k,j,i);

214 long long CheMPS2::FourIndex::getPtrIrrepOrderOK(const int irrep_i, const int irrep_j, const int irrep_k, const int irrep_l, const int i, const int j, const int k, const int l) const {
215 
216  //I_i <= I_j <= I_l and I_i <= I_k
217  int Icenter = Irreps::directProd(irrep_i,irrep_j);
218 
219  if (Icenter>0){ // I_i < I_j and I_k != I_l
220 
221  if (irrep_i == irrep_k){ //and hence I_j == I_l
222 
223  if (k>=i){
224  if (l>=j) return getPtrAllOK(5, Icenter, irrep_i, irrep_k, i, j, k, l);
225  else return getPtrAllOK(5, Icenter, irrep_i, irrep_k, i, l, k, j);
226  } else {
227  if (l>=j) return getPtrAllOK(5, Icenter, irrep_k, irrep_i, k, j, i, l);
228  else return getPtrAllOK(5, Icenter, irrep_k, irrep_i, k, l, i, j);
229  }
230 
231  } else return getPtrAllOK(6, Icenter, irrep_i, irrep_k, i, j, k, l);
232 
233  } else {
234 
235  if (irrep_i == irrep_k){ // all irreps the same
236 
237  //i en j
238  if ((i < j) && (i <= k) && (j <= l)) return getPtrAllOK(2, Icenter, irrep_i, irrep_k, i, j, k, l); // (ijkl ordering)
239  if ((i == j) && (i <= k) && (j <= l)){
240  if (l>=k) return getPtrAllOK(1, Icenter, irrep_i, irrep_k, i, j, k, l); // (ijkl ordering)
241  else return getPtrAllOK(1, Icenter, irrep_j, irrep_l, j, i, l, k); // (jilk ordering)
242  }
243  if ((j < i) && (j <= l) && (i <= k)) return getPtrAllOK(2, Icenter, irrep_j, irrep_l, j, i, l, k); // (jilk ordering)
244 
245  //k en l
246  if ((k < l) && (k <= i) && (l <= j)) return getPtrAllOK(2, Icenter, irrep_k, irrep_i, k, l, i, j); // (klij ordering)
247  if ((k == l) && (k <= i) && (l <= j)){
248  if (j>=i) return getPtrAllOK(1, Icenter, irrep_k, irrep_i, k, l, i, j); // (klij ordering)
249  else return getPtrAllOK(1, Icenter, irrep_l, irrep_j, l, k, j, i); // (lkji ordering)
250  }
251  if ((l < k) && (l <= j) && (k <= i)) return getPtrAllOK(2, Icenter, irrep_l, irrep_j, l, k, j, i); // (lkji ordering)
252 
253  // k en j
254  if ((k < j) && (k <= i) && (j <= l)) return getPtrAllOK(2, Icenter, irrep_k, irrep_i, k, j, i, l); // (kjil ordering)
255  if ((k == j) && (k <= i) && (j <= l)){
256  if (l>=i) return getPtrAllOK(1, Icenter, irrep_k, irrep_i, k, j, i, l); // (kjil ordering)
257  else return getPtrAllOK(1, Icenter, irrep_j, irrep_l, j, k, l, i); // (jkli ordering)
258  }
259  if ((j < k) && (j <= l) && (k <= i)) return getPtrAllOK(2, Icenter, irrep_j, irrep_l, j, k, l, i); // (jkli ordering)
260 
261  // i en l
262  if ((i < l) && (i <= k) && (l <= j)) return getPtrAllOK(2, Icenter, irrep_i, irrep_k, i, l, k, j); // (ilkj ordering)
263  if ((i == l) && (i <= k) && (l <= j)){
264  if (j>=k) return getPtrAllOK(1, Icenter, irrep_i, irrep_k, i, l, k, j); // (ilkj ordering)
265  else return getPtrAllOK(1, Icenter, irrep_l, irrep_j, l, i, j, k); // (lijk ordering)
266  }
267  if ((l < i) && (l <= j) && (i <= k)) return getPtrAllOK(2, Icenter, irrep_l, irrep_j, l, i, j, k); // (lijk ordering)
268 
269  } else {
270 
271  if (j==i){
272  if (l>=k) return getPtrAllOK(3, Icenter, irrep_i, irrep_k, i, j, k, l);
273  else return getPtrAllOK(3, Icenter, irrep_j, irrep_l, j, i, l, k);
274  } else {
275  if (j>i) return getPtrAllOK(4, Icenter, irrep_i, irrep_k, i, j, k, l);
276  else return getPtrAllOK(4, Icenter, irrep_j, irrep_l, j, i, l, k);
277  }
278 
279  }
280 
281  }
282 
283  return -1;

287 long long CheMPS2::FourIndex::getPtrAllOK(const int number, const int Icent, const int irrep_i, const int irrep_k, const int i, const int j, const int k, const int l) const {
288 
289  switch (number){
290  case 1:
291  return storage[Icent][irrep_i][irrep_k][i + k*(k+1)/2][j-i] + l-k;
292  case 2:
293  return storage[Icent][irrep_i][irrep_k][i + k*(k+1)/2][j-i] + l-j;
294  case 3:
295  return storage[Icent][irrep_i][irrep_k][i + Isizes[irrep_i]*k][j-i] + l-k;
296  case 4:
297  return storage[Icent][irrep_i][irrep_k][i + Isizes[irrep_i]*k][j-i] + l;
298  case 5:
299  return storage[Icent][irrep_i][irrep_k][i + k*(k+1)/2][j] + l-j;
300  case 6:
301  return storage[Icent][irrep_i][irrep_k][i + Isizes[irrep_i]*k][j] + l;
302  }
