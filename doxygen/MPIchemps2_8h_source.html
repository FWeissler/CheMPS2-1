268  static void sendreceive_tensor(Tensor * object, int SENDER, int RECEIVER, int tag){
269  if ( SENDER != RECEIVER ){
270  const int MPIRANK = mpi_rank();
271  if ( SENDER == MPIRANK ){
272  int arraysize = object->gKappa2index(object->gNKappa());
273  MPI_Send(object->gStorage(), arraysize, MPI_DOUBLE, RECEIVER, tag, MPI_COMM_WORLD);
274  }
275  if ( RECEIVER == MPIRANK ){
276  int arraysize = object->gKappa2index(object->gNKappa());
277  MPI_Recv(object->gStorage(), arraysize, MPI_DOUBLE, SENDER, tag, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
278  }
279  }
280  }
281  #endif
