171 void CheMPS2::TensorOperator::update_moving_right(const int ikappa, TensorOperator * previous, TensorT * mps_tensor, double * workmem){
172 
173  const int n_right_up = sectorN1[ ikappa ];
174  const int n_right_down = n_right_up + n_elec;
175  const int two_s_right_up = sectorTwoS1[ ikappa ];
176  const int two_s_right_down = sector_2S_down[ ikappa ];
177  const int irrep_right_up = sectorI1[ ikappa ];
178  const int irrep_right_down = Irreps::directProd( irrep_right_up, n_irrep );
179 
180  int dim_right_up = denBK->gCurrentDim( index, n_right_up, two_s_right_up, irrep_right_up );
181  int dim_right_down = denBK->gCurrentDim( index, n_right_down, two_s_right_down, irrep_right_down );
182 
183  for (int geval = 0; geval < 6; geval++){
184  int n_left_up, n_left_down, two_s_left_up, two_s_left_down, irrep_left_up, irrep_left_down;
185  switch ( geval ){
186  case 0: // MPS tensor sector (I,J,N) = (0,0,0)
187  two_s_left_up = two_s_right_up;
188  two_s_left_down = two_s_right_down;
189  n_left_up = n_right_up;
190  n_left_down = n_right_down;
191  irrep_left_up = irrep_right_up;
192  irrep_left_down = irrep_right_down;
193  break;
194  case 1: // MPS tensor sector (I,J,N) = (0,0,2)
195  two_s_left_up = two_s_right_up;
196  two_s_left_down = two_s_right_down;
197  n_left_up = n_right_up - 2;
198  n_left_down = n_right_down - 2;
199  irrep_left_up = irrep_right_up;
200  irrep_left_down = irrep_right_down;
201  break;
202  case 2: // MPS tensor sector (I,J,N) = (Ilocal,1/2,1)
203  two_s_left_up = two_s_right_up - 1;
204  two_s_left_down = two_s_right_down - 1;
205  n_left_up = n_right_up - 1;
206  n_left_down = n_right_down - 1;
207  irrep_left_up = Irreps::directProd( irrep_right_up, denBK->gIrrep(index-1) );
208  irrep_left_down = Irreps::directProd( irrep_right_down, denBK->gIrrep(index-1) );
209  break;
210  case 3: // MPS tensor sector (I,J,N) = (Ilocal,1/2,1)
211  two_s_left_up = two_s_right_up - 1;
212  two_s_left_down = two_s_right_down + 1;
213  n_left_up = n_right_up - 1;
214  n_left_down = n_right_down - 1;
215  irrep_left_up = Irreps::directProd( irrep_right_up, denBK->gIrrep(index-1) );
216  irrep_left_down = Irreps::directProd( irrep_right_down, denBK->gIrrep(index-1) );
217  break;
218  case 4: // MPS tensor sector (I,J,N) = (Ilocal,1/2,1)
219  two_s_left_up = two_s_right_up + 1;
220  two_s_left_down = two_s_right_down - 1;
221  n_left_up = n_right_up - 1;
222  n_left_down = n_right_down - 1;
223  irrep_left_up = Irreps::directProd( irrep_right_up, denBK->gIrrep(index-1) );
224  irrep_left_down = Irreps::directProd( irrep_right_down, denBK->gIrrep(index-1) );
225  break;
226  case 5: // MPS tensor sector (I,J,N) = (Ilocal,1/2,1)
227  two_s_left_up = two_s_right_up + 1;
228  two_s_left_down = two_s_right_down + 1;
229  n_left_up = n_right_up - 1;
230  n_left_down = n_right_down - 1;
231  irrep_left_up = Irreps::directProd( irrep_right_up, denBK->gIrrep(index-1) );
232  irrep_left_down = Irreps::directProd( irrep_right_down, denBK->gIrrep(index-1) );
233  break;
234  }
235 
236  if ( abs( two_s_left_up - two_s_left_down ) <= two_j ){
237 
238  int dim_left_up = denBK->gCurrentDim( index-1, n_left_up, two_s_left_up, irrep_left_up );
239  int dim_left_down = denBK->gCurrentDim( index-1, n_left_down, two_s_left_down, irrep_left_down );
240 
241  if (( dim_left_up > 0 ) && ( dim_left_down > 0 )){
242 
243  double * mps_block_up = mps_tensor->gStorage( n_left_up, two_s_left_up, irrep_left_up, n_right_up, two_s_right_up, irrep_right_up );
244  double * mps_block_down = mps_tensor->gStorage( n_left_down, two_s_left_down, irrep_left_down, n_right_down, two_s_right_down, irrep_right_down );
245  double * left_block = previous->gStorage( n_left_up, two_s_left_up, irrep_left_up, n_left_down, two_s_left_down, irrep_left_down );
246 
247  // Prefactor
248  double alpha = 1.0;
249  if ( geval >= 2 ){
250  if ( two_j == 0 ){
251  alpha = ( ( jw_phase ) ? -1.0 : 1.0 );
252  } else {
253  if ( prime_last ){
254  alpha = CheMPS2::Heff::phase( two_s_right_up + two_s_left_down + two_j + ( ( jw_phase ) ? 3 : 1 ) )
255  * sqrt( ( two_s_left_down + 1.0 ) * ( two_s_right_up + 1.0 ) )
256  * gsl_sf_coupling_6j( two_s_left_up, two_s_left_down, two_j, two_s_right_down, two_s_right_up, 1 );
257  } else {
258  alpha = CheMPS2::Heff::phase( two_s_right_down + two_s_left_up + two_j + ( ( jw_phase ) ? 3 : 1 ) )
259  * sqrt( ( two_s_left_up + 1.0 ) * ( two_s_right_down + 1.0 ) )
260  * gsl_sf_coupling_6j( two_s_left_down, two_s_left_up, two_j, two_s_right_up, two_s_right_down, 1 );
261  }
262  }
263  }
264 
265  // prefactor * mps_block_up^T * left_block --> mem
266  char trans = 'T';
267  char notr = 'N';
268  double beta = 0.0; //set
269  dgemm_(&trans, &notr, &dim_right_up, &dim_left_down, &dim_left_up,
270  &alpha, mps_block_up, &dim_left_up, left_block, &dim_left_up,
271  &beta, workmem, &dim_right_up);
272 
273  // mem * mps_block_down --> storage
274  alpha = 1.0;
275  beta = 1.0; //add
276  dgemm_(&notr, &notr, &dim_right_up, &dim_right_down, &dim_left_down,
277  &alpha, workmem, &dim_right_up, mps_block_down, &dim_left_down,
278  &beta, storage + kappa2index[ikappa], &dim_right_up);
279  }
280  }
281  }

285 void CheMPS2::TensorOperator::update_moving_left(const int ikappa, TensorOperator * previous, TensorT * mps_tensor, double * workmem){
286 
287  const int n_left_up = sectorN1[ ikappa ];
288  const int n_left_down = n_left_up + n_elec;
289  const int two_s_left_up = sectorTwoS1[ ikappa ];
290  const int two_s_left_down = sector_2S_down[ ikappa ];
291  const int irrep_left_up = sectorI1[ ikappa ];
292  const int irrep_left_down = Irreps::directProd( irrep_left_up, n_irrep );
293 
294  int dim_left_up = denBK->gCurrentDim( index, n_left_up, two_s_left_up, irrep_left_up );
295  int dim_left_down = denBK->gCurrentDim( index, n_left_down, two_s_left_down, irrep_left_down );
296 
297  for (int geval = 0; geval < 6; geval++){
298  int n_right_up, n_right_down, two_s_right_up, two_s_right_down, irrep_right_up, irrep_right_down;
299  switch ( geval ){
300  case 0: // MPS tensor sector (I,J,N) = (0,0,0)
301  two_s_right_up = two_s_left_up;
302  two_s_right_down = two_s_left_down;
303  n_right_up = n_left_up;
304  n_right_down = n_left_down;
305  irrep_right_up = irrep_left_up;
306  irrep_right_down = irrep_left_down;
307  break;
308  case 1: // MPS tensor sector (I,J,N) = (0,0,2)
309  two_s_right_up = two_s_left_up;
310  two_s_right_down = two_s_left_down;
311  n_right_up = n_left_up + 2;
312  n_right_down = n_left_down + 2;
313  irrep_right_up = irrep_left_up;
314  irrep_right_down = irrep_left_down;
315  break;
316  case 2: // MPS tensor sector (I,J,N) = (Ilocal,1/2,1)
317  two_s_right_up = two_s_left_up - 1;
318  two_s_right_down = two_s_left_down - 1;
319  n_right_up = n_left_up + 1;
320  n_right_down = n_left_down + 1;
321  irrep_right_up = Irreps::directProd( irrep_left_up, denBK->gIrrep(index) );
322  irrep_right_down = Irreps::directProd( irrep_left_down, denBK->gIrrep(index) );
323  break;
324  case 3: // MPS tensor sector (I,J,N) = (Ilocal,1/2,1)
325  two_s_right_up = two_s_left_up - 1;
326  two_s_right_down = two_s_left_down + 1;
327  n_right_up = n_left_up + 1;
328  n_right_down = n_left_down + 1;
329  irrep_right_up = Irreps::directProd( irrep_left_up, denBK->gIrrep(index) );
330  irrep_right_down = Irreps::directProd( irrep_left_down, denBK->gIrrep(index) );
331  break;
332  case 4: // MPS tensor sector (I,J,N) = (Ilocal,1/2,1)
333  two_s_right_up = two_s_left_up + 1;
334  two_s_right_down = two_s_left_down - 1;
335  n_right_up = n_left_up + 1;
336  n_right_down = n_left_down + 1;
337  irrep_right_up = Irreps::directProd( irrep_left_up, denBK->gIrrep(index) );
338  irrep_right_down = Irreps::directProd( irrep_left_down, denBK->gIrrep(index) );
339  break;
340  case 5: // MPS tensor sector (I,J,N) = (Ilocal,1/2,1)
341  two_s_right_up = two_s_left_up + 1;
342  two_s_right_down = two_s_left_down + 1;
343  n_right_up = n_left_up + 1;
344  n_right_down = n_left_down + 1;
345  irrep_right_up = Irreps::directProd( irrep_left_up, denBK->gIrrep(index) );
346  irrep_right_down = Irreps::directProd( irrep_left_down, denBK->gIrrep(index) );
347  break;
348  }
349 
350  if ( abs( two_s_right_up - two_s_right_down ) <= two_j ){
351 
352  int dim_right_up = denBK->gCurrentDim( index+1, n_right_up, two_s_right_up, irrep_right_up );
353  int dim_right_down = denBK->gCurrentDim( index+1, n_right_down, two_s_right_down, irrep_right_down );
354 
355  if (( dim_right_up > 0 ) && ( dim_right_down > 0 )){
356 
357  double * mps_block_up = mps_tensor->gStorage( n_left_up, two_s_left_up, irrep_left_up, n_right_up, two_s_right_up, irrep_right_up );
358  double * mps_block_down = mps_tensor->gStorage( n_left_down, two_s_left_down, irrep_left_down, n_right_down, two_s_right_down, irrep_right_down );
359  double * right_block = previous->gStorage( n_right_up, two_s_right_up, irrep_right_up, n_right_down, two_s_right_down, irrep_right_down );
360 
361  // Prefactor
362  double alpha = 1.0;
363  if ( geval >= 2 ){
364  if ( two_j == 0 ){
365  alpha = ( ( jw_phase ) ? -1.0 : 1.0 ) * ( two_s_right_up + 1.0 ) / ( two_s_left_up + 1.0 );
366  } else {
367  if ( prime_last ){
368  alpha = CheMPS2::Heff::phase( two_s_right_up + two_s_left_down + two_j + ( ( jw_phase ) ? 3 : 1 ) )
369  * ( two_s_right_down + 1 ) * sqrt( ( two_s_right_up + 1.0 ) / ( two_s_left_down + 1.0 ) )
370  * gsl_sf_coupling_6j( two_s_right_up, two_s_right_down, two_j, two_s_left_down, two_s_left_up, 1 );
371  } else {
372  alpha = CheMPS2::Heff::phase( two_s_right_down + two_s_left_up + two_j + ( ( jw_phase ) ? 3 : 1 ) )
373  * ( two_s_right_up + 1 ) * sqrt( ( two_s_right_down + 1.0 ) / ( two_s_left_up + 1.0 ) )
374  * gsl_sf_coupling_6j( two_s_right_down, two_s_right_up, two_j, two_s_left_up, two_s_left_down, 1 );
375  }
376  }
377  }
378 
379  // prefactor * mps_block_up * right_block --> mem
380  char notr = 'N';
381  double beta = 0.0; //set
382  dgemm_(&notr, &notr, &dim_left_up, &dim_right_down, &dim_right_up,
383  &alpha, mps_block_up, &dim_left_up, right_block, &dim_right_up,
384  &beta, workmem, &dim_left_up);
385 
386  // mem * mps_block_down^T --> storage
387  char trans = 'T';
388  alpha = 1.0;
389  beta = 1.0; //add
390  dgemm_(&notr, &trans, &dim_left_up, &dim_left_down, &dim_right_down,
391  &alpha, workmem, &dim_left_up, mps_block_down, &dim_left_down,
392  &beta, storage + kappa2index[ikappa], &dim_left_up);
393  }
394  }
395  }

407 void CheMPS2::TensorOperator::daxpy_transpose_tensorCD(const double alpha, TensorOperator * to_add){
408 
409  assert( kappa2index[nKappa] == to_add->gKappa2index(to_add->gNKappa()) );
410  assert( n_elec == 0 );
411  assert( ( two_j == 0 ) || ( two_j == 2 ) );
412 
413  for (int ikappa=0; ikappa<nKappa; ikappa++){
414 
415  const int irrep_up = sectorI1[ikappa];
416  const int irrep_down = Irreps::directProd( irrep_up, n_irrep );
417  const int two_s_up = sectorTwoS1[ikappa];
418  const int two_s_down = sector_2S_down[ikappa];
419  const int n_updown = sectorN1[ikappa];
420 
421  const int dim_up = denBK->gCurrentDim( index, n_updown, two_s_up, irrep_up );
422  const int dim_down = denBK->gCurrentDim( index, n_updown, two_s_down, irrep_down );
423 
424  double prefactor = alpha;
425  /*
426  This phase factor comes historically from the TensorD and is not valid in general,
427  as it is tightly coupled to the specific change from (for moving_right == true ):
428  < 1/2 m1 1/2 -m2 | 1 (m1-m2) > * (-1)^{1/2-m2} * < j_L' j_L^z' 1 (m1-m2) | j_L j_L^z >
429  = < 1/2 m2 1/2 -m1 | 1 (m2-m1) > * (-1)^{1/2-m1} * < j_L j_L^z 1 (m1-m2) | j_L' j_L^z' > * prefactor
430  */
431  if ( two_s_up != two_s_down ){
432  prefactor *= CheMPS2::Heff::phase( two_s_up - two_s_down )
433  * sqrt(( moving_right ) ? (( two_s_up + 1.0 ) / ( two_s_down + 1.0 )) : (( two_s_down + 1.0 ) / ( two_s_up + 1.0 )));
