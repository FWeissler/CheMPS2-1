195 void CheMPS2::Sobject::Join(TensorT * Tleft, TensorT * Tright){
196 
197  //PARALLEL
198  #pragma omp parallel for schedule(dynamic)
199  for (int ikappa=0; ikappa<nKappa; ikappa++){
200  int NM = sectorNL[ikappa] + sectorN1[ikappa];
201  int IM = ((sectorN1[ikappa]==1)?(Irreps::directProd(sectorIL[ikappa],Ilocal1)):sectorIL[ikappa]);
202 
203  int nCases = 1; // number of TwoJM possibilities --> most cases 1, in the case of the next line: 2
204  if ((sectorTwoSR[ikappa]==sectorTwoSL[ikappa]) && (sectorN1[ikappa]==1) && (sectorN2[ikappa]==1) && (sectorTwoSR[ikappa]>=1)) nCases = 2;
205 
206  int * TwoJM = new int[nCases];
207  if (nCases==2){
208  TwoJM[0] = sectorTwoSL[ikappa]-1;
209  TwoJM[1] = sectorTwoSL[ikappa]+1;
210  } else { // 1 case
211  if (((sectorN1[ikappa])%2)==0) TwoJM[0] = sectorTwoSL[ikappa];
212  else {
213  if (((sectorN2[ikappa])%2)==0) TwoJM[0] = sectorTwoSR[ikappa];
214  else { //N1==1 and N2==1
215  if ((sectorTwoSR[ikappa]==sectorTwoSL[ikappa]) && (sectorTwoSR[ikappa] == 0)) TwoJM[0] = 1;
216  else { //TwoSR != TwoSL
217  if (sectorTwoSR[ikappa]>sectorTwoSL[ikappa]) TwoJM[0] = sectorTwoSL[ikappa] + 1;
218  else TwoJM[0] = sectorTwoSL[ikappa] - 1;
219  }
220  }
221  }
222  }
223 
224  int dimL = denBK->gCurrentDim(index, sectorNL[ikappa],sectorTwoSL[ikappa],sectorIL[ikappa]); // dimL>0, checked at creation
225  int dimR = denBK->gCurrentDim(index+2,sectorNR[ikappa],sectorTwoSR[ikappa],sectorIR[ikappa]); // dimR>0, checked at creation
226  int phase = ((((sectorTwoSL[ikappa] + sectorTwoSR[ikappa] + ((sectorN2[ikappa]==1)?1:0) + ((sectorN1[ikappa]==1)?1:0))/2)%2)!=0)?-1:1;
227 
228  for (int casenr=0; casenr<nCases; casenr++){
229 
230  int dimM = denBK->gCurrentDim(index+1,NM,TwoJM[casenr],IM);
231  if (dimM>0){
232  double * BlockLeft = Tleft->gStorage(sectorNL[ikappa],sectorTwoSL[ikappa],sectorIL[ikappa],NM,TwoJM[casenr],IM);
233  double * BlockRight = Tright->gStorage(NM,TwoJM[casenr],IM,sectorNR[ikappa],sectorTwoSR[ikappa],sectorIR[ikappa]);
234 
235  double prefactor = gsl_sf_coupling_6j(sectorTwoSL[ikappa],sectorTwoSR[ikappa],sectorTwoJ[ikappa],((sectorN2[ikappa]==1)?1:0),((sectorN1[ikappa]==1)?1:0),TwoJM[casenr]) * sqrt((sectorTwoJ[ikappa]+1.0)*(TwoJM[casenr]+1)) * phase;
236 
237  double beta = (casenr==0)?0.0:1.0; //first time: set, other times: add.
238  char notrans = 'N';
239  dgemm_(&notrans,&notrans,&dimL,&dimR,&dimM,&prefactor,BlockLeft,&dimL,BlockRight,&dimM,&beta,storage+kappa2index[ikappa],&dimL);
240  } else {
241  if (casenr==0){ for (int cnt=0; cnt<dimL*dimR; cnt++){ storage[kappa2index[ikappa]+cnt] = 0.0; }}
242  }
243 
244  }
245 
246  delete [] TwoJM;
247 
248  }

252 double CheMPS2::Sobject::Split(TensorT * Tleft, TensorT * Tright, const int virtualdimensionD, const bool movingright, const bool change){
253 
254  //Get the number of central sectors
255  int nCenterSectors = 0;
256  for (int NM=denBK->gNmin(index+1); NM<=denBK->gNmax(index+1); NM++){
257  for (int TwoSM=denBK->gTwoSmin(index+1,NM); TwoSM<=denBK->gTwoSmax(index+1,NM); TwoSM+=2){
258  for (int IM=0; IM<denBK->getNumberOfIrreps(); IM++){
259  int dimM = denBK->gFCIdim(index+1,NM,TwoSM,IM); //FCIdim !! Whether possible hence.
260  if (dimM>0){
261  nCenterSectors++;
262  }
263  }
264  }
265  }
266 
267  //Get the labels of the central sectors
268  int * SplitSectNM = new int[nCenterSectors];
269  int * SplitSectTwoJM = new int[nCenterSectors];
270  int * SplitSectIM = new int[nCenterSectors];
271  nCenterSectors = 0;
272  for (int NM=denBK->gNmin(index+1); NM<=denBK->gNmax(index+1); NM++){
273  for (int TwoSM=denBK->gTwoSmin(index+1,NM); TwoSM<=denBK->gTwoSmax(index+1,NM); TwoSM+=2){
274  for (int IM=0; IM<denBK->getNumberOfIrreps(); IM++){
275  int dimM = denBK->gFCIdim(index+1,NM,TwoSM,IM); //FCIdim !! Whether possible hence.
276  if (dimM>0){
277  SplitSectNM[nCenterSectors] = NM;
278  SplitSectTwoJM[nCenterSectors] = TwoSM;
279  SplitSectIM[nCenterSectors] = IM;
280  nCenterSectors++;
281  }
282  }
283  }
284  }
285 
286  //SVD each sector --> Allocate memory
287  double ** Lambdas = new double*[nCenterSectors];
288  double ** Us = new double*[nCenterSectors];
289  double ** VTs = new double*[nCenterSectors];
290  int * CenterDims = new int[nCenterSectors];
291  int * DimLtotal = new int[nCenterSectors];
292  int * DimRtotal = new int[nCenterSectors];
293 
294  //PARALLEL
295  #pragma omp parallel for schedule(dynamic)
296  for (int iCenter=0; iCenter<nCenterSectors; iCenter++){
297 
298  //Determine left and right dimensions contributing to the center block iCenter
299  DimLtotal[iCenter] = 0;
300  for (int NL=SplitSectNM[iCenter]-2; NL<=SplitSectNM[iCenter]; NL++){
301  for (int TwoSL=SplitSectTwoJM[iCenter]-((NL==SplitSectNM[iCenter]-1)?1:0); TwoSL<SplitSectTwoJM[iCenter]+2; TwoSL+=2){
302  if (TwoSL>=0){
303  int IL = ((NL==SplitSectNM[iCenter]-1)?Irreps::directProd(Ilocal1,SplitSectIM[iCenter]):SplitSectIM[iCenter]);
304  int dimL = denBK->gCurrentDim(index,NL,TwoSL,IL);
305  if (dimL>0){
306  DimLtotal[iCenter] += dimL;
307  }
308  }
309  }
310  }
311  DimRtotal[iCenter] = 0;
312  for (int NR=SplitSectNM[iCenter]; NR<=SplitSectNM[iCenter]+2; NR++){
313  for (int TwoSR=SplitSectTwoJM[iCenter]-((NR==SplitSectNM[iCenter]+1)?1:0); TwoSR<SplitSectTwoJM[iCenter]+2; TwoSR+=2){
314  if (TwoSR>=0){
315  int IR = ((NR==SplitSectNM[iCenter]+1)?Irreps::directProd(Ilocal2,SplitSectIM[iCenter]):SplitSectIM[iCenter]);
316  int dimR = denBK->gCurrentDim(index+2,NR,TwoSR,IR);
317  if (dimR>0){
318  DimRtotal[iCenter] += dimR;
319  }
320  }
321  }
322  }
323  CenterDims[iCenter] = min(DimLtotal[iCenter],DimRtotal[iCenter]); //CenterDims contains the min. amount
324 
325  //Allocate memory to store the SVD in.
326  Lambdas[iCenter] = new double[CenterDims[iCenter]];
327  Us[iCenter] = new double[DimLtotal[iCenter]*CenterDims[iCenter]];
328  VTs[iCenter] = new double[CenterDims[iCenter]*DimRtotal[iCenter]];
329 
330  //Allocate memory to copy the different parts of the S-object. Use prefactor sqrt((2jR+1)/(2jM+1) * (2jM+1) * (2j+1)) W6J (-1)^(jL+jR+s1+s2) and sum over j.
331  if (CenterDims[iCenter]>0){
332  double * mem = new double[DimLtotal[iCenter]*DimRtotal[iCenter]];
333  for (int cnt=0; cnt<DimLtotal[iCenter]*DimRtotal[iCenter]; cnt++){ mem[cnt] = 0.0; }
334 
335  int dimLtotal2 = 0;
336  int dimRtotal2;
337  for (int NL=SplitSectNM[iCenter]-2; NL<=SplitSectNM[iCenter]; NL++){
338  for (int TwoSL=SplitSectTwoJM[iCenter]-((NL==SplitSectNM[iCenter]-1)?1:0); TwoSL<SplitSectTwoJM[iCenter]+2; TwoSL+=2){
339  if (TwoSL>=0){
340  int IL = ((NL==SplitSectNM[iCenter]-1)?Irreps::directProd(Ilocal1,SplitSectIM[iCenter]):SplitSectIM[iCenter]);
341  int dimL = denBK->gCurrentDim(index,NL,TwoSL,IL);
342  if (dimL>0){
343  dimRtotal2 = 0;
344  for (int NR=SplitSectNM[iCenter]; NR<=SplitSectNM[iCenter]+2; NR++){
345  for (int TwoSR=SplitSectTwoJM[iCenter]-((NR==SplitSectNM[iCenter]+1)?1:0); TwoSR<SplitSectTwoJM[iCenter]+2; TwoSR+=2){
346  if (TwoSR>=0){
347  int IR = ((NR==SplitSectNM[iCenter]+1)?Irreps::directProd(Ilocal2,SplitSectIM[iCenter]):SplitSectIM[iCenter]);
348  int dimR = denBK->gCurrentDim(index+2,NR,TwoSR,IR);
349  if (dimR>0){
350  //Loop over contributing TwoJ's
351  int TwoS2 = (NR==SplitSectNM[iCenter]+1)?1:0;
352  int TwoS1 = (NL==SplitSectNM[iCenter]-1)?1:0;
353  int fase = ((((TwoSL + TwoSR + TwoS1 + TwoS2)/2)%2)!=0)?-1:1;
354  for (int TwoJ = max(abs(TwoSR-TwoSL),abs(TwoS2-TwoS1)); TwoJ<=min(TwoS1+TwoS2,TwoSL+TwoSR); TwoJ+=2){
355  //calc prefactor
356  double prefact = gsl_sf_coupling_6j(TwoSL,TwoSR,TwoJ,TwoS2,TwoS1,SplitSectTwoJM[iCenter]) * sqrt((TwoJ+1.0)*(TwoSR+1)) * fase;
357 
358  //add them to mem.
359  double * Block = gStorage(NL,TwoSL,IL,SplitSectNM[iCenter]-NL,NR-SplitSectNM[iCenter],TwoJ,NR,TwoSR,IR);
360  for (int l=0; l<dimL; l++){
361  for (int r=0; r<dimR; r++){
362  mem[dimLtotal2 + l + DimLtotal[iCenter] * (dimRtotal2 + r)] += prefact * Block[l + dimL * r]; //+= because several TwoJ
363  }
364  }
365  }
366  dimRtotal2 += dimR;
367  }
368  }
369  }
370  }
371  dimLtotal2 += dimL;
372  }
373  }
374  }
375  }
376 
377  //Now mem contains sqrt((2jR+1)/(2jM+1)) * (TT)^{jM nM IM) --> SVD per central symmetry
378  char jobz = 'S'; //M x min(M,N) in U and min(M,N) x N in VT
379  int lwork = 3*CenterDims[iCenter] + max(max(DimLtotal[iCenter],DimRtotal[iCenter]),4*CenterDims[iCenter]*(CenterDims[iCenter]+1));
380  double * work = new double[lwork];
381  int * iwork = new int[8*CenterDims[iCenter]];
382  int info;
383 
384  //dgesdd is not thread-safe in every implementation (intel MKL is safe, Atlas is not safe)
385  #pragma omp critical
386  dgesdd_(&jobz, DimLtotal + iCenter, DimRtotal + iCenter, mem, DimLtotal + iCenter, Lambdas[iCenter], Us[iCenter], DimLtotal + iCenter, VTs[iCenter], CenterDims + iCenter, work, &lwork, iwork, &info);
387 
388  delete [] work;
389  delete [] iwork;
390  delete [] mem;
391  }
392  }
393 
394  double discardedWeight = 0.0; //Only if change==true; will the discardedWeight be meaningful and different from zero.
395 
396  //If change: determine new virtual dimensions.
397  if (change){
398 
399  int * NewDims = new int[nCenterSectors];
400  //First determine the total number of singular values
401  int totalDimSVD = 0;
402  int totalDimMPS = 0;
403  for (int iCenter=0; iCenter<nCenterSectors; iCenter++){
404  NewDims[iCenter] = CenterDims[iCenter];
405  totalDimSVD += NewDims[iCenter];
406  totalDimMPS += denBK->gCurrentDim(index+1,SplitSectNM[iCenter],SplitSectTwoJM[iCenter],SplitSectIM[iCenter]);
407  }
408 
409  //If larger then the required virtualdimensionD, new virtual dimensions will be set in NewDims.
410  if (totalDimSVD>virtualdimensionD){
411  //Copy them all in 1 array
412  double * values = new double[totalDimSVD];
413  totalDimSVD = 0;
414  int inc = 1;
415  for (int iCenter=0; iCenter<nCenterSectors; iCenter++){
416  if (NewDims[iCenter]>0){
417  dcopy_(NewDims + iCenter, Lambdas[iCenter], &inc, values + totalDimSVD, &inc);
418  totalDimSVD += NewDims[iCenter];
419  }
420  }
421 
422  //Sort them in decreasing order
423  char ID = 'D';
424  int info;
425  dlasrt_(&ID,&totalDimSVD,values,&info); //quicksort
426 
427  //The D+1'th value becomes the lower bound Schmidt value. Every value smaller than or equal to the D+1'th value is thrown out (hence Dactual <= Ddesired).
428  const double lowerBound = values[virtualdimensionD];
429  for (int iCenter=0; iCenter<nCenterSectors; iCenter++){
430  for (int cnt=0; cnt<NewDims[iCenter]; cnt++){
431  if (Lambdas[iCenter][cnt]<=lowerBound) NewDims[iCenter] = cnt;
432  }
433  }
434 
435  //Discarded weight
436  double totalSum = 0.0;
437  double discardedSum = 0.0;
438  for (int iCenter=0; iCenter<nCenterSectors; iCenter++){
439  for (int iLocal=0; iLocal<CenterDims[iCenter]; iLocal++){
440  double temp = (SplitSectTwoJM[iCenter]+1) * Lambdas[iCenter][iLocal] * Lambdas[iCenter][iLocal];
441  totalSum += temp;
442  if (Lambdas[iCenter][iLocal] <= lowerBound){ discardedSum += temp; }
443  }
444  }
445  discardedWeight = discardedSum / totalSum;
446 
447  //Clean-up
448  delete [] values;
449  }
450 
451  //Set NewDims only if relevant --> if totalDimSVD == totalDimMPS <= virtualdimensionD, no new symm sect virt D.
452  if ((totalDimSVD>virtualdimensionD) || (totalDimSVD!=totalDimMPS)){
453  for (int iCenter=0; iCenter<nCenterSectors; iCenter++){
454  denBK->SetDim(index+1,SplitSectNM[iCenter],SplitSectTwoJM[iCenter],SplitSectIM[iCenter],NewDims[iCenter]);
455  }
456  Tleft ->Reset();
457  Tright->Reset();
458  }
459 
460  delete [] NewDims;
461 
462  }
463 
464  //copy first gCurrentDimM per central symmetry sector to the relevant parts.
465  //PARALLEL
466  #pragma omp parallel for schedule(dynamic)
467  for (int iCenter=0; iCenter<nCenterSectors; iCenter++){
468  int dimM = denBK->gCurrentDim(index+1,SplitSectNM[iCenter],SplitSectTwoJM[iCenter],SplitSectIM[iCenter]);
469  if (dimM>0){
470  //U-part: copy
471  int dimLtotal2 = 0;
472  for (int NL=SplitSectNM[iCenter]-2; NL<=SplitSectNM[iCenter]; NL++){
473  for (int TwoSL=SplitSectTwoJM[iCenter]-((NL==SplitSectNM[iCenter]-1)?1:0); TwoSL<SplitSectTwoJM[iCenter]+2; TwoSL+=2){
474  if (TwoSL>=0){
475  int IL = ((NL==SplitSectNM[iCenter]-1)?Irreps::directProd(Ilocal1,SplitSectIM[iCenter]):SplitSectIM[iCenter]);
