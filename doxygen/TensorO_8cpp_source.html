177 void CheMPS2::TensorO::updateRight(const int ikappa, TensorT * denTup, TensorT * denTdown){
178 
179  int dimRdown = denBK->gCurrentDim(index, sectorN1[ikappa], sectorTwoS1[ikappa], sectorI1[ikappa]);
180  int dimRup = denBKup->gCurrentDim(index, sectorN1[ikappa], sectorTwoS1[ikappa], sectorI1[ikappa]);
181 
182  for (int geval=0; geval<4; geval++){
183  int IL, TwoSL, NL;
184  switch(geval){
185  case 0:
186  NL = sectorN1[ikappa];
187  TwoSL = sectorTwoS1[ikappa];
188  IL = sectorI1[ikappa];
189  break;
190  case 1:
191  NL = sectorN1[ikappa]-2;
192  TwoSL = sectorTwoS1[ikappa];
193  IL = sectorI1[ikappa];
194  break;
195  case 2:
196  NL = sectorN1[ikappa]-1;
197  TwoSL = sectorTwoS1[ikappa]-1;
198  IL = Irreps::directProd( sectorI1[ikappa] , denBK->gIrrep(index-1) );
199  break;
200  case 3:
201  NL = sectorN1[ikappa]-1;
202  TwoSL = sectorTwoS1[ikappa]+1;
203  IL = Irreps::directProd( sectorI1[ikappa] , denBK->gIrrep(index-1) );
204  break;
205  }
206 
207  int dimLup = denBKup->gCurrentDim(index-1, NL, TwoSL, IL);
208  int dimLdown = denBK->gCurrentDim(index-1, NL, TwoSL, IL);
209  if ((dimLup>0) && (dimLdown>0) && (dimLup==dimLdown)){
210 
211  double alpha = 1.0;
212  double beta = 1.0; //add
213  char trans = 'T';
214  char notrans = 'N';
215  double * Tup = denTup->gStorage(NL,TwoSL,IL,sectorN1[ikappa],sectorTwoS1[ikappa],sectorI1[ikappa]);
216  double * Tdown = denTdown->gStorage(NL,TwoSL,IL,sectorN1[ikappa],sectorTwoS1[ikappa],sectorI1[ikappa]);
217  dgemm_(&trans,&notrans,&dimRdown,&dimRup,&dimLup,&alpha,Tdown,&dimLdown,Tup,&dimLup,&beta,storage+gKappa2index(ikappa),&dimRdown);
218 
219  }
220  }
