41 int CheMPS2::CASSCF::orbital_rotation_size( const int maxlinsize, const int CUTOFF, const DMRGSCFindices * idx ){
42 
43  const int n_irreps = idx->getNirreps();
44 
45  int max_norb_ok = 0;
46  for ( int irrep = 0; irrep < n_irreps; irrep++ ){
47  const int NORB = idx->getNORB( irrep );
48  if (( NORB > max_norb_ok ) && ( NORB <= CUTOFF )){ max_norb_ok = NORB; }
49  }
50  if ( max_norb_ok == maxlinsize ){ return maxlinsize; }
51 
52  const int factor = (int) ( ceil( ( 1.0 * maxlinsize ) / CUTOFF ) + 0.1 );
53  const int newlinsize = (int) ( ceil( ( 1.0 * maxlinsize ) / factor ) + 0.1 );
54  const int block_size = max( newlinsize, max_norb_ok ); // If a particular index can be rotated at once...
55 
56  cout << "DMRGSCF info: The max. # orb per irrep = " << maxlinsize << endl;
57  cout << " The size cutoff for 2-body transfo = " << CUTOFF << endl;
58  cout << " The max. # orb per irrep <= cutoff = " << max_norb_ok << endl;
59  cout << " The blocksize for piecewise tfo = " << block_size << endl;
60 
61  return block_size;
62

65 double CheMPS2::CASSCF::solve(const int Nelectrons, const int TwoS, const int Irrep, ConvergenceScheme * OptScheme, const int rootNum, DMRGSCFoptions * theDMRGSCFoptions){
66 
67  const int num_elec = Nelectrons - 2 * iHandler->getNOCCsum();
68  assert( num_elec >= 0 );
69 
70  //Convergence variables
71  double gradNorm = 1.0;
72  double updateNorm = 1.0;
73  double * gradient = new double[unitary->getNumVariablesX()];
74  for (int cnt=0; cnt<unitary->getNumVariablesX(); cnt++){ gradient[cnt] = 0.0; }
75  double * theDIISparameterVector = NULL;
76  const int linsizeDIIS = iHandler->getROTparamsize();
77  double Energy = 1e8;
78 
79  //The CheMPS2::Problem for the inner DMRG calculation
80  Hamiltonian * HamDMRG = new Hamiltonian(nOrbDMRG, SymmInfo.getGroupNumber(), iHandler->getIrrepOfEachDMRGorbital());
81  Problem * Prob = new Problem(HamDMRG, TwoS, num_elec, Irrep);
82  Prob->SetupReorderD2h(); //Doesn't matter if the group isn't D2h, Prob checks it.
83 
84  //Determine the maximum NORB(irrep) and the max_block_size for the ERI orbital rotation
85  const int maxlinsize = iHandler->getNORBmax();
86  const bool block_wise = (( maxlinsize <= CheMPS2::DMRGSCF_maxlinsizeCutoff ) ? false : true ); //Only if bigger, do we want to work blockwise
87  const int block_size = (( block_wise ) ? orbital_rotation_size( maxlinsize, CheMPS2::DMRGSCF_maxlinsizeCutoff, iHandler ) : maxlinsize );
88 
89  //Allocate 2-body rotation memory: One array is approx (block_size/273.0)^4 * 42 GiB --> [block_size=100 --> 750 MB]
90  const int block_size_power4 = block_size * block_size * block_size * block_size; //Note that 273**4 overfloats the 32 bit integer!!!
91  const int tot_dmrg_power4 = nOrbDMRG * nOrbDMRG * nOrbDMRG * nOrbDMRG;
92  const int work_mem_size = max( max( block_size_power4 , maxlinsize * maxlinsize * 4 ) , tot_dmrg_power4 ); //For (ERI rotation, update unitary, block diagonalize, orbital localization)
93  double * mem1 = new double[ work_mem_size ];
94  double * mem2 = new double[ work_mem_size ];
95  double * mem3 = NULL;
96  if ( block_wise ){ mem3 = new double[ block_size_power4 ]; }
97 
98  //The two-body rotator and Edmiston-Ruedenberg active space localizer
99  DMRGSCFVmatRotations theRotator(HamOrig, iHandler);
100  EdmistonRuedenberg * theLocalizer = NULL;
101  if (theDMRGSCFoptions->getWhichActiveSpace()==2){ theLocalizer = new EdmistonRuedenberg(HamDMRG); }
102 
103  //Load unitary from disk
104  if (theDMRGSCFoptions->getStoreUnitary()){
105  struct stat stFileInfo;
106  int intStat = stat((theDMRGSCFoptions->getUnitaryStorageName()).c_str(),&stFileInfo);
107  if (intStat==0){ unitary->loadU(theDMRGSCFoptions->getUnitaryStorageName()); }
108  }
109 
110  //Load DIIS from disk
111  if ((theDMRGSCFoptions->getDoDIIS()) && (theDMRGSCFoptions->getStoreDIIS())){
112  struct stat stFileInfo;
113  int intStat = stat((theDMRGSCFoptions->getDIISStorageName()).c_str(),&stFileInfo);
114  if (intStat==0){
115  if (theDIIS == NULL){
116  theDIIS = new DIIS(linsizeDIIS, unitary->getNumVariablesX(), theDMRGSCFoptions->getNumDIISVecs());
117  theDIISparameterVector = new double[ linsizeDIIS ];
118  }
119  theDIIS->loadDIIS(theDMRGSCFoptions->getDIISStorageName());
120  }
121  }
122 
123  int nIterations = 0;
124 
125  /*******************************
126  *** Actual DMRGSCF loops ***
127  *******************************/
128  while ((gradNorm > theDMRGSCFoptions->getGradientThreshold()) && (nIterations < theDMRGSCFoptions->getMaxIterations())){
129 
130  nIterations++;
131 
132  //Update the unitary transformation
133  if (unitary->getNumVariablesX() > 0){
134 
135  unitary->updateUnitary(mem1, mem2, gradient, true, true); //multiply = compact = true
136 
137  if ((theDMRGSCFoptions->getDoDIIS()) && (updateNorm <= theDMRGSCFoptions->getDIISGradientBranch())){
138  if (theDMRGSCFoptions->getWhichActiveSpace()==1){
139  cout << "DMRGSCF::solve : DIIS has started. Active space not rotated to NOs anymore!" << endl;
140  }
141  if (theDMRGSCFoptions->getWhichActiveSpace()==2){
142  cout << "DMRGSCF::solve : DIIS has started. Active space not rotated to localized orbitals anymore!" << endl;
143  }
144  if (theDIIS == NULL){
145  theDIIS = new DIIS( linsizeDIIS, unitary->getNumVariablesX(), theDMRGSCFoptions->getNumDIISVecs());
146  theDIISparameterVector = new double[ linsizeDIIS ];
147  unitary->makeSureAllBlocksDetOne(mem1, mem2);
148  }
149  unitary->getLog(theDIISparameterVector, mem1, mem2);
150  theDIIS->appendNew(gradient, theDIISparameterVector);
151  theDIIS->calculateParam(theDIISparameterVector);
152  unitary->updateUnitary(mem1, mem2, theDIISparameterVector, false, false); //multiply = compact = false
153  }
154 
155  }
156  if ((theDMRGSCFoptions->getStoreUnitary()) && (gradNorm!=1.0)){ unitary->saveU( theDMRGSCFoptions->getUnitaryStorageName() ); }
157  if ((theDMRGSCFoptions->getStoreDIIS()) && (updateNorm!=1.0) && (theDIIS!=NULL)){ theDIIS->saveDIIS( theDMRGSCFoptions->getDIISStorageName() ); }
158 
159  //Fill HamDMRG
160  buildQmatOCC();
161  buildTmatrix();
162  fillConstAndTmatDMRG(HamDMRG);
163  if ( block_wise ){ theRotator.fillVmatDMRGBlockWise(HamDMRG, unitary, mem1, mem2, mem3, block_size); }
164  else { theRotator.fillVmatDMRG(HamDMRG, unitary, mem1, mem2); }
165 
166  //Localize the active space and reorder the orbitals within each irrep based on the exchange matrix
167  if ((theDMRGSCFoptions->getWhichActiveSpace()==2) && (theDIIS==NULL)){ //When the DIIS has started: stop
168  theLocalizer->Optimize(mem1, mem2, theDMRGSCFoptions->getStartLocRandom()); //Default EDMISTONRUED_gradThreshold and EDMISTONRUED_maxIter used
169  theLocalizer->FiedlerExchange(maxlinsize, mem1, mem2);
170  fillLocalizedOrbitalRotations(theLocalizer->getUnitary(), iHandler, mem1);
171  unitary->rotateActiveSpaceVectors(mem1, mem2);
172  buildQmatOCC(); //With an updated unitary, the Qocc, Tmat, and HamDMRG objects need to be updated as well.
173  buildTmatrix();
174  fillConstAndTmatDMRG(HamDMRG);
175  if ( block_wise ){ theRotator.fillVmatDMRGBlockWise(HamDMRG, unitary, mem1, mem2, mem3, block_size); }
176  else { theRotator.fillVmatDMRG(HamDMRG, unitary, mem1, mem2); }
177  cout << "DMRGSCF::solve : Rotated the active space to localized orbitals, sorted according to the exchange matrix." << endl;
178  }
179 
180  if (( OptScheme == NULL ) && ( rootNum == 1 )){ // Do FCI, and calculate the 2DM
181 
182  const int nalpha = ( num_elec + TwoS ) / 2;
183  const int nbeta = ( num_elec - TwoS ) / 2;
184  const double workmem = 1000.0; // 1GB
185  const int verbose = 2;
186  CheMPS2::FCI * theFCI = new CheMPS2::FCI( HamDMRG, nalpha, nbeta, Irrep, workmem, verbose );
187  double * inoutput = new double[ theFCI->getVecLength(0) ];
188  theFCI->ClearVector( theFCI->getVecLength(0), inoutput );
189  inoutput[ theFCI->LowestEnergyDeterminant() ] = 1.0;
190  Energy = theFCI->GSDavidson( inoutput );
191  theFCI->Fill2RDM( inoutput, DMRG2DM );
192  delete theFCI;
193  delete [] inoutput;
194 
195  } else { //Do the DMRG sweeps, and calculate the 2DM
196 
197  for ( int cnt = 0; cnt < tot_dmrg_power4; cnt++ ){ DMRG2DM[ cnt ] = 0.0; } //Clear the 2-RDM (to allow for state-averaged calculations)
198  DMRG * theDMRG = new DMRG(Prob, OptScheme);
199  for (int state = 0; state < rootNum; state++){
200  if (state > 0){ theDMRG->newExcitation( fabs( Energy ) ); }
201  Energy = theDMRG->Solve();
202  if ( theDMRGSCFoptions->getStateAveraging() ){ // When SA-DMRGSCF: 2DM += current 2DM
203  theDMRG->calc2DMandCorrelations();
204  copy2DMover( theDMRG->get2DM(), nOrbDMRG, DMRG2DM );
205  }
206  if ((state == 0) && (rootNum > 1)){ theDMRG->activateExcitations( rootNum-1 ); }
207  }
208  if ( !(theDMRGSCFoptions->getStateAveraging()) ){ // When SS-DMRGSCF: 2DM += last 2DM
209  theDMRG->calc2DMandCorrelations();
210  copy2DMover( theDMRG->get2DM(), nOrbDMRG, DMRG2DM );
211  }
212  if (theDMRGSCFoptions->getDumpCorrelations()){ theDMRG->getCorrelations()->Print(); } // Correlations have been calculated in the loop (SA) or outside of the loop (SS)
213  if (CheMPS2::DMRG_storeMpsOnDisk){ theDMRG->deleteStoredMPS(); }
214  if (CheMPS2::DMRG_storeRenormOptrOnDisk){ theDMRG->deleteStoredOperators(); }
215  delete theDMRG;
216  if ((theDMRGSCFoptions->getStateAveraging()) && (rootNum > 1)){
217  const double averagingfactor = 1.0 / rootNum;
218  for ( int cnt = 0; cnt < tot_dmrg_power4; cnt++ ){ DMRG2DM[ cnt ] *= averagingfactor; }
219  }
220 
221  }
222  setDMRG1DM(num_elec, nOrbDMRG, DMRG1DM, DMRG2DM);
223 
224  //Possibly rotate the active space to the natural orbitals
225  if ((theDMRGSCFoptions->getWhichActiveSpace()==1) && (theDIIS==NULL)){ //When the DIIS has started: stop
226  copy_active( DMRG1DM, theQmatWORK, iHandler, true );
227  block_diagonalize( 'A', theQmatWORK, unitary, mem1, mem2, iHandler, true, DMRG2DM ); // Unitary is updated and DMRG2DM rotated
228  setDMRG1DM( num_elec, nOrbDMRG, DMRG1DM, DMRG2DM );
229  buildQmatOCC(); //With an updated unitary, the Qocc and Tmat matrices need to be updated as well.
230  buildTmatrix();
231  cout << "DMRGSCF::solve : Rotated the active space to natural orbitals, sorted according to the NOON." << endl;
232  }
233 
234  //Calculate the matrix elements needed to calculate the gradient and hessian
235  buildQmatACT();
236  if ( block_wise ){ theRotator.fillRotatedTEIBlockWise(theRotatedTEI, unitary, mem1, mem2, mem3, block_size); }
237  else { theRotator.fillRotatedTEI( theRotatedTEI, unitary, mem1, mem2 ); }
238  buildFmat( theFmatrix, theTmatrix, theQmatOCC, theQmatACT, iHandler, theRotatedTEI, DMRG2DM, DMRG1DM);
239  buildWtilde(wmattilde, theTmatrix, theQmatOCC, theQmatACT, iHandler, theRotatedTEI, DMRG2DM, DMRG1DM);
240 
241  //Calculate the gradient, hessian and corresponding update. On return, gradient contains the rescaled gradient == the update.
242  augmentedHessianNR(theFmatrix, wmattilde, iHandler, unitary, gradient, &updateNorm, &gradNorm);
243 
244  }
245 
246  delete [] mem1;
247  delete [] mem2;
248  if ( block_wise ){ delete [] mem3; }
249 
250  delete Prob;
251  delete HamDMRG;
252  delete [] gradient;
253  if (theDIISparameterVector!=NULL){ delete [] theDIISparameterVector; }
254  if (theLocalizer!=NULL){ delete theLocalizer; }
255 
256  return Energy;
257 
258 }

260 void CheMPS2::CASSCF::augmentedHessianNR(const DMRGSCFmatrix * localFmat, const DMRGSCFwtilde * localwtilde, const DMRGSCFindices * localIdx, const DMRGSCFunitary * localUmat, double * theupdate, double * updateNorm, double * gradNorm){
261 
262  /* A good read to understand
263  (1) how the augmented Hessian arises from a rational function optimization
264  (2) where the parameter lambda in Eq. (22) of Yanai, IJQC 109, 2178-2190 (2009) comes from
265  (3) why the smallest algebraic eigenvalue + corresponding eigenvector should be retained for minimizations
266  Banerjee, Adams, Simons, Shepard, "Search for stationary points on surfaces",
267  J. Phys. Chem. 1985, volume 89, pages 52-57, doi:10.1021/j100247a015 */
268 
269  //Calculate the gradient
270  const int x_linearlength = localUmat->getNumVariablesX();
271  gradNorm[0] = calcGradient(localFmat, localIdx, localUmat, theupdate);
272 
273  //Calculate the Hessian
274  int dim = x_linearlength + 1;
275  int size = dim * dim;
276  double * hessian = new double[size];
277  calcHessian(localFmat, localwtilde, localIdx, localUmat, hessian, dim);
278 
279  //Augment the gradient into the Hessian matrix
280  for (int cnt=0; cnt<x_linearlength; cnt++){
281  hessian[cnt + x_linearlength*dim] = theupdate[cnt];
282  hessian[x_linearlength + dim*cnt] = theupdate[cnt];
283  }
284  hessian[x_linearlength + dim*x_linearlength] = 0.0;
285 
286  //Find the lowest eigenvalue and corresponding eigenvector of the augmented hessian
287  {
288  Davidson deBoskabouter( dim, CheMPS2::DAVIDSON_NUM_VEC,
289  CheMPS2::DAVIDSON_NUM_VEC_KEEP,
290  CheMPS2::DAVIDSON_FCI_RTOL,
291  CheMPS2::DAVIDSON_PRECOND_CUTOFF, false ); // No debug printing
292  double ** whichpointers = new double*[2];
293 
294  char instruction = deBoskabouter.FetchInstruction( whichpointers );
295  assert( instruction == 'A' );
296  for (int cnt = 0; cnt < dim; cnt++){ whichpointers[1][cnt] = hessian[cnt*(1+dim)]; } // Preconditioner = diagonal elements of the augmented Hessian
297  for (int cnt = 0; cnt < x_linearlength; cnt++){ // Initial guess = [ -gradient / diag(hessian) , 1 ]
298  const double denom = ( whichpointers[1][cnt] > CheMPS2::DAVIDSON_PRECOND_CUTOFF ) ? whichpointers[1][cnt] : CheMPS2::DAVIDSON_PRECOND_CUTOFF;
299  whichpointers[0][cnt] = - theupdate[cnt] / denom;
300  }
301  whichpointers[0][x_linearlength] = 1.0;
302 
303  instruction = deBoskabouter.FetchInstruction( whichpointers );
304  while ( instruction == 'B' ){
305  char notrans = 'N';
306  int one = 1;
307  double alpha = 1.0;
308  double beta = 0.0;
309  dgemm_(&notrans, &notrans, &dim, &one, &dim, &alpha, hessian, &dim, whichpointers[0], &dim, &beta, whichpointers[1], &dim);
310  instruction = deBoskabouter.FetchInstruction( whichpointers );
311  }
312 
313  assert( instruction == 'C' );
314  double scalar = 1.0 / whichpointers[0][x_linearlength];
315  cout << "DMRGSCF::augmentedHessianNR : Augmented Hessian update found with " << deBoskabouter.GetNumMultiplications() << " Davidson iterations." << endl;
316  if (CheMPS2::DMRGSCF_debugPrint){
317  cout << "DMRGSCF::augmentedHessianNR : Lowest eigenvalue = " << whichpointers[1][0] << endl;
318  cout << "DMRGSCF::augmentedHessianNR : The last number of the eigenvector (which will be rescaled to one) = " << scalar << endl;
319  }
320  for (int cnt = 0; cnt < x_linearlength; cnt++){ theupdate[cnt] = scalar * whichpointers[0][cnt]; }
321  delete [] whichpointers;
322  }
323 
324  //Calculate the update norm
325  updateNorm[0] = 0.0;
326  for (int cnt = 0; cnt < x_linearlength; cnt++){ updateNorm[0] += theupdate[cnt] * theupdate[cnt]; }
327  updateNorm[0] = sqrt(updateNorm[0]);
328  cout << "DMRGSCF::augmentedHessianNR : Norm of the update = " << updateNorm[0] << endl;
329 
330  delete [] hessian;
331 
332 }

334 double CheMPS2::CASSCF::calcGradient(const DMRGSCFmatrix * localFmat, const DMRGSCFindices * localIdx, const DMRGSCFunitary * localUmat, double * gradient){
335 
336  for (int cnt=0; cnt<localUmat->getNumVariablesX(); cnt++){
337  const int index1 = localUmat->getFirstIndex(cnt);
338  const int index2 = localUmat->getSecondIndex(cnt);
339  // irrep1 == irrep2 due to construction DMRGSCFunitary
340  const int irrep = localIdx->getOrbitalIrrep( index1 );
341  const int shift = localIdx->getOrigNOCCstart( irrep );
342  const int relIndex1 = index1 - shift;
343  const int relIndex2 = index2 - shift;
344  gradient[cnt] = 2 * ( localFmat->get( irrep, relIndex1, relIndex2 ) - localFmat->get( irrep, relIndex2, relIndex1 ) );
345  }
346 
347  double gradNorm = 0.0;
348  for (int cnt=0; cnt<localUmat->getNumVariablesX(); cnt++){ gradNorm += gradient[cnt] * gradient[cnt]; }
349  gradNorm = sqrt(gradNorm);
350  cout << "DMRGSCF::calcGradient : Norm of the gradient = " << gradNorm << endl;
351  return gradNorm;
352 
353 }

355 void CheMPS2::CASSCF::calcHessian(const DMRGSCFmatrix * localFmat, const DMRGSCFwtilde * localwtilde, const DMRGSCFindices * localIdx, const DMRGSCFunitary * localUmat, double * hessian, const int rowjump){
356 
357  const int lindim = ( localUmat->getNumVariablesX() * ( localUmat->getNumVariablesX() + 1 ))/2;
358 
359  #pragma omp parallel for schedule(static)
360  for (int count=0; count<lindim; count++){
361 
362  int col = 1;
363  while ( (col*(col+1))/2 <= count ) col++;
364  col -= 1;
365  int row = count - (col*(col+1))/2;
366 
367  const int p_index = localUmat->getFirstIndex(row);
368  const int q_index = localUmat->getSecondIndex(row);
369  // irrep_p == irrep_q due to construction DMRGSCFunitary
370  const int irrep_pq = localIdx->getOrbitalIrrep( p_index );
371 
372  const int r_index = localUmat->getFirstIndex(col);
373  const int s_index = localUmat->getSecondIndex(col);
374  // irrep_r == irrep_s due to construction DMRGSCFunitary
375  const int irrep_rs = localIdx->getOrbitalIrrep( r_index );
376 
377  const int rel_p_index = p_index - localIdx->getOrigNOCCstart( irrep_pq );
378  const int rel_q_index = q_index - localIdx->getOrigNOCCstart( irrep_pq );
379  const int rel_r_index = r_index - localIdx->getOrigNOCCstart( irrep_rs );
380  const int rel_s_index = s_index - localIdx->getOrigNOCCstart( irrep_rs );
381 
382  hessian[row + rowjump * col] = Wmat(localFmat, localwtilde, localIdx, irrep_pq, irrep_rs, rel_p_index, rel_q_index, rel_r_index, rel_s_index)
383  - Wmat(localFmat, localwtilde, localIdx, irrep_pq, irrep_rs, rel_q_index, rel_p_index, rel_r_index, rel_s_index)
384  - Wmat(localFmat, localwtilde, localIdx, irrep_pq, irrep_rs, rel_p_index, rel_q_index, rel_s_index, rel_r_index)
385  + Wmat(localFmat, localwtilde, localIdx, irrep_pq, irrep_rs, rel_q_index, rel_p_index, rel_s_index, rel_r_index);
386  hessian[col + rowjump * row] = hessian[row + rowjump * col];
387 
388  }
389 
390 }

392 double CheMPS2::CASSCF::Wmat(const DMRGSCFmatrix * localFmat, const DMRGSCFwtilde * localwtilde, const DMRGSCFindices * localIdx, const int irrep_pq, const int irrep_rs, const int relindexP, const int relindexQ, const int relindexR, const int relindexS){
393 
394  double value = 0.0;
395 
396  if ( ( irrep_pq == irrep_rs ) && ( relindexQ == relindexR ) ) {
397  value = localFmat->get(irrep_pq, relindexP, relindexS) + localFmat->get(irrep_pq, relindexS, relindexP);
398  }
399 
400  if (relindexP >= localIdx->getNOCC(irrep_pq) + localIdx->getNDMRG(irrep_pq)){ return value; }

402 
403  value += localwtilde->get( irrep_pq, irrep_rs, relindexP, relindexQ, relindexR, relindexS );
404  return value;
405

408 void CheMPS2::CASSCF::buildWtilde(DMRGSCFwtilde * localwtilde, const DMRGSCFmatrix * localTmat, const DMRGSCFmatrix * localJKocc, const DMRGSCFmatrix * localJKact, const DMRGSCFindices * localIdx, const DMRGSCFintegrals * theInts, double * local2DM, double * local1DM){
409 
410  localwtilde->clear();
411  const int numIrreps = localIdx->getNirreps();
412  const int totOrbDMRG = localIdx->getDMRGcumulative( numIrreps );
413  for (int irrep_pq = 0; irrep_pq < numIrreps; irrep_pq++){
414 
415  const int NumOCCpq = localIdx->getNOCC( irrep_pq );
416  const int NumDMRGpq = localIdx->getNDMRG( irrep_pq );
417  const int NumORBpq = localIdx->getNORB( irrep_pq );
418  const int NumCOREpq = NumOCCpq + NumDMRGpq;
419 
420  //If irrep_pq == irrep_rs and P == R occupied --> QS only active or virtual
421  #pragma omp parallel for schedule(static)
422  for (int relindexP = 0; relindexP < NumOCCpq; relindexP++){
