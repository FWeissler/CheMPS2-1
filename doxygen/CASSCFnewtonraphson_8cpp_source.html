56 double CheMPS2::CASSCF::solve( const int Nelectrons, const int TwoS, const int Irrep, ConvergenceScheme * OptScheme, const int rootNum, DMRGSCFoptions * scf_options ){
57 
58  #ifdef CHEMPS2_MPI_COMPILATION
59  const bool am_i_master = ( MPIchemps2::mpi_rank() == MPI_CHEMPS2_MASTER );
60  #else
61  const bool am_i_master = true;
62  #endif
63 
64  const int num_elec = Nelectrons - 2 * iHandler->getNOCCsum();
65  assert( num_elec >= 0 );
66  assert(( OptScheme != NULL ) || (( OptScheme == NULL ) && ( rootNum == 1 )));
67 
68  // Convergence variables
69  double gradNorm = 1.0;
70  double updateNorm = 1.0;
71  double * gradient = NULL;
72  if ( am_i_master ){
73  gradient = new double[ unitary->getNumVariablesX() ];
74  for ( int cnt = 0; cnt < unitary->getNumVariablesX(); cnt++ ){ gradient[ cnt ] = 0.0; }
75  }
76  double * diis_vec = NULL;
77  double Energy = 1e8;
78 
79  // The CheMPS2::Problem for the inner DMRG calculation
80  Hamiltonian * HamDMRG = new Hamiltonian( nOrbDMRG, SymmInfo.getGroupNumber(), iHandler->getIrrepOfEachDMRGorbital() );
81  Problem * Prob = new Problem( HamDMRG, TwoS, num_elec, Irrep );
82  Prob->SetupReorderD2h(); // Doesn't matter if the group isn't D2h, Prob checks it.
83 
84  // Determine the maximum NORB( irrep ) and the max_block_size for the ERI orbital rotation
85  const int maxlinsize = iHandler->getNORBmax();
86  const long long fullsize = ((long long) maxlinsize ) * ((long long) maxlinsize ) * ((long long) maxlinsize ) * ((long long) maxlinsize );
87  const string tmp_filename = tmp_folder + "/" + CheMPS2::DMRGSCF_eri_storage_name;
88  const int dmrgsize_power4 = nOrbDMRG * nOrbDMRG * nOrbDMRG * nOrbDMRG;
89  // For ( ERI rotation, update unitary, block diagonalize, orbital localization )
90  DMRGSCFintegrals * theRotatedTEI = new DMRGSCFintegrals( iHandler );
91  DMRGSCFwtilde * wmattilde = new DMRGSCFwtilde( iHandler );
92  const int temp_work_size = (( fullsize > CheMPS2::DMRGSCF_max_mem_eri_tfo ) ? CheMPS2::DMRGSCF_max_mem_eri_tfo : fullsize );
93  const int work_mem_size = max( max( temp_work_size , maxlinsize * maxlinsize * 4 ) , dmrgsize_power4 );
94  double * mem1 = new double[ work_mem_size ];
95  double * mem2 = new double[ work_mem_size ];
96 
97  // Only MASTER process has Edmiston-Ruedenberg active space localizer
98  EdmistonRuedenberg * theLocalizer = NULL;
99  if (( scf_options->getWhichActiveSpace() == 2 ) && ( am_i_master )){ theLocalizer = new EdmistonRuedenberg( HamDMRG->getVmat(), iHandler->getGroupNumber() ); }
100 
101  // Load unitary from disk
102  if ( scf_options->getStoreUnitary() ){
103  if ( am_i_master ){
104  struct stat file_info;
105  int master_stat = stat( (scf_options->getUnitaryStorageName()).c_str(), &file_info );
106  if ( master_stat == 0 ){ unitary->loadU( scf_options->getUnitaryStorageName() ); }
107  }
108  #ifdef CHEMPS2_MPI_COMPILATION
109  unitary->broadcast( MPI_CHEMPS2_MASTER ); // Unitary was set to the identity at construction
110  #endif
111  }
112 
113  // Only master has DIIS; load DIIS from disk
114  DIIS * diis = NULL;
115  if (( scf_options->getDoDIIS() ) && ( scf_options->getStoreDIIS() ) && ( am_i_master )){
116  struct stat file_info;
117  int master_stat = stat( (scf_options->getDIISStorageName()).c_str(), &file_info );
118  if ( master_stat == 0 ){
119  const int diis_vec_size = iHandler->getROTparamsize();
120  diis = new DIIS( diis_vec_size, unitary->getNumVariablesX(), scf_options->getNumDIISVecs() );
121  diis->loadDIIS( scf_options->getDIISStorageName() );
122  diis_vec = new double[ diis_vec_size ];
123  }
124  }
125 
126  int nIterations = 0;
127 
128  /*******************************
129  *** Actual DMRGSCF loops ***
130  *******************************/
131  while (( gradNorm > scf_options->getGradientThreshold() ) && ( nIterations < scf_options->getMaxIterations() )){
132 
133  nIterations++;
134 
135  // Update the unitary transformation
136  if (( unitary->getNumVariablesX() > 0 ) && ( am_i_master )){
137 
138  unitary->updateUnitary( mem1, mem2, gradient, true, true ); //multiply = compact = true
139 
140  if (( scf_options->getDoDIIS() ) && ( updateNorm <= scf_options->getDIISGradientBranch() )){
141  if ( scf_options->getWhichActiveSpace() == 1 ){
142  cout << "DMRGSCF::solve : DIIS has started. Active space not rotated to NOs anymore!" << endl;
143  }
144  if ( scf_options->getWhichActiveSpace() == 2 ){
145  cout << "DMRGSCF::solve : DIIS has started. Active space not rotated to localized orbitals anymore!" << endl;
146  }
147  if ( diis == NULL ){
148  const int diis_vec_size = iHandler->getROTparamsize();
149  diis = new DIIS( diis_vec_size, unitary->getNumVariablesX(), scf_options->getNumDIISVecs() );
150  diis_vec = new double[ diis_vec_size ];
151  unitary->makeSureAllBlocksDetOne( mem1, mem2 );
152  }
153  unitary->getLog( diis_vec, mem1, mem2 );
154  diis->appendNew( gradient, diis_vec );
155  diis->calculateParam( diis_vec );
156  unitary->updateUnitary( mem1, mem2, diis_vec, false, false ); //multiply = compact = false
157  }
158  }
159  if (( am_i_master ) && ( scf_options->getStoreUnitary() ) && ( gradNorm != 1.0 )){ unitary->saveU( scf_options->getUnitaryStorageName() ); }
160  if (( am_i_master ) && ( scf_options->getStoreDIIS() ) && ( updateNorm != 1.0 ) && ( diis != NULL )){ diis->saveDIIS( scf_options->getDIISStorageName() ); }
161  int master_diis = (( diis != NULL ) ? 1 : 0 );
162  #ifdef CHEMPS2_MPI_COMPILATION
163  MPIchemps2::broadcast_array_int( &master_diis, 1, MPI_CHEMPS2_MASTER );
164  unitary->broadcast( MPI_CHEMPS2_MASTER );
165  #endif
166 
167  // Fill HamDMRG
168  buildTmatrix();
169  buildQmatOCC();
170  fillConstAndTmatDMRG( HamDMRG );
171  if ( am_i_master ){
172  DMRGSCFrotations::rotate( VMAT_ORIG, HamDMRG->getVmat(), NULL, 'A', 'A', 'A', 'A', iHandler, unitary, mem1, mem2, work_mem_size, tmp_filename );
173  }
174  #ifdef CHEMPS2_MPI_COMPILATION
175  HamDMRG->getVmat()->broadcast( MPI_CHEMPS2_MASTER );
176  #endif
177 
178  // Localize the active space and reorder the orbitals within each irrep based on the exchange matrix
179  if (( scf_options->getWhichActiveSpace() == 2 ) && ( master_diis == 0 )){ // When the DIIS has started: stop
180  if ( am_i_master ){
181  theLocalizer->Optimize(mem1, mem2, scf_options->getStartLocRandom()); // Default EDMISTONRUED_gradThreshold and EDMISTONRUED_maxIter used
182  theLocalizer->FiedlerExchange(maxlinsize, mem1, mem2);
183  fillLocalizedOrbitalRotations(theLocalizer->getUnitary(), iHandler, mem1);
184  unitary->rotateActiveSpaceVectors(mem1, mem2);
185  }
186  #ifdef CHEMPS2_MPI_COMPILATION
187  unitary->broadcast( MPI_CHEMPS2_MASTER );
188  #endif
189  buildTmatrix(); //With an updated unitary, the Qocc, Tmat, and HamDMRG objects need to be updated as well.
190  buildQmatOCC();
191  fillConstAndTmatDMRG( HamDMRG );
192  if ( am_i_master ){
193  DMRGSCFrotations::rotate( VMAT_ORIG, HamDMRG->getVmat(), NULL, 'A', 'A', 'A', 'A', iHandler, unitary, mem1, mem2, work_mem_size, tmp_filename );
194  cout << "DMRGSCF::solve : Rotated the active space to localized orbitals, sorted according to the exchange matrix." << endl;
195  }
196  #ifdef CHEMPS2_MPI_COMPILATION
197  HamDMRG->getVmat()->broadcast( MPI_CHEMPS2_MASTER );
198  #endif
199  }
200 
201  if (( OptScheme == NULL ) && ( rootNum == 1 )){ // Do FCI, and calculate the 2DM
202 
203  if ( am_i_master ){
204  const int nalpha = ( num_elec + TwoS ) / 2;
205  const int nbeta = ( num_elec - TwoS ) / 2;
206  const double workmem = 1000.0; // 1GB
207  const int verbose = 2;
208  CheMPS2::FCI * theFCI = new CheMPS2::FCI( HamDMRG, nalpha, nbeta, Irrep, workmem, verbose );
209  double * inoutput = new double[ theFCI->getVecLength(0) ];
210  theFCI->ClearVector( theFCI->getVecLength(0), inoutput );
211  inoutput[ theFCI->LowestEnergyDeterminant() ] = 1.0;
212  Energy = theFCI->GSDavidson( inoutput );
213  theFCI->Fill2RDM( inoutput, DMRG2DM );
214  delete theFCI;
215  delete [] inoutput;
216  }
217  #ifdef CHEMPS2_MPI_COMPILATION
218  MPIchemps2::broadcast_array_double( &Energy, 1, MPI_CHEMPS2_MASTER );
219  MPIchemps2::broadcast_array_double( DMRG2DM, dmrgsize_power4, MPI_CHEMPS2_MASTER );
220  #endif
221 
222  } else { // Do the DMRG sweeps, and calculate the 2DM
223 
224  assert( OptScheme != NULL );
225  for ( int cnt = 0; cnt < dmrgsize_power4; cnt++ ){ DMRG2DM[ cnt ] = 0.0; } // Clear the 2-RDM ( to allow for state-averaged calculations )
226  DMRG * theDMRG = new DMRG( Prob, OptScheme, CheMPS2::DMRG_storeMpsOnDisk, tmp_folder );
227  for ( int state = 0; state < rootNum; state++ ){
228  if ( state > 0 ){ theDMRG->newExcitation( fabs( Energy ) ); }
229  Energy = theDMRG->Solve();
230  if ( scf_options->getStateAveraging() ){ // When SA-DMRGSCF: 2DM += current 2DM
231  theDMRG->calc2DMandCorrelations();
232  copy2DMover( theDMRG->get2DM(), nOrbDMRG, DMRG2DM );
233  }
234  if (( state == 0 ) && ( rootNum > 1 )){ theDMRG->activateExcitations( rootNum - 1 ); }
235  }
236  if ( !( scf_options->getStateAveraging() )){ // When SS-DMRGSCF: 2DM += last 2DM
237  theDMRG->calc2DMandCorrelations();
238  copy2DMover( theDMRG->get2DM(), nOrbDMRG, DMRG2DM );
239  }
240  if (( scf_options->getDumpCorrelations() ) && ( am_i_master )){ theDMRG->getCorrelations()->Print(); }
241  if (CheMPS2::DMRG_storeMpsOnDisk){ theDMRG->deleteStoredMPS(); }
242  if (CheMPS2::DMRG_storeRenormOptrOnDisk){ theDMRG->deleteStoredOperators(); }
243  delete theDMRG;
244  if (( scf_options->getStateAveraging() ) && ( rootNum > 1 )){
245  const double averagingfactor = 1.0 / rootNum;
246  for ( int cnt = 0; cnt < dmrgsize_power4; cnt++ ){ DMRG2DM[ cnt ] *= averagingfactor; }
247  }
248 
249  }
250  setDMRG1DM( num_elec, nOrbDMRG, DMRG1DM, DMRG2DM );
251 
252  // Possibly rotate the active space to the natural orbitals
253  if (( scf_options->getWhichActiveSpace() == 1 ) && ( master_diis == 0 )){ // When the DIIS has started: stop
254  copy_active( DMRG1DM, theQmatWORK, iHandler, true );
255  block_diagonalize( 'A', theQmatWORK, unitary, mem1, mem2, iHandler, true, DMRG2DM, NULL, NULL ); // Unitary is updated and DMRG2DM rotated
256  setDMRG1DM( num_elec, nOrbDMRG, DMRG1DM, DMRG2DM );
257  buildTmatrix(); // With an updated unitary, the Qocc and Tmat matrices need to be updated as well.
258  buildQmatOCC();
259  if ( am_i_master ){ cout << "DMRGSCF::solve : Rotated the active space to natural orbitals, sorted according to the NOON." << endl; }
260  }
261 
262  // Calculate the matrix elements needed to calculate the gradient and hessian
263  buildQmatACT();
264  if ( am_i_master ){
265  DMRGSCFrotations::rotate( VMAT_ORIG, NULL, theRotatedTEI, 'C', 'C', 'F', 'F', iHandler, unitary, mem1, mem2, work_mem_size, tmp_filename );
266  DMRGSCFrotations::rotate( VMAT_ORIG, NULL, theRotatedTEI, 'C', 'V', 'C', 'V', iHandler, unitary, mem1, mem2, work_mem_size, tmp_filename );
267  buildFmat( theFmatrix, theTmatrix, theQmatOCC, theQmatACT, iHandler, theRotatedTEI, DMRG2DM, DMRG1DM );
268  buildWtilde( wmattilde, theTmatrix, theQmatOCC, theQmatACT, iHandler, theRotatedTEI, DMRG2DM, DMRG1DM );
269  augmentedHessianNR( theFmatrix, wmattilde, iHandler, unitary, gradient, &updateNorm, &gradNorm ); // On return the gradient contains the update
270  }
271  #ifdef CHEMPS2_MPI_COMPILATION
272  MPIchemps2::broadcast_array_double( &updateNorm, 1, MPI_CHEMPS2_MASTER );
273  MPIchemps2::broadcast_array_double( &gradNorm, 1, MPI_CHEMPS2_MASTER );
274  #endif
275 
276  }
277 
278  delete [] mem1;
279  delete [] mem2;
280  delete theRotatedTEI;
281  delete wmattilde;
282  if ( am_i_master ){ delete_file( tmp_filename ); }
283 
284  delete Prob;
285  delete HamDMRG;
286  if ( gradient != NULL ){ delete [] gradient; }
287  if ( diis_vec != NULL ){ delete [] diis_vec; }
288  if ( diis != NULL ){ delete diis; }
289  if ( theLocalizer != NULL ){ delete theLocalizer; }
290 
291  successful_solve = true;
292  return Energy;
293

296 void CheMPS2::CASSCF::augmentedHessianNR( DMRGSCFmatrix * localFmat, DMRGSCFwtilde * localwtilde, const DMRGSCFindices * localIdx, const DMRGSCFunitary * localUmat, double * theupdate, double * updateNorm, double * gradNorm ){
297 
298  /* A good read to understand
299  (1) how the augmented Hessian arises from a rational function optimization
300  (2) where the parameter lambda in Eq. (22) of Yanai, IJQC 109, 2178-2190 (2009) comes from
301  (3) why the smallest algebraic eigenvalue + corresponding eigenvector should be retained for minimizations
302  Banerjee, Adams, Simons, Shepard, "Search for stationary points on surfaces",
303  J. Phys. Chem. 1985, volume 89, pages 52-57, doi:10.1021/j100247a015 */
304 
305  //Calculate the gradient
306  const int x_linearlength = localUmat->getNumVariablesX();
307  gradNorm[ 0 ] = construct_gradient( localFmat, localIdx, theupdate );
308 
309  //Find the lowest eigenvalue and corresponding eigenvector of the augmented hessian
310  {
311  Davidson deBoskabouter( x_linearlength + 1, CheMPS2::DAVIDSON_NUM_VEC,
312  CheMPS2::DAVIDSON_NUM_VEC_KEEP,
313  CheMPS2::DAVIDSON_FCI_RTOL,
314  CheMPS2::DAVIDSON_PRECOND_CUTOFF, false ); // No debug printing
315  double ** whichpointers = new double*[ 2 ];
316  char instruction = deBoskabouter.FetchInstruction( whichpointers );
317  assert( instruction == 'A' );
318  diag_hessian( localFmat, localwtilde, localIdx, whichpointers[ 1 ] );
319  whichpointers[ 1 ][ x_linearlength ] = 0.0;
320  for ( int cnt = 0; cnt < x_linearlength; cnt++ ){ // Initial guess = [ -gradient / diag(hessian) , 1 ]
321  const double denom = ( whichpointers[ 1 ][ cnt ] > CheMPS2::DAVIDSON_PRECOND_CUTOFF ) ? whichpointers[ 1 ][ cnt ] : CheMPS2::DAVIDSON_PRECOND_CUTOFF;
322  whichpointers[ 0 ][ cnt ] = - theupdate[ cnt ] / denom;
323  }
324  whichpointers[ 0 ][ x_linearlength ] = 1.0;
325  instruction = deBoskabouter.FetchInstruction( whichpointers );
326  while ( instruction == 'B' ){
327  augmented_hessian( localFmat, localwtilde, localIdx, whichpointers[ 0 ], whichpointers[ 1 ], theupdate, x_linearlength );
328  instruction = deBoskabouter.FetchInstruction( whichpointers );
329  }
330  assert( instruction == 'C' );
331  double scalar = 1.0 / whichpointers[ 0 ][ x_linearlength ];
332  cout << "DMRGSCF::augmentedHessianNR : Augmented Hessian update found with " << deBoskabouter.GetNumMultiplications() << " Davidson iterations." << endl;
333  if ( CheMPS2::DMRGSCF_debugPrint ){
334  cout << "DMRGSCF::augmentedHessianNR : Lowest eigenvalue = " << whichpointers[ 1 ][ 0 ] << endl;
335  cout << "DMRGSCF::augmentedHessianNR : The last number of the eigenvector (which will be rescaled to one) = " << scalar << endl;
336  }
337  for ( int cnt = 0; cnt < x_linearlength; cnt++ ){ theupdate[ cnt ] = scalar * whichpointers[ 0 ][ cnt ]; }
338  delete [] whichpointers;
339  }
340 
341  //Calculate the update norm
342  updateNorm[ 0 ] = 0.0;
343  for ( int cnt = 0; cnt < x_linearlength; cnt++ ){ updateNorm[ 0 ] += theupdate[ cnt ] * theupdate[ cnt ]; }
344  updateNorm[ 0 ] = sqrt( updateNorm[ 0 ] );
345  cout << "DMRGSCF::augmentedHessianNR : Norm of the update = " << updateNorm[ 0 ] << endl;
346 
347 }

349 double CheMPS2::CASSCF::construct_gradient( DMRGSCFmatrix * Fmatrix, const DMRGSCFindices * idx, double * gradient ){
350 
351  const int n_irreps = idx->getNirreps();
352 
353  int jump = 0;
354  for ( int irrep = 0; irrep < n_irreps; irrep++ ){
355 
356  const int NORB = idx->getNORB( irrep );
357  const int NOCC = idx->getNOCC( irrep );
358  const int NACT = idx->getNDMRG( irrep );
359  const int NVIR = idx->getNVIRT( irrep );
360  const int N_OA = NOCC + NACT;
361  double * FMAT = Fmatrix->getBlock( irrep );
362 
363  // Type 0: act-occ
364  for ( int occ = 0; occ < NOCC; occ++ ){
365  for ( int act = 0; act < NACT; act++ ){
366  gradient[ jump + act + NACT * occ ] = 2 * ( FMAT[ NOCC + act + NORB * occ ] - FMAT[ occ + NORB * ( NOCC + act ) ] );
367  }
368  }
369  jump += NOCC * NACT;
370 
371  // Type 1: vir-act
372  for ( int act = 0; act < NACT; act++ ){
373  for ( int vir = 0; vir < NVIR; vir++ ){
374  gradient[ jump + vir + NVIR * act ] = 2 * ( FMAT[ N_OA + vir + NORB * ( NOCC + act ) ] - FMAT[ NOCC + act + NORB * ( N_OA + vir ) ] );
375  }
376  }
377  jump += NACT * NVIR;
378 
379  // Type 2: vir-occ
380  for ( int occ = 0; occ < NOCC; occ++ ){
381  for ( int vir = 0; vir < NVIR; vir++ ){
382  gradient[ jump + vir + NVIR * occ ] = 2 * ( FMAT[ N_OA + vir + NORB * occ ] - FMAT[ occ + NORB * ( N_OA + vir ) ] );
383  }
384  }
385  jump += NOCC * NVIR;
386  }
387 
388  double gradient_norm = 0.0;
389  for ( int cnt = 0; cnt < jump; cnt++ ){ gradient_norm += gradient[ cnt ] * gradient[ cnt ]; }
390  gradient_norm = sqrt( gradient_norm );
391  cout << "DMRGSCF::construct_gradient : Norm of the gradient = " << gradient_norm << endl;
392  return gradient_norm;
393 
394 }

409 void CheMPS2::CASSCF::diag_hessian( DMRGSCFmatrix * Fmatrix, const DMRGSCFwtilde * Wtilde, const DMRGSCFindices * idx, double * diagonal ){
410 
411  const int n_irreps = idx->getNirreps();
412 
413  int jump = 0;
414  for ( int irrep = 0; irrep < n_irreps; irrep++ ){
415 
416  const int NORB = idx->getNORB( irrep );
417  const int NOCC = idx->getNOCC( irrep );
418  const int NACT = idx->getNDMRG( irrep );
419  const int NVIR = idx->getNVIRT( irrep );
420  const int N_OA = NOCC + NACT;
421  double * FMAT = Fmatrix->getBlock( irrep );
422 
423  for ( int occ = 0; occ < NOCC; occ++ ){
424  const double F_occ = FMAT[ occ * ( NORB + 1 ) ];
425  for ( int act = 0; act < NACT; act++ ){
426  const double F_act = FMAT[ ( NOCC + act ) * ( NORB + 1 ) ];
427  diagonal[ jump + act + NACT * occ ] = - 2 * ( F_occ + F_act ) + ( Wtilde->get( irrep, irrep, NOCC + act, occ, NOCC + act, occ )
428  - Wtilde->get( irrep, irrep, NOCC + act, occ, occ, NOCC + act )
429  - Wtilde->get( irrep, irrep, occ, NOCC + act, NOCC + act, occ )
430  + Wtilde->get( irrep, irrep, occ, NOCC + act, occ, NOCC + act ) );
431  }
432  }
433  jump += NOCC * NACT;
434 
435  for ( int act = 0; act < NACT; act++ ){
436  const double F_act = FMAT[ ( NOCC + act ) * ( NORB + 1 ) ];
437  for ( int vir = 0; vir < NVIR; vir++ ){
438  const double F_vir = FMAT[ ( N_OA + vir ) * ( NORB + 1 ) ];
439  diagonal[ jump + vir + NVIR * act ] = - 2 * ( F_act + F_vir ) + Wtilde->get( irrep, irrep, NOCC + act, N_OA + vir, NOCC + act, N_OA + vir );
440  }
441  }
442  jump += NACT * NVIR;
443 
444  for ( int occ = 0; occ < NOCC; occ++ ){
445  const double F_occ = FMAT[ occ * ( NORB + 1 ) ];
446  for ( int vir = 0; vir < NVIR; vir++ ){
447  const double F_vir = FMAT[ ( N_OA + vir ) * ( NORB + 1 ) ];
