62 double CheMPS2::EdmistonRuedenberg::Optimize(double * temp1, double * temp2, const bool startFromRandomUnitary, const double gradThreshold, const int maxIter){
63 
64  //Clear the unitary
65  for (int irrep=0; irrep<SymmInfo.getNumberOfIrreps(); irrep++){
66  const int linsize = iHandler->getNORB(irrep);
67  double * block = unitary->getBlock(irrep);
68  for (int cnt=0; cnt<linsize*linsize; cnt++){ block[cnt] = 0.0; }
69  for (int cnt=0; cnt<linsize; cnt++){ block[cnt*(1+linsize)] = 1.0; }
70  }
71 
72  //Setting up the variables for the gradient
73  double gradNorm = 1.0;
74  int numVariables = 0;
75  for (int irrep=0; irrep<SymmInfo.getNumberOfIrreps(); irrep++){
76  if (iHandler->getNORB(irrep) > 1){
77  numVariables += iHandler->getNORB(irrep) * (iHandler->getNORB(irrep) - 1) / 2;
78  }
79  }
80  double * gradient = new double[numVariables];
81  for (int cnt=0; cnt<numVariables; cnt++){ gradient[cnt] = 0.0; }
82 
83  //Randomize the unitary if asked
84  if (startFromRandomUnitary){
85  for (int cnt=0; cnt<numVariables; cnt++){ gradient[cnt] = ((double) rand())/RAND_MAX - 0.5; }
86  unitary->updateUnitary(temp1, temp2, gradient, false, false); //multiply = compact = false
87  for (int cnt=0; cnt<numVariables; cnt++){ gradient[cnt] = 0.0; }
88  }
89 
90  DMRGSCFVmatRotations theRotator(Ham, iHandler);
91  theRotator.fillVmatRotated(VmatRotated, unitary, temp1, temp2);
92 
93  //Setting up the variables for the cost function
94  double Icost = costFunction();
95  if (printLevel>0){ cout << " EdmistonRuedenberg::Optimize : Cost function at start = " << Icost << endl; }
96  double Icost_previous = 0.0;
97 
98  int nIterations = 0;
99 
100  while ((gradNorm > gradThreshold) && (nIterations < maxIter)){
101 
102  nIterations++;
103 
104  //Update the unitary
105  unitary->updateUnitary(temp1, temp2, gradient, true, false); //multiply = true; compact = false
106 
107  //Rotate the Vmat
108  Icost_previous = Icost;
109  theRotator.fillVmatRotated(VmatRotated, unitary, temp1, temp2);
110  Icost = costFunction();
111 
112  /* What if the cost function has dimished? Then make the rotation step a bit smaller!
113  Rotate back ever closer to unitary_effective = I; untill Icost_previous < Icost
114  1 - 0.5 - 0.5^2 - 0.5^3 - 0.5^4 - ... = 2 - ( 1 + 0.5 + 0.5^2 + ... ) = 2 - 1/(1-0.5) = 0 */
115  if (Icost_previous > Icost){
116  if (printLevel>1){ cout << " WARNING : Icost = " << Icost << " < Icost_previous = " << Icost_previous << endl; }
117  for (int cnt=0; cnt<numVariables; cnt++){ gradient[cnt] = -gradient[cnt]; } //Switch the sign of the update : rotate back!
118  int nIterationsBACK = 0;
119  while ((Icost_previous > Icost) && (nIterationsBACK < CheMPS2::EDMISTONRUED_maxIterBackTfo)){
120  nIterationsBACK++;
121  for (int cnt=0; cnt<numVariables; cnt++){ gradient[cnt] *= 0.5; }
122  unitary->updateUnitary(temp1, temp2, gradient, true, false); //multiply = true; compact = false
123  theRotator.fillVmatRotated(VmatRotated, unitary, temp1, temp2);
124  Icost = costFunction();
125  }
126  if (printLevel>1){ cout << " WARNING : Rotated back a bit. Now Icost = " << Icost << endl; }
127  }
128 
129  //Calculate the gradient and Hessian and update
130  gradNorm = augmentedHessianNewtonRaphson(gradient, temp1, temp2);
131 
132  if (printLevel>1){
133  cout << " After iteration " << nIterations << " : Icost = " << Icost << " has gradNorm = " << gradNorm << endl;
134  }
135 
136  }
137 
138  delete [] gradient;
139 
140  if (printLevel>0){
141  cout << " Cost function at stop = " << Icost << endl;
142  cout << " Gradient norm = " << gradNorm << " after " << nIterations << " iterations." << endl;
143  }
144 
145  return Icost;
146

169 double CheMPS2::EdmistonRuedenberg::calcHessianValue(const int irrep, const int p, const int q, const int r, const int s) const{
170 
171  double value = 0.0;
172  if (p==r){
173  value += ( 8*VmatRotated->get(irrep, irrep, irrep, irrep, p, p, q, s)
174  + 4*VmatRotated->get(irrep, irrep, irrep, irrep, p, q, p, s)
175  - 2*VmatRotated->get(irrep, irrep, irrep, irrep, q, q, q, s)
176  - 2*VmatRotated->get(irrep, irrep, irrep, irrep, s, s, s, q) );
177  }
178  if (q==s){
179  value += ( 8*VmatRotated->get(irrep, irrep, irrep, irrep, q, q, p, r)
180  + 4*VmatRotated->get(irrep, irrep, irrep, irrep, q, p, q, r)
181  - 2*VmatRotated->get(irrep, irrep, irrep, irrep, p, p, p, r)
182  - 2*VmatRotated->get(irrep, irrep, irrep, irrep, r, r, r, p) );
183  }
184  if (p==s){
185  value -= ( 8*VmatRotated->get(irrep, irrep, irrep, irrep, p, p, q, r)
186  + 4*VmatRotated->get(irrep, irrep, irrep, irrep, p, q, p, r)
187  - 2*VmatRotated->get(irrep, irrep, irrep, irrep, q, q, q, r)
188  - 2*VmatRotated->get(irrep, irrep, irrep, irrep, r, r, r, q) );
189  }
190  if (q==r){
191  value -= ( 8*VmatRotated->get(irrep, irrep, irrep, irrep, q, q, p, s)
192  + 4*VmatRotated->get(irrep, irrep, irrep, irrep, q, p, q, s)
193  - 2*VmatRotated->get(irrep, irrep, irrep, irrep, p, p, p, s)
194  - 2*VmatRotated->get(irrep, irrep, irrep, irrep, s, s, s, p) );
195  }
196  return value;

200 double CheMPS2::EdmistonRuedenberg::augmentedHessianNewtonRaphson(double * gradient, double * temp1, double * temp2) const{
201 
202  int jump = 0;
203  double gradNorm = 0.0;
204 
205  for (int irrep=0; irrep<SymmInfo.getNumberOfIrreps(); irrep++){
206 
207  int linsize = iHandler->getNORB(irrep);
208 
209  if (linsize>1){
210 
211  int hessianlinearsize = linsize*(linsize-1)/2 + 1;
212 
213  /* linsize linsize^4 hesslinsize hesslinsize^2 3*hesslinsize-1 4*hesslinsize-1 (hesslinsize+1)*hesslinsize
214  2 16 2 4 5 7 6
215  3 81 4 16 11 15 20
216  4 256 7 49 20 27 56
217  5 625 11 121 32 43 132
218  6 1296 16 256 47 63 272
219 
220  Conclusion : - temp1 is large enough for hessian (hessianlinearsize^2)
221  - temp2 is large enough for workspace (max(hesslinsize^2, 3*hesslinsize-1)) AND eigenvecs (hesslinsize)
222  */
223 
224  double * hessian = temp1;
225  double * eigen = temp2;
226  double * work = temp2 + hessianlinearsize;
227  int lwork = linsize*linsize*linsize*linsize - hessianlinearsize;
228 
229  /* Maximizing the cost function O(x) = O(0) + g^T * x + x^T * H * x / 2
230  == Minimizing the cost function -O(x) = -O(0) - g^T * x - x^T * H * x / 2 */
231  for (int row=0; row<linsize; row++){
232  for (int col=row+1; col<linsize; col++){
233  const double waarde = calcGradientValue(irrep, row, col);
234  gradient[jump + row + col*(col-1)/2] = waarde;
235  gradNorm += waarde * waarde;
236  for (int row2=0; row2<linsize; row2++){
237  for (int col2=row2+1; col2<linsize; col2++){
238  const double value = calcHessianValue(irrep, row, col, row2, col2);
239  hessian[row + col*(col-1)/2 + hessianlinearsize * (row2 + col2*(col2-1)/2)] = - value; //MINUS SIGN FOR OTHER DIRECTION
240  }
241  }
242  }
243  }
244 
245  for (int cnt=0; cnt<hessianlinearsize-1; cnt++){
246  hessian[hessianlinearsize-1 + hessianlinearsize * cnt] = - gradient[jump + cnt]; //MINUS SIGN FOR OTHER DIRECTION
247  hessian[cnt + hessianlinearsize * (hessianlinearsize-1)] = - gradient[jump + cnt]; //MINUS SIGN FOR OTHER DIRECTION
248  }
249  hessian[hessianlinearsize*hessianlinearsize-1] = 0.0;
250 
251  //Find its lowest eigenvalue and vector
252  char jobz = 'V';
253  char uplo = 'U';
254  int info;
255  dsyev_(&jobz,&uplo,&hessianlinearsize,hessian,&hessianlinearsize,eigen,work,&lwork,&info);
256 
257  double scalar = 1.0/hessian[hessianlinearsize-1];
258  int inc = 1;
259  dscal_(&hessianlinearsize,&scalar,hessian,&inc);
260 
261  for (int cnt=0; cnt<hessianlinearsize-1; cnt++){ gradient[jump + cnt] = hessian[cnt]; }
262 
263  jump += hessianlinearsize-1;
264 
265  }
266 
267  }
268 
269  gradNorm = sqrt(gradNorm);
270 
271  return gradNorm;
272 
273 }

275 void CheMPS2::EdmistonRuedenberg::Fiedler(const int irrep, int * reorder, double * laplacian, double * temp2){
276 
277  //For information on the Fiedler vector: see http://web.eecs.utk.edu/~mberry/order/node9.html
278 
279  int linsize = iHandler->getNORB(irrep);
280  assert( linsize>=2 );
281 
282  //Preamble: linsize>=2 at this point
283  double * work = temp2; //temp2 at least of size 4*linsize*linsize
284  int lwork = 3*linsize*linsize; //3*linsize*linsize > 3*linsize-1 for linsize>=2
285  double * eigs = temp2 + lwork; //has size linsize*linsize > linsize for linsize>=2
286 
287  //Calculate the eigenspectrum of the Laplacian
288  char jobz = 'V';
289  char uplo = 'U';
290  int info;
291  dsyev_(&jobz, &uplo, &linsize, laplacian, &linsize, eigs, work, &lwork, &info);
292  if (printLevel>1){
293  cout << " EdmistonRuedenberg::Fiedler : Smallest eigs(Laplacian[" << irrep << "]) = [ " << eigs[0] << " , " << eigs[1] << " ]." << endl;
294  }
295 
296  //The second eigenvector is the Fiedler vector
297  double * FiedlerVector = laplacian + linsize;
298  double * FiedlerVectorCopy = laplacian;
299  for (int cnt=0; cnt<linsize; cnt++){ FiedlerVectorCopy[cnt] = FiedlerVector[cnt]; }
300 
301  //Find the Fiedler ordering
302  const double upperBound = 2.0; //Sum_i FiedlerVector[i]^2 = 1 --> fabs(FiedlerVector[i]) <= 1.0
303  for (int value=0; value<linsize; value++){
304  int index=0;
305  for (int cnt=1; cnt<linsize; cnt++){
306  if (FiedlerVectorCopy[cnt] < FiedlerVectorCopy[index]){ index = cnt; }
307  } //Index now corresponds to the smallest value in FiedlerVectorCopy
308  reorder[value] = index;
309  FiedlerVectorCopy[index] = upperBound; //Remove from the interesting entries in FiedlerVectorCopy
310  }
311 
312  if (printLevel>1){
313  bool isOK = true;
314  for (int cnt=0; cnt<linsize-1; cnt++){
315  if (FiedlerVector[reorder[cnt]] > FiedlerVector[reorder[cnt+1]]){ isOK = false; }
316  }
317  assert( isOK );
318  cout << " Reorder[" << irrep << "] = [ ";
319  for (int cnt=0; cnt<linsize-1; cnt++){ cout << reorder[cnt] << " , "; }
320  cout << reorder[linsize-1] << " ]." << endl;
321  }
322 
323  //Reorder the vectors in the unitary
324  double * blockU = unitary->getBlock(irrep);
325  for (int row=0; row<linsize; row++){
326  for (int col=0; col<linsize; col++){
327  work[row + linsize * col] = blockU[reorder[row] + linsize * col];
328  }
329  }
330 
331  int size = linsize*linsize;
332  int inc = 1;
333  dcopy_(&size, work, &inc, blockU, &inc);
334 
335  if (printLevel>1){
336  char trans = 'T';
337  char notrans = 'N';
338  double alpha = 1.0;
339  double beta = 0.0;
340  dgemm_(&trans, &notrans, &linsize, &linsize, &linsize, &alpha, blockU, &linsize, blockU, &linsize, &beta, work, &linsize);
341  double sum = 0.0;
342  for (int row=0; row<linsize; row++){
343  sum += (work[row+linsize*row]-1.0)*(work[row+linsize*row]-1.0);
344  for (int col=row+1; col<linsize; col++){
345  sum += work[row+linsize*col]*work[row+linsize*col] + work[col+linsize*row]*work[col+linsize*row];
346  }
347  }
348  sum = sqrt(sum);
349  cout << " 2-norm of Unitary[" << irrep << "]^T * Unitary[" << irrep << "] - I = " << sum << endl;
350  }
351 
352 }

373 void CheMPS2::EdmistonRuedenberg::FiedlerExchange(const int maxlinsize, double * temp1, double * temp2){
374 
375  //For information on the Fiedler vector: see http://web.eecs.utk.edu/~mberry/order/node9.html
376 
377  if (printLevel>0){ cout << " EdmistonRuedenberg::FiedlerExchange : Cost function at start = " << FiedlerExchangeCost() << endl; }
378 
379  int * reorder = new int[maxlinsize];
380 
381  for (int irrep=0; irrep<SymmInfo.getNumberOfIrreps(); irrep++){
382 
383  const int linsize = iHandler->getNORB(irrep);
384  if (linsize>1){
385 
386  //temp1 and temp2 both at least of size 4*linsize*linsize
387  double * laplacian = temp1;
388 
389  //Fill the weighted graph Laplacian
390  for (int row=0; row<linsize; row++){
391  laplacian[row*(1+linsize)] = 0.0;
392  for (int col=row+1; col<linsize; col++){
393  laplacian[row + linsize*col] = - VmatRotated->get(irrep,irrep,irrep,irrep,row,col,col,row); //minus the exchange matrix
394  laplacian[col + linsize*row] = laplacian[row + linsize*col]; //Symmetric matrix
395  laplacian[row + linsize*row] -= laplacian[row + linsize*col]; //On diagonal : Minus the sum of the other terms on that row
396  }
397  for (int col=0; col<row; col++){
398  laplacian[row + linsize*row] -= laplacian[row + linsize*col]; //On diagonal : Minus the sum of the other terms on that row
399  }
400  }
401 
402  Fiedler(irrep, reorder, laplacian, temp2);
403 
404  }
405  }
406 
407  delete [] reorder;
408 
409  DMRGSCFVmatRotations theRotator(Ham, iHandler);
410  theRotator.fillVmatRotated(VmatRotated, unitary, temp1, temp2);
411 
412  if (printLevel>0){ cout << " EdmistonRuedenberg::FiedlerExchange : Cost function at end = " << FiedlerExchangeCost() << endl; }
