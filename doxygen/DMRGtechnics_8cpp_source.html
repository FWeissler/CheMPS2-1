39 void CheMPS2::DMRG::calc2DMandCorrelations(){
40 
41  //First get the whole MPS into left-canonical form
42  int index = L-2;
43  Sobject * denS = new Sobject(index,denBK->gIrrep(index),denBK->gIrrep(index+1),denBK);
44  denS->Join(MPS[index],MPS[index+1]);
45  Heff Solver(denBK, Prob);
46  double Energy = 0.0;
47  double ** VeffTilde = NULL;
48  if (Exc_activated){
49  VeffTilde = new double*[nStates-1];
50  for (int cnt=0; cnt<nStates-1; cnt++){
51  VeffTilde[cnt] = new double[denS->gKappa2index(denS->gNKappa())];
52  calcVeffTilde(VeffTilde[cnt], denS, cnt);
53  }
54  }
55  Energy = Solver.SolveDAVIDSON(denS, Ltensors, Atensors, Btensors, Ctensors, Dtensors, S0tensors, S1tensors, F0tensors, F1tensors, Qtensors, Xtensors, nStates-1, VeffTilde);
56  if (Exc_activated){
57  for (int cnt=0; cnt<nStates-1; cnt++){ delete [] VeffTilde[cnt]; }
58  delete [] VeffTilde;
59  }
60  Energy += Prob->gEconst();
61  if (Energy<TotalMinEnergy){ TotalMinEnergy = Energy; }
62  denS->Split(MPS[index],MPS[index+1],OptScheme->getD(OptScheme->getNInstructions()-1),true,true);
63  delete denS;
64 
65  cout << "**************************************" << endl;
66  cout << "** 2DM and Correlations calculation **" << endl;
67  cout << "**************************************" << endl;
68  updateMovingRightSafe(index);
69 
70  TensorDiag * Norm = new TensorDiag(L, denBK);
71  MPS[L-1]->QR(Norm);
72  delete Norm;
73 
74  //Allocate space for the 2DM
75  if (the2DMallocated){
76  delete the2DM;
77  the2DMallocated = false;
78  }
79  the2DM = new TwoDM(denBK, Prob);
80  the2DMallocated = true;
81 
82  //Then calculate step by step the 2DM
83  for (int siteindex=L-1; siteindex>=0; siteindex--){
84  the2DM->FillSite(MPS[siteindex], Ltensors, F0tensors, F1tensors, S0tensors, S1tensors);
85  if (siteindex>0){
86  TensorDiag * Left = new TensorDiag(siteindex, denBK);
87  MPS[siteindex]->LQ(Left);
88  MPS[siteindex-1]->RightMultiply(Left);
89  delete Left;
90  updateMovingLeftSafe2DM(siteindex-1);
91  }
92  }
93 
94  //Then perform two checks: double trace & energy
95  double NtimesNminus1 = the2DM->doubletrace2DMA();
96  cout << " N(N-1) = " << denBK->gN() * (denBK->gN() - 1) << " and calculated by double trace of the 2DM-A = " << NtimesNminus1 << endl;
97 
98  double Energy2DMA = the2DM->calcEnergy();
99  cout << " Energy obtained by Heffective at edge = " << Energy << " and as Econst + 0.5*trace(2DM-A*Ham) = " << Energy2DMA << endl;
100 
101  //Now the MPS has the gauge form CRRRRRRRRR
102  //Allocate space for the Correlations
103  if (theCorrAllocated){
104  delete theCorr;
105  theCorrAllocated = false;
106  }
107  theCorr = new Correlations(denBK, Prob, the2DM);
108  theCorrAllocated = true;
109 
110  //Then calculate step by step the mutual information.
111  //Define the following tensor arrays thereto. The native DMRG ones are at the edge and are hence small.
112  TensorGYZ ** Gtensors = new TensorGYZ * [L-1];
113  TensorGYZ ** Ytensors = new TensorGYZ * [L-1];
114  TensorGYZ ** Ztensors = new TensorGYZ * [L-1];
115  TensorK ** Ktensors = new TensorK * [L-1];
116  TensorM ** Mtensors = new TensorM * [L-1];
117 
118  //Do the actual work
119  for (int siteindex=1; siteindex<L; siteindex++){
120 
121  //Switch MPS gauge
122  TensorDiag * Right = new TensorDiag(siteindex, denBK);
123  MPS[siteindex-1]->QR(Right);
124  MPS[siteindex]->LeftMultiply(Right);
125  delete Right;
126

128  const int dimL = denBK->gMaxDimAtBound(siteindex-1);
129  const int dimR = denBK->gMaxDimAtBound(siteindex);
130  double * workmemLR = new double[dimL*dimR];
131  for (int previousindex=0; previousindex<siteindex-1; previousindex++){
132  TensorGYZ * newG = new TensorGYZ(siteindex, 'G', denBK);
133  TensorGYZ * newY = new TensorGYZ(siteindex, 'Y', denBK);
134  TensorGYZ * newZ = new TensorGYZ(siteindex, 'Z', denBK);
135  TensorK * newK = new TensorK(siteindex, denBK->gIrrep(previousindex), denBK);
136  TensorM * newM = new TensorM(siteindex, denBK->gIrrep(previousindex), denBK);
137 
138  newG->update(MPS[siteindex-1], Gtensors[previousindex], workmemLR);
139  newY->update(MPS[siteindex-1], Ytensors[previousindex], workmemLR);
140  newZ->update(MPS[siteindex-1], Ztensors[previousindex], workmemLR);
141  newK->update(Ktensors[previousindex], MPS[siteindex-1], workmemLR, false);
142  newM->update(Mtensors[previousindex], MPS[siteindex-1], workmemLR, false);
143 
144  delete Gtensors[previousindex];
145  delete Ytensors[previousindex];
146  delete Ztensors[previousindex];
147  delete Ktensors[previousindex];
148  delete Mtensors[previousindex];
149 
150  Gtensors[previousindex] = newG;
151  Ytensors[previousindex] = newY;
152  Ztensors[previousindex] = newZ;
153  Ktensors[previousindex] = newK;
154  Mtensors[previousindex] = newM;
155  }
156  delete [] workmemLR;
157 
158  //Construct the new tensors
159  Gtensors[siteindex-1] = new TensorGYZ(siteindex, 'G', denBK);
160  Ytensors[siteindex-1] = new TensorGYZ(siteindex, 'Y', denBK);
161  Ztensors[siteindex-1] = new TensorGYZ(siteindex, 'Z', denBK);
162  Ktensors[siteindex-1] = new TensorK(siteindex, denBK->gIrrep(siteindex-1), denBK);
163  Mtensors[siteindex-1] = new TensorM(siteindex, denBK->gIrrep(siteindex-1), denBK);
164 
165  Gtensors[siteindex-1]->construct(MPS[siteindex-1]);
166  Ytensors[siteindex-1]->construct(MPS[siteindex-1]);
167  Ztensors[siteindex-1]->construct(MPS[siteindex-1]);
168  Ktensors[siteindex-1]->construct(MPS[siteindex-1]);
169  Mtensors[siteindex-1]->construct(MPS[siteindex-1]);
170 
171  //Use the tensors to fill in the Correlations
172  theCorr->FillSite(MPS[siteindex], Gtensors, Ytensors, Ztensors, Ktensors, Mtensors);
173 
174  }
175 
176  //Clean-up
177  for (int previousindex=0; previousindex<L-1; previousindex++){
178  delete Gtensors[previousindex];
179  delete Ytensors[previousindex];
180  delete Ztensors[previousindex];
181  delete Ktensors[previousindex];
182  delete Mtensors[previousindex];
183  }
184 
185  delete [] Gtensors;
186  delete [] Ytensors;
187  delete [] Ztensors;
188  delete [] Ktensors;
189  delete [] Mtensors;
190 
191  cout << " Single-orbital entropies (Hamiltonian index order is used!) = [ ";
192  for (int index=0; index < L-1; index++){ cout << theCorr->SingleOrbitalEntropy_HAM(index) << " , "; }
193  cout << theCorr->SingleOrbitalEntropy_HAM(L-1) << " ]." << endl;
194 
195  for (int power=0; power<=2; power++){
196  cout << " Idistance(" << power << ") = " << theCorr->MutualInformationDistance((double)power) << endl;
197  }
198  cout << "**************************************" << endl;

206 double CheMPS2::DMRG::getSpecificCoefficient(int * coeff) const{
207 
208  int * alpha = new int[ L ];
209  int * beta = new int[ L ];
210  for ( int orb=0; orb<L; orb++ ){
211  assert( ( coeff[orb] >= 0 ) && ( coeff[orb] <= 2 ) );
212  if ( coeff[orb] == 0 ){ alpha[orb] = 0; beta[orb] = 0; }
213  if ( coeff[orb] == 1 ){ alpha[orb] = 1; beta[orb] = 0; }
214  if ( coeff[orb] == 2 ){ alpha[orb] = 1; beta[orb] = 1; }
215  }
216  const double FCIcoeff = getFCIcoefficient( alpha, beta );
217  delete [] alpha;
218  delete [] beta;
219  return FCIcoeff;

223 double CheMPS2::DMRG::getFCIcoefficient(int * alpha, int * beta) const{ //DMRGcoeff = alpha/beta[Hamindex = Prob->gf2(DMRGindex)]
224 
225  //Check if it's possible
226  {
227  int nTot = 0;
228  int twoSz = 0;
229  int iTot = 0;
230  for (int DMRGindex=0; DMRGindex<L; DMRGindex++){
231  const int HamIndex = (Prob->gReorderD2h()) ? Prob->gf2(DMRGindex) : DMRGindex;
232  assert( ( alpha[HamIndex] == 0 ) || ( alpha[HamIndex] == 1 ) );
233  assert( ( beta[HamIndex] == 0 ) || ( beta[HamIndex] == 1 ) );
234  nTot += alpha[HamIndex] + beta[HamIndex];
235  twoSz += alpha[HamIndex] - beta[HamIndex];
236  if ((alpha[HamIndex]+beta[HamIndex])==1){ iTot = Irreps::directProd(iTot,denBK->gIrrep(DMRGindex)); }
237  }
238  if ( Prob->gN() != nTot ){
239  cout << "DMRG::getFCIcoefficient : Ndesired = " << Prob->gN() << " and Ntotal in alpha and beta strings = " << nTot << endl;
240  return 0.0;
241  }
242  // 2Sz can be -Prob->2S() ; -Prob->2S()+2 ; -Prob->2S()+4 ; ... ; Prob->2S()
243  if ( ( Prob->gTwoS() < twoSz ) || ( twoSz < -Prob->gTwoS() ) || ( ( Prob->gTwoS() - twoSz ) % 2 != 0 ) ){
244  cout << "DMRG::getFCIcoefficient : 2Sdesired = " << Prob->gTwoS() << " and Sz in alpha and beta strings = " << twoSz << endl;
245  return 0.0;
246  }
247  if ( Prob->gIrrep() != iTot ){
248  cout << "DMRG::getFCIcoefficient : Idesired = " << Prob->gIrrep() << " and Irrep of alpha and beta strings = " << iTot << endl;
249  return 0.0;
250  }
251  }
252 
253  //Construct necessary arrays
254  int Dmax = 1;
255  for (int DMRGindex=1; DMRGindex<L; DMRGindex++){
256  const int DtotBound = denBK->gTotDimAtBound(DMRGindex);
257  if (DtotBound>Dmax){ Dmax = DtotBound; }
258  }
259  double * arrayL = new double[Dmax];
260  double * arrayR = new double[Dmax];
261  int * twoSL = new int[L];
262  int * twoSR = new int[L];
263  int * jumpL = new int[L+1];
264  int * jumpR = new int[L+1];
265 
266  //Start the iterator
267  int num_SL = 0;
268  jumpL[num_SL] = 0;
269  int dimFirst = 1;
270  jumpL[num_SL+1] = jumpL[num_SL] + dimFirst;
271  twoSL[num_SL] = 0;
272  num_SL++;
273  arrayL[0] = 1.0;
274  int NL = 0;
275  int IL = 0;
276  int twoSLz = 0;
277 
278  for (int DMRGindex=0; DMRGindex<L; DMRGindex++){
279 
280  //Clear the right array
281  for (int count = 0; count < Dmax; count++){ arrayR[count] = 0.0; }
282 
283  //The local occupation
284  const int HamIndex = (Prob->gReorderD2h()) ? Prob->gf2(DMRGindex) : DMRGindex;
285  const int Nlocal = alpha[HamIndex] + beta[HamIndex];
286  const int twoSzloc = alpha[HamIndex] - beta[HamIndex];
287 
288  //The right symmetry sectors
289  const int NR = NL + Nlocal;
290  const int twoSRz = twoSLz + twoSzloc;
291  const int IR = (( Nlocal == 1 ) ? (Irreps::directProd(IL,denBK->gIrrep(DMRGindex))) : IL);
292 
293  int num_SR = 0;
294  jumpR[num_SR] = 0;
295  const int spread = ( ( Nlocal == 1 ) ? 1 : 0 );
296  for ( int cntSL = 0; cntSL < num_SL; cntSL++ ){
297  for ( int TwoSRattempt = twoSL[cntSL] - spread; TwoSRattempt <= twoSL[cntSL] + spread; TwoSRattempt+=2 ){
298  bool encountered = false;
299  for ( int cntSR = 0; cntSR < num_SR; cntSR++ ){
300  if ( twoSR[cntSR] == TwoSRattempt ){
301  encountered = true;
302  }
303  }
304  if ( encountered == false ){
305  const int dimR = denBK->gCurrentDim(DMRGindex+1,NR,TwoSRattempt,IR);
306  if ( dimR > 0 ){
307  jumpR[num_SR+1] = jumpR[num_SR] + dimR;
308  twoSR[num_SR] = TwoSRattempt;
309  num_SR++;
310  }
311  }
312  }
313  }
314  assert( jumpR[num_SR] <= Dmax );
315 
316  for ( int cntSR = 0; cntSR < num_SR; cntSR++ ){
317  int TwoSRvalue = twoSR[ cntSR ];
318  int dimR = jumpR[ cntSR+1 ] - jumpR[ cntSR ];
319  for ( int TwoSLvalue = TwoSRvalue - spread; TwoSLvalue <= TwoSRvalue + spread; TwoSLvalue += 2 ){
320 
321  int indexSL = -1;
322  for ( int cntSL = 0; cntSL < num_SL; cntSL++ ){
323  if ( twoSL[cntSL] == TwoSLvalue ){
324  indexSL = cntSL;
325  cntSL = num_SL; //exit loop
326  }
327  }
328  if ( indexSL != -1 ){
329  int dimL = jumpL[ indexSL+1 ] - jumpL[ indexSL ];
330  double * Tblock = MPS[DMRGindex]->gStorage(NL,TwoSLvalue,IL,NR,TwoSRvalue,IR);
331  double prefactor = sqrt( TwoSRvalue + 1 )
332  * gsl_sf_coupling_3j(TwoSLvalue, spread, TwoSRvalue, twoSLz, twoSzloc, -twoSRz)
333  * Heff::phase( -TwoSLvalue + spread - twoSRz );
334  double add2array = 1.0;
335  char notrans = 'N';
336  dgemm_( &notrans, &notrans, &dimFirst, &dimR, &dimL, &prefactor, arrayL + jumpL[indexSL], &dimFirst, Tblock, &dimL, &add2array, arrayR + jumpR[cntSR], &dimFirst);
337  }
338  }
339  }
340 
341  //Swap L <--> R
342  {
343  double * temp = arrayR;
344  arrayR = arrayL;
345  arrayL = temp;
346  int * temp2 = twoSR;
347  twoSR = twoSL;
348  twoSL = temp2;
349  temp2 = jumpR;
350  jumpR = jumpL;
351  jumpL = temp2;
352  num_SL = num_SR;
353  NL = NR;
354  IL = IR;
355  twoSLz = twoSRz;
356  }
357  }
358 
359  const double theCoeff = arrayL[0];
360 
361  assert( num_SL == 1 );
362  assert( jumpL[1] == 1 );
363  assert( twoSL[0] == Prob->gTwoS() );
364  assert( NL == Prob->gN() );
365  assert( IL == Prob->gIrrep() );
366 
367  delete [] arrayL;
368  delete [] arrayR;
369  delete [] twoSL;
370  delete [] twoSR;
371  delete [] jumpL;
372  delete [] jumpR;
373 
374  return theCoeff;
375 
376 }

378 void CheMPS2::DMRG::calcVeffTilde(double * result, Sobject * currentS, int state_number){
379 
380  int dimTot = currentS->gKappa2index(currentS->gNKappa());
381  for (int cnt=0; cnt<dimTot; cnt++){ result[cnt] = 0.0; }
382  int index = currentS->gIndex();
383 
384  const int dimL = std::max(denBK->gMaxDimAtBound(index), Exc_BKs[state_number]->gMaxDimAtBound(index) );
385  const int dimR = std::max(denBK->gMaxDimAtBound(index+2), Exc_BKs[state_number]->gMaxDimAtBound(index+2) );
386  double * workmem = new double[dimL * dimR];
387 
388  //Construct Sup
389  Sobject * Sup = new Sobject(index,Exc_BKs[state_number]->gIrrep(index),Exc_BKs[state_number]->gIrrep(index+1),Exc_BKs[state_number]);
390  Sup->Join(Exc_MPSs[state_number][index],Exc_MPSs[state_number][index+1]);
391 
392  //Construct VeffTilde
393  const double prefactor = sqrt(Exc_Eshifts[state_number]) / (Prob->gTwoS() + 1.0);
394  for (int ikappa=0; ikappa<currentS->gNKappa(); ikappa++){
395  int NL = currentS->gNL(ikappa);
396  int TwoSL = currentS->gTwoSL(ikappa);
397  int IL = currentS->gIL(ikappa);
398  int N1 = currentS->gN1(ikappa);
399  int N2 = currentS->gN2(ikappa);
400  int TwoJ = currentS->gTwoJ(ikappa);
401  int NR = currentS->gNR(ikappa);
402  int TwoSR = currentS->gTwoSR(ikappa);
403  int IR = currentS->gIR(ikappa);
404 
405  //Check if block also exists for other MPS
406  int kappaSup = Sup->gKappa(NL, TwoSL, IL, N1, N2, TwoJ, NR, TwoSR, IR);
407  if (kappaSup!=-1){
408 
409  int dimLdown = denBK->gCurrentDim(index, NL,TwoSL,IL);
410  int dimLup = Exc_BKs[state_number]->gCurrentDim(index, NL,TwoSL,IL);
411  int dimRdown = denBK->gCurrentDim(index+2,NR,TwoSR,IR);
412  int dimRup = Exc_BKs[state_number]->gCurrentDim(index+2,NR,TwoSR,IR);
413 
414  //Do sqrt( (TwoJR+1) * Eshift ) / (TwoStarget+1) times (OL * Sup)_{block} --> workmem
415  double * SupPart = Sup->gStorage() + Sup->gKappa2index(kappaSup);
416  double alpha = prefactor * sqrt(TwoSR+1.0);
417  if (index==0){
418 
419  int dimBlock = dimLup * dimRup;
420  int inc = 1;
421  dcopy_(&dimBlock,SupPart,&inc,workmem,&inc);
422  dscal_(&dimBlock,&alpha,workmem,&inc);
423 
424  } else {
425 
426  char notrans = 'N';
427  double beta = 0.0;
428  double * Opart = Exc_Overlaps[state_number][index-1]->gStorage(NL,TwoSL,IL,NL,TwoSL,IL);
429  dgemm_(&notrans,&notrans,&dimLdown,&dimRup,&dimLup,&alpha,Opart,&dimLdown,SupPart,&dimLup,&beta,workmem,&dimLdown);
430 
431  }
432 
433  //Do (workmem * OR)_{block} --> result + jumpCurrentS
434  int jumpCurrentS = currentS->gKappa2index(ikappa);
435  if (index==L-2){
436 
437  int dimBlock = dimLdown * dimRdown;
438  int inc = 1;
439  dcopy_(&dimBlock, workmem, &inc, result + jumpCurrentS, &inc);
440 
441  } else {
442 
443  char trans = 'T';
444  char notrans = 'N';
445  alpha = 1.0;
446  double beta = 0.0; //set
447  double * Opart = Exc_Overlaps[state_number][index+1]->gStorage(NR,TwoSR,IR,NR,TwoSR,IR);
448  dgemm_(&notrans,&trans,&dimLdown,&dimRdown,&dimRup,&alpha,workmem,&dimLdown,Opart,&dimRdown,&beta,result+jumpCurrentS,&dimLdown);
449 
450  }
451  }
452  }
453 
454  //Deallocate everything
455  delete Sup;
456  delete [] workmem;
457 
458 }
