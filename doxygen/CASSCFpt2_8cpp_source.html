42 double CheMPS2::CASSCF::caspt2( const int Nelectrons, const int TwoS, const int Irrep, ConvergenceScheme * OptScheme, const int rootNum, DMRGSCFoptions * theDMRGSCFoptions, const double IPEA, const double IMAG ){
43 
44  const int num_elec = Nelectrons - 2 * iHandler->getNOCCsum();
45  assert( num_elec >= 0 );
46 
47  //Determine the maximum NORB(irrep) and the max_block_size for the ERI orbital rotation
48  const int maxlinsize = iHandler->getNORBmax();
49  const bool block_wise = (( maxlinsize <= CheMPS2::DMRGSCF_maxlinsizeCutoff ) ? false : true ); //Only if bigger, do we want to work blockwise
50  const int block_size = (( block_wise ) ? orbital_rotation_size( maxlinsize, CheMPS2::DMRGSCF_maxlinsizeCutoff, iHandler ) : maxlinsize );
51 
52  //Allocate 2-body rotation memory: One array is approx (block_size/273.0)^4 * 42 GiB --> [block_size=100 --> 750 MB]
53  const int block_size_power4 = block_size * block_size * block_size * block_size; //Note that 273**4 overfloats the 32 bit integer!!!
54  const int tot_dmrg_power4 = nOrbDMRG * nOrbDMRG * nOrbDMRG * nOrbDMRG;
55  const int tot_dmrg_power6 = tot_dmrg_power4 * nOrbDMRG * nOrbDMRG;
56  const int work_mem_size = max( max( block_size_power4 , maxlinsize * maxlinsize * 4 ) , tot_dmrg_power4 ); //For (ERI rotation, update unitary, block diagonalize, orbital localization)
57  double * mem1 = new double[ work_mem_size ];
58  double * mem2 = new double[ work_mem_size ];
59  double * mem3 = NULL;
60  if ( block_wise ){ mem3 = new double[ block_size_power4 ]; }
61 
62  // Rotate to pseudocanonical orbitals
63  buildTmatrix();
64  buildQmatOCC();
65  buildQmatACT(); // DMRG1RDM needs to be set by CASSCF::solve
66  construct_fock( theFmatrix, theTmatrix, theQmatOCC, theQmatACT, iHandler );
67  block_diagonalize( 'O', theFmatrix, unitary, mem1, mem2, iHandler, false, NULL );
68  block_diagonalize( 'A', theFmatrix, unitary, mem1, mem2, iHandler, false, DMRG2DM );
69  block_diagonalize( 'V', theFmatrix, unitary, mem1, mem2, iHandler, false, NULL );
70 
71  // Fill active space Hamiltonian
72  Hamiltonian * HamAS = new Hamiltonian(nOrbDMRG, SymmInfo.getGroupNumber(), iHandler->getIrrepOfEachDMRGorbital());
73  Problem * Prob = new Problem(HamAS, TwoS, num_elec, Irrep);
74  Prob->SetupReorderD2h(); //Doesn't matter if the group isn't D2h, Prob checks it.
75  buildQmatOCC();
76  buildTmatrix();
77  fillConstAndTmatDMRG(HamAS);
78  DMRGSCFVmatRotations theRotator(HamOrig, iHandler);
79  if ( block_wise ){ theRotator.fillVmatDMRGBlockWise(HamAS, unitary, mem1, mem2, mem3, block_size); }
80  else { theRotator.fillVmatDMRG(HamAS, unitary, mem1, mem2); }
81  double E_CASSCF = 0.0;
82  double * three_dm = new double[ tot_dmrg_power6 ];
83  double * contract = new double[ tot_dmrg_power6 ];
84 
85  // Solve the active space problem
86  if (( OptScheme == NULL ) && ( rootNum == 1 )){ // Do FCI
87 
88  const int nalpha = ( num_elec + TwoS ) / 2;
89  const int nbeta = ( num_elec - TwoS ) / 2;
90  const double workmem = 1000.0; // 1GB
91  const int verbose = 2;
92  CheMPS2::FCI * theFCI = new CheMPS2::FCI( HamAS, nalpha, nbeta, Irrep, workmem, verbose );
93  double * inoutput = new double[ theFCI->getVecLength(0) ];
94  theFCI->ClearVector( theFCI->getVecLength(0), inoutput );
95  inoutput[ theFCI->LowestEnergyDeterminant() ] = 1.0;
96  E_CASSCF = theFCI->GSDavidson( inoutput );
97  theFCI->Fill2RDM( inoutput, DMRG2DM ); // 2-RDM
98  theFCI->Fill3RDM( inoutput, three_dm ); // 3-RDM
99  setDMRG1DM( num_elec, nOrbDMRG, DMRG1DM, DMRG2DM ); // 1-RDM
100  buildQmatACT();
101  construct_fock( theFmatrix, theTmatrix, theQmatOCC, theQmatACT, iHandler );
102  copy_active( theFmatrix, mem2, iHandler ); // Fock
103  theFCI->Fock4RDM( inoutput, three_dm, mem2, contract ); // trace( Fock * 4-RDM )
104  delete theFCI;
105  delete [] inoutput;
106 
107  } else { // Do the DMRG sweeps
108 
109  for ( int cnt = 0; cnt < tot_dmrg_power4; cnt++ ){ DMRG2DM[ cnt ] = 0.0; } // Clear the 2-RDM
110  CheMPS2::DMRG * theDMRG = new DMRG(Prob, OptScheme);
111  for (int state = 0; state < rootNum; state++){
112  if (state > 0){ theDMRG->newExcitation( fabs( E_CASSCF ) ); }
113  E_CASSCF = theDMRG->Solve();
114  if ((state == 0) && (rootNum > 1)){ theDMRG->activateExcitations( rootNum-1 ); }
115  }
116  theDMRG->calc_rdms_and_correlations( true );
117  copy2DMover( theDMRG->get2DM(), nOrbDMRG, DMRG2DM ); // 2-RDM
118  setDMRG1DM( num_elec, nOrbDMRG, DMRG1DM, DMRG2DM ); // 1-RDM
119  buildQmatACT();
120  construct_fock( theFmatrix, theTmatrix, theQmatOCC, theQmatACT, iHandler );
121  copy_active( theFmatrix, mem2, iHandler ); // Fock
122  //CheMPS2::Cumulant::gamma4_fock_contract_ham( Prob, theDMRG->get3DM(), theDMRG->get2DM(), mem2, contract );
123  for ( int cnt = 0; cnt < tot_dmrg_power6; cnt++ ){ contract[ cnt ] = 0.0; }
124  for ( int ham_orbz = 0; ham_orbz < nOrbDMRG; ham_orbz++ ){
125  theDMRG->Diag4RDM( three_dm, ham_orbz, false );
126  int size = tot_dmrg_power6;
127  double f_zz = mem2[ ham_orbz + nOrbDMRG * ham_orbz ];
128  int inc1 = 1;
129  daxpy_( &size, &f_zz, three_dm, &inc1, contract, &inc1 ); // trace( Fock * 4-RDM )
130  }
131  copy3DMover( theDMRG->get3DM(), nOrbDMRG, three_dm ); // 3-RDM --> three_dm was used as work space for the constracted 4-RDM
132  if (CheMPS2::DMRG_storeMpsOnDisk){ theDMRG->deleteStoredMPS(); }
133  if (CheMPS2::DMRG_storeRenormOptrOnDisk){ theDMRG->deleteStoredOperators(); }
134  delete theDMRG;
135 
136  }
137 
138  delete Prob;
139  delete HamAS;
140 
141  //Calculate the matrix elements needed to calculate the gradient and hessian
142  if ( block_wise ){ theRotator.fillRotatedTEIBlockWise(theRotatedTEI, unitary, mem1, mem2, mem3, block_size); }
143  else { theRotator.fillRotatedTEI( theRotatedTEI, unitary, mem1, mem2 ); }
144 
145  delete [] mem1;
146  delete [] mem2;
147  if ( block_wise ){ delete [] mem3; }
148 
149  cout << "CASPT2 : Norm F - F_pseudocan = " << deviation_from_blockdiag( theFmatrix, iHandler ) << endl;
150  CheMPS2::CASPT2 * myCASPT2 = new CheMPS2::CASPT2( iHandler, theRotatedTEI, theTmatrix, theFmatrix, DMRG1DM, DMRG2DM, three_dm, contract, IPEA );
151  const double E_CASPT2 = myCASPT2->solve( IMAG );
152 
153  delete myCASPT2;
154  delete [] three_dm;
155  delete [] contract;
156 
157  return E_CASPT2;
158 
159 }

177 double CheMPS2::CASSCF::deviation_from_blockdiag( DMRGSCFmatrix * matrix, const DMRGSCFindices * idx ){
178 
179  double rms_deviation = 0.0;
180  const int n_irreps = idx->getNirreps();
181  for ( int irrep = 0; irrep < n_irreps; irrep++ ){
182  const int NOCC = idx->getNOCC( irrep );
183  for ( int row = 0; row < NOCC; row++ ){
184  for ( int col = row + 1; col < NOCC; col++ ){
185  rms_deviation += matrix->get( irrep, row, col ) * matrix->get( irrep, row, col );
186  rms_deviation += matrix->get( irrep, col, row ) * matrix->get( irrep, col, row );
187  }
188  }
189  const int NACT = idx->getNDMRG( irrep );
190  for ( int row = 0; row < NACT; row++ ){
191  for ( int col = row + 1; col < NACT; col++ ){
192  rms_deviation += matrix->get( irrep, NOCC + row, NOCC + col ) * matrix->get( irrep, NOCC + row, NOCC + col );
193  rms_deviation += matrix->get( irrep, NOCC + col, NOCC + row ) * matrix->get( irrep, NOCC + col, NOCC + row );
194  }
195  }
196  const int NVIR = idx->getNVIRT( irrep );
197  const int N_OA = NOCC + NACT;
198  for ( int row = 0; row < NVIR; row++ ){
199  for ( int col = row + 1; col < NVIR; col++ ){
200  rms_deviation += matrix->get( irrep, N_OA + row, N_OA + col ) * matrix->get( irrep, N_OA + row, N_OA + col );
201  rms_deviation += matrix->get( irrep, N_OA + col, N_OA + row ) * matrix->get( irrep, N_OA + col, N_OA + row );
202  }
203  }
204  }
205  rms_deviation = sqrt( rms_deviation );
206  return rms_deviation;
207 
208 }
