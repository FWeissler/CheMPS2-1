33 CheMPS2::Molden::Molden( const int L, const int group, int * irrep_sizes ){

35  this->L = L;

37  this->num_irreps = SymmInfo.getNumberOfIrreps();

39  Isizes = new int[ num_irreps ];

68 void CheMPS2::Molden::read_molden( const string filename ){
69 
70  int * Icounter = new int[ num_irreps ];
71  for ( int irrep = 0; irrep < num_irreps; irrep++ ){ Icounter[ irrep ] = 0; }
72 
73  int * psi2molpro = new int[ num_irreps ];
74  CheMPS2::Irreps::symm_psi2molpro( psi2molpro, SymmInfo.getGroupName() );
75 
76  string line, part;
77 
78  std::ifstream inputfile( filename.c_str() );
79 
80  bool stop = false;
81  string start = "[MO]";
82  do {
83  getline( inputfile, line );
84  const int pos = line.find( start );
85  if ( pos != string::npos ){ stop = true; }
86  } while ( stop == false );
87 
88  int total_num_orbs = 0;
89  for ( int irrep = 0; irrep < num_irreps; irrep++ ){ total_num_orbs += Isizes[ irrep ]; }
90 
91  for ( int orbs = 0; orbs < total_num_orbs; orbs++ ){ // Only read in the alpha orbitals
92 
93  // Read in the Symm
94  getline( inputfile, line );
95  const int pos_equal = line.find( '=' );
96  const int pos_dot = line.find( '.', pos_equal + 1 );
97  int psi4_irrep = -1;
98  if ( pos_dot != string::npos ){ // Molpro
99  part = line.substr( pos_dot + 1, line.size() - 1 - pos_dot );
100  const int molpro_irrep = atoi( part.c_str() );
101  for ( int irrep = 0; irrep < num_irreps; irrep++ ){
102  if ( psi2molpro[ irrep ] == molpro_irrep ){ psi4_irrep = irrep; }
103  }
104  std::cout << "MO " << orbs + 1 << " has molpro_irrep " << molpro_irrep << " and psi4_irrep " << SymmInfo.getIrrepName( psi4_irrep ) << "." << std::endl;
105  } else { // Psi4
106  part = line.substr( pos_equal + 1, line.size() - 1 - pos_equal );
107  for ( int irrep = 0; irrep < num_irreps; irrep++ ){
108  if ( part.find( SymmInfo.getIrrepName( irrep ) ) != string::npos ){ psi4_irrep = irrep; }
109  }
110  std::cout << "MO " << orbs + 1 << " has psi4_irrep " << SymmInfo.getIrrepName( psi4_irrep ) << "." << std::endl;
111  }
112  assert( psi4_irrep != -1 );
113 
114  // Read in the Ene, Spin, Occup lines
115  for ( int cnt = 0; cnt < 3; cnt++ ){ getline( inputfile, line ); }
116 
117  for ( int cnt = 0; cnt < L; cnt++ ){
118  getline( inputfile, line );
119  const int pos_number = line.find_first_not_of( ' ' ); // Start of the primitive counter number
120  const int pos_gap = line.find( ' ', pos_number ); // Gap between the primitive number and the actual coefficient
121  part = line.substr( pos_gap + 1, line.size() - 1 - pos_gap );
122  const double value = atof( part.c_str() );
123  molden[ psi4_irrep ][ cnt + L * Icounter[ psi4_irrep ] ] = value;
124  std::cout << cnt+1 << " " << value << std::endl;
125  }
126 
127  Icounter[ psi4_irrep ] += 1;
128 
129  }
130 
131  for ( int irrep = 0; irrep < num_irreps; irrep++ ){
132  assert( Icounter[ irrep ] == Isizes[ irrep ] );
133  }
134 
135  delete [] Icounter;
136  delete [] psi2molpro;
137 
138  inputfile.close();
139 
140 }

148 void CheMPS2::Molden::print( const string original, const string output ){
149 
150  char trans = 'T';
151  char notrans = 'N';
152  double one = 1.0;
153  double set = 0.0;
154  for ( int irrep = 0; irrep < num_irreps; irrep++ ){
155  if ( Isizes[ irrep ] > 0 ){
156  dgemm_( &notrans, &trans, &L, Isizes + irrep, Isizes + irrep, &one, molden[ irrep ], &L, unitary[ irrep ], Isizes + irrep, &set, product[ irrep ], &L );
157  }
158  }
159 
160  string line, part;
161 
162  std::ifstream inputfile( original.c_str() );
163  std::ofstream outputfile( output.c_str(), std::ofstream::trunc );
164 
165  // First go to the start of the orbital coefficient dump
166  bool stop = false;
167  string start = "[MO]";
168  do {
169  getline( inputfile,line );
170  outputfile << line << std::endl;
171  const int pos = line.find( start );
172  if ( pos != string::npos ){ stop = true; }
173  } while ( stop == false );
174 
175  inputfile.close();
176 
177  for ( int irrep = 0; irrep < num_irreps; irrep++ ){
178  for ( int orb = 0; orb < Isizes[ irrep ]; orb++ ){
179  outputfile << " Sym= " << SymmInfo.getIrrepName( irrep ) << std::endl;
180  outputfile << " Ene= N/A" << std::endl;
181  outputfile << " Spin= Restricted" << std::endl;
182  outputfile << " Occup= N/A" << std::endl;
183  for ( int cnt = 0; cnt < L; cnt++ ){
184  outputfile << cnt + 1 << " " << product[ irrep ][ cnt + L * orb ] << std::endl;
185  }
186  }
187  }
188 
189  outputfile.close();
190 
191 }
