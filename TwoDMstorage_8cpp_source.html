46 long long CheMPS2::TwoDMstorage::calcNumberOfElements(const bool allocate){
47 
48  //The object size: see text above storage in TwoDMstorage.h
49  long long theTotalSize = 0;
50  const int NumIrreps = SymmInfo.getNumberOfIrreps();
51 
52  for (int Icenter=0; Icenter<NumIrreps; Icenter++){
53 
54  if (allocate){ storage[Icenter] = new long long * [NumIrreps]; } //ALLOCATION
55 
56  for (int I_i=0; I_i<NumIrreps; I_i++){
57  const int I_j = Irreps::directProd(Icenter,I_i);
58  if ((I_i <= I_j) && (Isizes[I_i]>0) && (Isizes[I_j]>0)){ //I_i <= I_j and both I_i and I_j have orbitals
59 
60  if (allocate){ storage[Icenter][I_i] = new long long[NumIrreps - I_i]; } //ALLOCATION
61 
62  for (int I_k=I_i; I_k<NumIrreps; I_k++){ //I_i <= I_k
63  const int I_l = Irreps::directProd(Icenter,I_k);
64  if ((I_i <= I_l) && (Isizes[I_k]>0) && (Isizes[I_l]>0)){ //I_i <= I_l and both I_k and I_l have orbitals
65 
66  storage[Icenter][I_i][I_k - I_i] = theTotalSize;
67  theTotalSize += Isizes[I_i] * Isizes[I_j] * Isizes[I_k] * Isizes[I_l];
68 
69  }
70  }
71  if (!allocate){ delete [] storage[Icenter][I_i]; } //DEALLOCATION
72  }
73  }
74  if (!allocate){ delete [] storage[Icenter]; } //DEALLOCATION
75  }
76 
77  return theTotalSize;
78 
79 }

102 long long CheMPS2::TwoDMstorage::getPointer(const int irrep_i, const int irrep_j, const int irrep_k, const int irrep_l, const int i, const int j, const int k, const int l) const {
103 
104  const int Icenter = Irreps::directProd(irrep_i,irrep_j);
105 
106  if ( Icenter == Irreps::directProd(irrep_k,irrep_l) ){
107 
108  if ((irrep_i <= irrep_j) && (irrep_i <= irrep_k) && (irrep_i <= irrep_l)){ // Irrep I_i is the smallest : (ijkl)
109  return storage[Icenter][irrep_i][irrep_k - irrep_i] + i + Isizes[irrep_i] * ( j + Isizes[irrep_j] * ( k + Isizes[irrep_k] * l ) );
110  }
111 
112  if ((irrep_j <= irrep_i) && (irrep_j <= irrep_k) && (irrep_j <= irrep_l)){ // Irrep I_j is the smallest : (jilk)
113  return storage[Icenter][irrep_j][irrep_l - irrep_j] + j + Isizes[irrep_j] * ( i + Isizes[irrep_i] * ( l + Isizes[irrep_l] * k ) );
114  }
115 
116  if ((irrep_k <= irrep_i) && (irrep_k <= irrep_j) && (irrep_k <= irrep_l)){ // Irrep I_k is the smallest : (klij)
117  return storage[Icenter][irrep_k][irrep_i - irrep_k] + k + Isizes[irrep_k] * ( l + Isizes[irrep_l] * ( i + Isizes[irrep_i] * j ) );
118  }
119 
120  if ((irrep_l <= irrep_i) && (irrep_l <= irrep_j) && (irrep_l <= irrep_k)){ // Irrep I_l is the smallest : (lkji)
121  return storage[Icenter][irrep_l][irrep_j - irrep_l] + l + Isizes[irrep_l] * ( k + Isizes[irrep_k] * ( j + Isizes[irrep_j] * i ) );
122  }
123 
124  }
125 
126  return -1;
127 
128 }

204 
205  hid_t attribute_id2 = H5Aopen_by_name(group_id,"IrrepSizes", "nIrreps", H5P_DEFAULT, H5P_DEFAULT);
206  int nIrreps;
207  H5Aread(attribute_id2, H5T_NATIVE_INT, &nIrreps);
208  assert( nIrreps==SymmInfo.getNumberOfIrreps() );
209 
210  hid_t attribute_id3 = H5Aopen_by_name(group_id,"IrrepSizes", "theTotalSize", H5P_DEFAULT, H5P_DEFAULT);
211  long long theTotalSize;
212  H5Aread(attribute_id3, H5T_NATIVE_LLONG, &theTotalSize);
213  assert( theTotalSize==arrayLength );
214 
215  H5Aclose(attribute_id1);
216  H5Aclose(attribute_id2);
217  H5Aclose(attribute_id3);
218 
219  int * IsizesAgain = new int[SymmInfo.getNumberOfIrreps()];
220  H5Dread(dataset_id, H5T_NATIVE_INT, H5S_ALL, H5S_ALL, H5P_DEFAULT, IsizesAgain);
221  for (int cnt=0; cnt<SymmInfo.getNumberOfIrreps(); cnt++){
222  assert( IsizesAgain[cnt]==Isizes[cnt] );
223  }
224  delete [] IsizesAgain;
225  H5Dclose(dataset_id);
226 
227  H5Gclose(group_id);
228 
229  std::cout << "TwoDMstorage::read : loading " << arrayLength << " doubles." << std::endl;
230 
231  //The object itself.
232  hid_t group_id7 = H5Gopen(file_id, "/TwoDMstorageObject", H5P_DEFAULT);
233 
234  hid_t dataset_id7 = H5Dopen(group_id7, "TwoDM elements", H5P_DEFAULT);
235  H5Dread(dataset_id7, H5T_NATIVE_DOUBLE, H5S_ALL, H5S_ALL, H5P_DEFAULT, theElements);

238  H5Gclose(group_id7);

240  H5Fclose(file_id);
