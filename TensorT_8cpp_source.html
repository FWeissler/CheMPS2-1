157 void CheMPS2::TensorT::QR(TensorDiag * Rstorage){
158 
159  //Left normalization occurs in T-convention: no pre or after multiplication
160  //Work per right symmetry sector
161 
162  //PARALLEL
163  #pragma omp parallel for schedule(dynamic)
164  for (int NR=denBK->gNmin(index+1); NR<=denBK->gNmax(index+1); NR++){
165  for (int TwoSR=denBK->gTwoSmin(index+1,NR); TwoSR<=denBK->gTwoSmax(index+1,NR); TwoSR+=2){
166  for (int IR=0; IR<denBK->getNumberOfIrreps(); IR++){
167  int dimR = denBK->gCurrentDim(index+1,NR,TwoSR,IR);
168  if (dimR>0){
169 
170  //Find out the total left dimension
171  int dimLtotal = 0;
172  for (int ikappa=0; ikappa<nKappa; ikappa++){
173  if ((NR==sectorNR[ikappa])&&(TwoSR==sectorTwoSR[ikappa])&&(IR==sectorIR[ikappa])){
174  dimLtotal += denBK->gCurrentDim(index,sectorN1[ikappa],sectorTwoS1[ikappa],sectorI1[ikappa]);
175  }
176  }
177 
178  if (dimLtotal>0){ //Due to the initial truncation, it is possible that dimLtotal is temporarily smaller than dimR ...
179 
180  double * mem = new double[dimLtotal*dimR];
181  //Copy the relevant parts from storage to mem
182  int dimLtotal2 = 0;
183  for (int ikappa=0; ikappa<nKappa; ikappa++){
184  if ((NR==sectorNR[ikappa])&&(TwoSR==sectorTwoSR[ikappa])&&(IR==sectorIR[ikappa])){
185  int dimL = denBK->gCurrentDim(index,sectorN1[ikappa],sectorTwoS1[ikappa],sectorI1[ikappa]);
186  if (dimL>0){
187  for (int l=0; l<dimL; l++){
188  for (int r=0; r<dimR; r++){
189  mem[dimLtotal2 + l + dimLtotal * r] = storage[kappa2index[ikappa] + l + dimL * r];
190  }
191  }
192  dimLtotal2 += dimL;
193  }
194  }
195  }
196 
197  //QR mem --> m = dimLtotal ; n = dimR
198  int info;
199  int minofdims = min(dimR,dimLtotal);
200  double * tau = new double[minofdims];
201  double * work = new double[dimR];
202  dgeqrf_(&dimLtotal,&dimR,mem,&dimLtotal,tau,work,&dimR,&info);
203 
204  //Copy R to Rstorage
205  double * wheretoput = Rstorage->gStorage(NR,TwoSR,IR,NR,TwoSR,IR); //dimR x dimR
206 
207  for (int irow = 0; irow<minofdims; irow++){
208  for (int icol = 0; icol<irow; icol++){
209  wheretoput[irow + dimR * icol] = 0.0;
210  }
211  for (int icol = irow; icol<dimR; icol++){
212  wheretoput[irow + dimR * icol] = mem[irow + dimLtotal * icol];
213  }
214  }
215  for (int irow = minofdims; irow<dimR; irow++){
216  for (int icol = 0; icol<dimR; icol++){
217  wheretoput[irow + dimR * icol] = 0.0;
218  }
219  }
220 
221  //Construct Q
222  dorgqr_(&dimLtotal,&minofdims,&minofdims,mem,&dimLtotal,tau,work,&dimR,&info);
223  if (dimLtotal < dimR){ //if number of cols larger than number of rows, rest of cols zero.
224  for (int irow=0; irow<dimLtotal; irow++){
225  for (int icol=dimLtotal; icol<dimR; icol++){
226  mem[irow + dimLtotal * icol] = 0.0;
227  }
228  }
229  }
230 
231  //Copy from mem to storage
232  dimLtotal2 = 0;
233  for (int ikappa=0; ikappa<nKappa; ikappa++){
234  if ((NR==sectorNR[ikappa])&&(TwoSR==sectorTwoSR[ikappa])&&(IR==sectorIR[ikappa])){
235  int dimL = denBK->gCurrentDim(index,sectorN1[ikappa],sectorTwoS1[ikappa],sectorI1[ikappa]);
236  if (dimL>0){
237  for (int l=0; l<dimL; l++){
238  for (int r=0; r<dimR; r++){
239  storage[kappa2index[ikappa] + l + dimL * r] = mem[dimLtotal2 + l + dimLtotal * r];
240  }
241  }
242  dimLtotal2 += dimL;
243  }
244  }
245  }
246 
247  //Clear the memory
248  delete [] work;
249  delete [] tau;
250  delete [] mem;
251 
252  }
253  }
254  }
255  }

260 void CheMPS2::TensorT::LQ(TensorDiag * Lstorage){
261 
262  //Right normalization occurs in U-convention: pre-multiplication with sqrt{2jR+1/2jL+1} and after multiplication with sqrt{2jL+1/2jR+1}
263  //Work per left symmetry sector
264 
265  //PARALLEL
266  #pragma omp parallel for schedule(dynamic)
267  for (int NL=denBK->gNmin(index); NL<=denBK->gNmax(index); NL++){
268  for (int TwoSL=denBK->gTwoSmin(index,NL); TwoSL<=denBK->gTwoSmax(index,NL); TwoSL+=2){
269  for (int IL=0; IL<denBK->getNumberOfIrreps(); IL++){
270  int dimL = denBK->gCurrentDim(index,NL,TwoSL,IL);
271  if (dimL>0){
272 
273  //Find out the total right dimension
274  int dimRtotal = 0;
275  for (int ikappa=0; ikappa<nKappa; ikappa++){
276  if ((NL==sectorN1[ikappa])&&(TwoSL==sectorTwoS1[ikappa])&&(IL==sectorI1[ikappa])){
277  dimRtotal += denBK->gCurrentDim(index+1,sectorNR[ikappa],sectorTwoSR[ikappa],sectorIR[ikappa]);
278  }
279  }
280 
281  if (dimRtotal>0){ //Due to the initial truncation, it is possible that dimRtotal is temporarily smaller than dimL ...
282 
283  double * mem = new double[dimRtotal*dimL];
284  //Copy the relevant parts from storage to mem & multiply with factor !!
285  int dimRtotal2 = 0;
286  for (int ikappa=0; ikappa<nKappa; ikappa++){
287  if ((NL==sectorN1[ikappa])&&(TwoSL==sectorTwoS1[ikappa])&&(IL==sectorI1[ikappa])){
288  int dimR = denBK->gCurrentDim(index+1,sectorNR[ikappa],sectorTwoSR[ikappa],sectorIR[ikappa]);
289  if (dimR>0){
290  double factor = sqrt((sectorTwoSR[ikappa]+1.0)/(TwoSL+1.0));
291  for (int l=0; l<dimL; l++){
292  for (int r=0; r<dimR; r++){
293  mem[l + dimL * (dimRtotal2 + r)] = factor * storage[kappa2index[ikappa] + l + dimL * r];
294  }
295  }
296  dimRtotal2 += dimR;
297  }
298  }
299  }
300 
301  //LQ mem --> m = dimL ; n = dimRtotal
302  int info;
303  int minofdims = min(dimL,dimRtotal);
304  double * tau = new double[minofdims];
305  double * work = new double[dimL];
306  dgelqf_(&dimL,&dimRtotal,mem,&dimL,tau,work,&dimL,&info);
307 
308 
309 
310  //Copy L to Lstorage
311  double * wheretoput = Lstorage->gStorage(NL,TwoSL,IL,NL,TwoSL,IL); //dimL x dimL
312 
313  for (int irow = 0; irow<dimL; irow++){
314  for (int icol = 0; icol<min(irow+1,dimRtotal); icol++){ //icol can be max. irow and max. dimRtotal-1
315  wheretoput[irow + dimL * icol] = mem[irow + dimL * icol];
316  }
317  for (int icol = min(irow+1,dimRtotal); icol<dimL; icol++){
318  wheretoput[irow + dimL * icol] = 0.0;
319  }
320  }
321 
322  //Construct Q
323  dorglq_(&minofdims,&dimRtotal,&minofdims,mem,&dimL,tau,work,&dimL,&info);
324  if (dimRtotal < dimL){ //if number of rows larger than number of cols, rest of rows zero.
325  for (int irow=dimRtotal; irow<dimL; irow++){
326  for (int icol=0; icol<dimRtotal; icol++){
327  mem[irow + dimL * icol] = 0.0;
328  }
329  }
330  }
331 
332  //Copy from mem to storage & multiply with factor !!
333  dimRtotal2 = 0;
334  for (int ikappa=0; ikappa<nKappa; ikappa++){
335  if ((NL==sectorN1[ikappa])&&(TwoSL==sectorTwoS1[ikappa])&&(IL==sectorI1[ikappa])){
336  int dimR = denBK->gCurrentDim(index+1,sectorNR[ikappa],sectorTwoSR[ikappa],sectorIR[ikappa]);
337  if (dimR>0){
338  double factor = sqrt((TwoSL+1.0)/(sectorTwoSR[ikappa]+1.0));
339  for (int l=0; l<dimL; l++){
340  for (int r=0; r<dimR; r++){
341  storage[kappa2index[ikappa] + l + dimL * r] = factor * mem[l + dimL * (r + dimRtotal2)];
342  }
343  }
344  dimRtotal2 += dimR;
345  }
346  }
347  }
348 
349  //Clear the memory
350  delete [] work;
351  delete [] tau;
352  delete [] mem;
353 
354  }
355  }
356  }
357  }

404 bool CheMPS2::TensorT::CheckLeftNormal() const{
405 
406  bool isLeftNormal = true;
407 
408  for (int NR=denBK->gNmin(index+1); NR<=denBK->gNmax(index+1); NR++){
409  for (int TwoSR=denBK->gTwoSmin(index+1,NR); TwoSR<=denBK->gTwoSmax(index+1,NR); TwoSR+=2){
410  for (int IR=0; IR<denBK->getNumberOfIrreps(); IR++){
411  int dimR = denBK->gCurrentDim(index+1,NR,TwoSR,IR);
412  if (dimR>0){
413  double * result = new double[dimR*dimR];
414  bool firsttime = true;
415  for (int NL=NR-2; NL<=NR; NL++){
416  for (int TwoSL=TwoSR-((NR==NL+1)?1:0); TwoSL<TwoSR+2; TwoSL+=2){
417  int IL = (NR==NL+1)?(denBK->directProd(Ilocal,IR)):IR;
418  int dimL = denBK->gCurrentDim(index,NL,TwoSL,IL);
419  if (dimL>0){
420  double * Block = storage + kappa2index[gKappa(NL,TwoSL,IL,NR,TwoSR,IR)];
421  char trans = 'T';
422  char notrans = 'N';
423  double one = 1.0;
424  double beta = (firsttime)?0.0:1.0;
425  dgemm_(&trans,&notrans,&dimR,&dimR,&dimL,&one,Block,&dimL,Block,&dimL,&beta,result,&dimR);
426  firsttime = false;
427  }
428  }
429  }
430  for (int cnt=0; cnt<dimR; cnt++) result[(dimR+1)*cnt] -= 1.0;
431  char norm = 'F'; //Frobenius norm
432  char uplo = 'U'; //Doesn't matter as result is fully filled
433  double TwoNorm = dlansy_(&norm,&uplo,&dimR,result,&dimR,result); //last result is work space, not referenced as norm='F';
434  if (TwoNorm>CheMPS2::TENSORT_orthoComparison) isLeftNormal = false;
435  delete [] result;
436  }
437  }
438  }
439  }
440 
441  return isLeftNormal;
