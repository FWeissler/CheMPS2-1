205 int CheMPS2::Irreps::directProdPrivate(const int nGroup, const int n1, const int n2){
206 
207  if (n1 == n2) return 0;
208  if (n1 > n2) return directProdPrivate(nGroup, n2, n1);
209 
210  //n1<n2 remains
211  if (nGroup <= 3){ //ci, cs or c2 (c1 only n1=n2 possible)
212  return 1; //Off-diag terms are always the non-trivial group
213  }
214 
215  if (nGroup <= 6){ //d2, c2v, c2h (with the conventional order, both diagonals are symmetry axes of the multiplication table)
216  if (n1 == 0) return n2;
217  if (n1 == 1){
218  if (n2 == 2) return 3;
219  if (n2 == 3) return 2;
220  }
221  return 1;
222  }
223 
224  if (nGroup == 7){ //d2h (with the conventional order, both diagonals are symmetry axes of the multiplication table)
225  if (n1 == 0) return n2;
226  if (n1 == 1){
227  if (n2 == 2) return 3;
228  if (n2 == 3) return 2;
229  if (n2 == 4) return 5;
230  if (n2 == 5) return 4;
231  if (n2 == 6) return 7;
232  if (n2 == 7) return 6;
233  }
234  if (n1 == 2){
235  if (n2 == 3) return 1;
236  if (n2 == 4) return 6;
237  if (n2 == 5) return 7;
238  if (n2 == 6) return 4;
239  if (n2 == 7) return 5;
240  }
241  if (n1 == 3){
242  if (n2 == 4) return 7;
243  if (n2 == 5) return 6;
244  if (n2 == 6) return 5;
245  if (n2 == 7) return 4;
246  }
247  if (n1 == 4){
248  if (n2 == 5) return 1;
249  if (n2 == 6) return 2;
250  if (n2 == 7) return 3;
251  }
252  if (n1 == 5){
253  if (n2 == 6) return 3;
254  if (n2 == 7) return 2;
255  }
256  return 1;
257  }
258 
259  return -1;
