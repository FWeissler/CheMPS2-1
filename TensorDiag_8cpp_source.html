29  for (int TwoS=denBK->gTwoSmin(index,N); TwoS<=denBK->gTwoSmax(index,N); TwoS+=2){
30  for (int Icnt=0; Icnt<denBK->getNumberOfIrreps(); Icnt++){
31  int dim = denBK->gCurrentDim(index,N,TwoS,Icnt);
32  if (dim>0) nKappa++;
33  }
34  }
35  }
36 
37  sectorN1 = new int[nKappa];
38  sectorTwoS1 = new int[nKappa];
39  sectorI1 = new int[nKappa];
40  kappa2index = new int[nKappa+1];
41  kappa2index[0] = 0;
42 
43  nKappa = 0;
44  for (int N=denBK->gNmin(index); N<=denBK->gNmax(index); N++){
45  for (int TwoS=denBK->gTwoSmin(index,N); TwoS<=denBK->gTwoSmax(index,N); TwoS+=2){
46  for (int Icnt=0; Icnt<denBK->getNumberOfIrreps(); Icnt++){
47  int dim = denBK->gCurrentDim(index,N,TwoS,Icnt);
48  if (dim>0){
49  sectorN1[nKappa] = N;
50  sectorTwoS1[nKappa] = TwoS;
51  sectorI1[nKappa] = Icnt;
52  nKappa++;
53  kappa2index[nKappa] = kappa2index[nKappa-1] + dim*dim;
54  }
55  }
56  }
57  }
58

77 int CheMPS2::TensorDiag::gKappa(const int N1, const int TwoS1, const int I1, const int N2, const int TwoS2, const int I2) const{
78 
79  if ((N1!=N2) || (TwoS1!=TwoS2) || (I1!=I2)) return -1;
80 
81  for (int cnt=0; cnt<nKappa; cnt++){
82  if ( (sectorN1[cnt]==N1) && (sectorTwoS1[cnt]==TwoS1) && (sectorI1[cnt]==I1) ) return cnt;
83  }
84 
85  return -1;
