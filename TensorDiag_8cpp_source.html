102 void CheMPS2::TensorDiag::updateRight(const int ikappa, Tensor * denT, TensorDiag * diagPrevious, double * workmemLR){
103 
104  int dimR = denBK->gCurrentDim(index, sectorN1[ikappa], sectorTwoS1[ikappa], sectorI1[ikappa]);
105  for (int geval=0; geval<4; geval++){
106  int NL,TwoSL,IL;
107  switch(geval){
108  case 0:
109  NL = sectorN1[ikappa];
110  TwoSL = sectorTwoS1[ikappa];
111  IL = sectorI1[ikappa];
112  break;
113  case 1:
114  NL = sectorN1[ikappa]-2;
115  TwoSL = sectorTwoS1[ikappa];
116  IL = sectorI1[ikappa];
117  break;
118  case 2:
119  NL = sectorN1[ikappa]-1;
120  TwoSL = sectorTwoS1[ikappa]-1;
121  IL = Irreps::directProd( sectorI1[ikappa] , denBK->gIrrep(index-1) );
122  break;
123  case 3:
124  NL = sectorN1[ikappa]-1;
125  TwoSL = sectorTwoS1[ikappa]+1;
126  IL = Irreps::directProd( sectorI1[ikappa] , denBK->gIrrep(index-1) );
127  break;
128  }
129  int dimL = denBK->gCurrentDim(index-1,NL,TwoSL,IL);
130  if (dimL>0){
131 
132  double * BlockT = denT->gStorage(NL,TwoSL,IL,sectorN1[ikappa],sectorTwoS1[ikappa],sectorI1[ikappa]);
133  double * BlockDiag = diagPrevious->gStorage(NL,TwoSL,IL,NL,TwoSL,IL);
134 
135  //T^T * diag --> mem
136  char trans = 'T';
137  char notr = 'N';
138  double alpha = 1.0;
139  double beta = 0.0; //set
140  dgemm_(&trans,&notr,&dimR,&dimL,&dimL,&alpha,BlockT,&dimL,BlockDiag,&dimL,&beta,workmemLR,&dimR);
141 
142  //mem * T --> storage
143  beta = 1.0; //add
144  dgemm_(&notr,&notr,&dimR,&dimR,&dimL,&alpha,workmemLR,&dimR,BlockT,&dimL,&beta,storage+kappa2index[ikappa],&dimR);
145 
146  }
147  }
148 
149 }

155  int NR,TwoSR,IR;

171  break;
172  case 3:
173  NR = sectorN1[ikappa]+1;
174  TwoSR = sectorTwoS1[ikappa]+1;
175  IR = Irreps::directProd( sectorI1[ikappa] , denBK->gIrrep(index) );
176  break;
177  }
178  int dimR = denBK->gCurrentDim(index+1,NR,TwoSR,IR);
179  if (dimR>0){
180 
181  double * BlockT = denT->gStorage(sectorN1[ikappa],sectorTwoS1[ikappa],sectorI1[ikappa],NR,TwoSR,IR);
182  double * BlockDiag = diagPrevious->gStorage(NR,TwoSR,IR,NR,TwoSR,IR);
183 
184  //factor * T * diag --> mem
185  char notr = 'N';
186  double alpha = (geval>1) ? (TwoSR+1.0)/(sectorTwoS1[ikappa]+1.0) : 1.0;
187  double beta = 0.0; //set
188  dgemm_(&notr,&notr,&dimL,&dimR,&dimR,&alpha,BlockT,&dimL,BlockDiag,&dimR,&beta,workmemLR,&dimL);
189 
190  //mem * T^T --> storage

192  alpha = 1.0;
193  beta = 1.0; //add
194  dgemm_(&notr,&trans,&dimL,&dimL,&dimR,&alpha,workmemLR,&dimL,BlockT,&dimL,&beta,storage+kappa2index[ikappa],&dimL);

196  }
