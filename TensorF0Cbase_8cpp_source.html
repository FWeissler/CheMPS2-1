55  nKappa = 0;

134 void CheMPS2::TensorF0Cbase::updateMovingRight(TensorF0Cbase * F0CbasePrevious, TensorT * denT, double * workmem){
135 
136  Clear();
137 
138  for (int ikappa=0; ikappa<nKappa; ikappa++){
139  int dimUR = denBK->gCurrentDim(index, sectorN1[ikappa], sectorTwoS1[ikappa], sectorI1[ikappa]);
140  const int IDR = Irreps::directProd(sectorI1[ikappa], Idiff);
141  int dimDR = denBK->gCurrentDim(index, sectorN1[ikappa], sectorTwoS1[ikappa], IDR);
142 
143  for (int geval=0; geval<4; geval++){
144  int NL,TwoSL,IUL,IDL; //NDL = NUL; TwoSDL = TwoSUL
145  switch(geval){
146  case 0: //N = 0
147  NL = sectorN1[ikappa];
148  TwoSL = sectorTwoS1[ikappa];
149  IUL = sectorI1[ikappa];
150  IDL = IDR;
151  break;
152  case 1: //N = 2
153  NL = sectorN1[ikappa] - 2;
154  TwoSL = sectorTwoS1[ikappa];
155  IUL = sectorI1[ikappa];
156  IDL = IDR;
157  break;
158  case 2: //N = 1
159  NL = sectorN1[ikappa] - 1;
160  TwoSL = sectorTwoS1[ikappa] - 1;
161  IUL = Irreps::directProd( sectorI1[ikappa] , denBK->gIrrep(index-1));
162  IDL = Irreps::directProd( IDR , denBK->gIrrep(index-1) );
163  break;
164  case 3: //N = 1
165  NL = sectorN1[ikappa] - 1;
166  TwoSL = sectorTwoS1[ikappa] + 1;
167  IUL = Irreps::directProd( sectorI1[ikappa] , denBK->gIrrep(index-1));
168  IDL = Irreps::directProd( IDR , denBK->gIrrep(index-1) );
169  break;
170  }
171  int dimUL = denBK->gCurrentDim(index-1,NL,TwoSL,IUL);
172  int dimDL = denBK->gCurrentDim(index-1,NL,TwoSL,IDL);
173 
174  if ((dimUL>0) && (dimDL>0)){
175 
176  double * BlockTup = denT->gStorage(NL,TwoSL,IUL,sectorN1[ikappa],sectorTwoS1[ikappa],sectorI1[ikappa]);
177  double * BlockTdown = denT->gStorage(NL,TwoSL,IDL,sectorN1[ikappa],sectorTwoS1[ikappa],IDR );
178  double * BlockF0CbaseLeft = F0CbasePrevious->gStorage(NL,TwoSL,IUL,NL,TwoSL,IDL);
179 
180  //factor * Tup^T * F0Cbase --> mem
181  double alpha = 1.0; //For tensorF0 always factor = 1
182  char trans = 'T';
183  char notr = 'N';
184  double beta = 0.0; //set mem
185  dgemm_(&trans,&notr,&dimUR,&dimDL,&dimUL,&alpha,BlockTup,&dimUL,BlockF0CbaseLeft,&dimUL,&beta,workmem,&dimUR);
186 
187  //mem * Tdown --> storage + kappa2index[ikappa]
188  beta = 1.0; //add to storage
189  dgemm_(&notr,&notr,&dimUR,&dimDR,&dimDL,&alpha,workmem,&dimUR,BlockTdown,&dimDL,&beta,storage+kappa2index[ikappa],&dimUR);
190 
191  }
192 
193  }
194  }

198 void CheMPS2::TensorF0Cbase::updateMovingLeft(TensorF0Cbase * F0CbasePrevious, TensorT * denT, double * workmem){
199 
200  Clear();
201 
202  for (int ikappa=0; ikappa<nKappa; ikappa++){
203  int dimUL = denBK->gCurrentDim(index,sectorN1[ikappa],sectorTwoS1[ikappa],sectorI1[ikappa]);
204  const int IDL = Irreps::directProd(sectorI1[ikappa],Idiff);
205  int dimDL = denBK->gCurrentDim(index,sectorN1[ikappa],sectorTwoS1[ikappa],IDL);
206 
207  for (int geval=0; geval<4; geval++){
208  int NR,TwoSR,IUR,IDR; //NDR = NUR; TwoSDR = TwoSUR
209  switch(geval){
210  case 0: //N = 0
211  NR = sectorN1[ikappa];
212  TwoSR = sectorTwoS1[ikappa];
213  IUR = sectorI1[ikappa];
214  IDR = IDL;
215  break;
216  case 1: //N = 2
217  NR = sectorN1[ikappa] + 2;
218  TwoSR = sectorTwoS1[ikappa];
219  IUR = sectorI1[ikappa];
220  IDR = IDL;
221  break;
222  case 2: //N = 1
223  NR = sectorN1[ikappa] + 1;
224  TwoSR = sectorTwoS1[ikappa] - 1;
225  IUR = Irreps::directProd( sectorI1[ikappa] , denBK->gIrrep(index) );
226  IDR = Irreps::directProd( IDL , denBK->gIrrep(index) );
227  break;
228  case 3: //N = 1
229  NR = sectorN1[ikappa] + 1;
230  TwoSR = sectorTwoS1[ikappa] + 1;
231  IUR = Irreps::directProd( sectorI1[ikappa] , denBK->gIrrep(index) );
232  IDR = Irreps::directProd( IDL , denBK->gIrrep(index) );
233  break;
234  }
235  int dimUR = denBK->gCurrentDim(index+1,NR,TwoSR,IUR);
236  int dimDR = denBK->gCurrentDim(index+1,NR,TwoSR,IDR);
237 
238  if ((dimUR>0) && (dimDR>0)){
239 
240  double * BlockTup = denT->gStorage(sectorN1[ikappa],sectorTwoS1[ikappa],sectorI1[ikappa],NR,TwoSR,IUR);
241  double * BlockTdown = denT->gStorage(sectorN1[ikappa],sectorTwoS1[ikappa],IDL, NR,TwoSR,IDR);
242  double * BlockF0CbaseRight = F0CbasePrevious->gStorage(NR,TwoSR,IUR,NR,TwoSR,IDR);
243 
244  //factor * Tup * F0CbaseRight --> mem
245  double alpha = 1.0;
246  if (geval>=2){
247  alpha = (TwoSR+1.0)/(sectorTwoS1[ikappa]+1.0);
248  }
249  char notr = 'N';
250  double beta = 0.0; //set mem
251  dgemm_(&notr,&notr,&dimUL,&dimDR,&dimUR,&alpha,BlockTup,&dimUL,BlockF0CbaseRight,&dimUR,&beta,workmem,&dimUL);
252 
253  //mem * Tdown^T --> storage + kappa2index[ikappa]
254  beta = 1.0; //add to storage
255  alpha = 1.0; //factor already with previous multiplication
256  char trans = 'T';
257  dgemm_(&notr,&trans,&dimUL,&dimDL,&dimDR,&alpha,workmem,&dimUL,BlockTdown,&dimDL,&beta,storage+kappa2index[ikappa],&dimUL);
258 
259  }
260  }
261  }
