88  double * BlockTdown = denT->gStorage(NLU+1, TwoSLD, ILD, sectorN1[ikappa]+2, sectorTwoSD[ikappa], IDR);
89  double * BlockL = denL->gStorage(NLU, TwoSLU, ILU, NLU+1, TwoSLD, ILD);

93  char notrans = 'N';

95  if (geval<=1){

97  alpha = fase * sqrt(3.0*(TwoSLD+1)) * gsl_sf_coupling_6j(1,1,2,sectorTwoS1[ikappa],sectorTwoSD[ikappa],TwoSLD);

101  }
102  double beta = 0.0; //set
103  dgemm_(&trans,&notrans,&dimUR,&dimLD,&dimLU,&alpha,BlockTup,&dimLU,BlockL,&dimLU,&beta,workmem,&dimUR);

107  beta = 1.0; // add
