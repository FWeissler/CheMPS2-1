51 void CheMPS2::Problem::SetupReorderD2h(){
52 
53  if (gSy()==7){ //Only if D2h of course
54 
55  bReorderD2h = true;
56  f1 = new int[Ham->getL()];
57  f2 = new int[Ham->getL()];
58 
59  int DMRGirrepOrder[8];
60  DMRGirrepOrder[0] = 0; //Ag sigma
61  DMRGirrepOrder[1] = 5; //B1u sigma^*
62  DMRGirrepOrder[2] = 7; //B3u pi_x
63  DMRGirrepOrder[3] = 2; //B2g pi_x^*
64  DMRGirrepOrder[4] = 6; //B2u pi_y
65  DMRGirrepOrder[5] = 3; //B3g pi_y^*
66  DMRGirrepOrder[6] = 1; //B1g
67  DMRGirrepOrder[7] = 4; //Au
68 
69  int DMRGOrb = 0;
70  for (int irrep=0; irrep<8; irrep++){
71  for (int HamOrb=0; HamOrb<Ham->getL(); HamOrb++){
72  if (Ham->getOrbitalIrrep(HamOrb)==DMRGirrepOrder[irrep]){
73  f1[HamOrb] = DMRGOrb;
74  f2[DMRGOrb] = HamOrb;
75  DMRGOrb++;
76  }
77  }
78  }
79  assert( DMRGOrb==Ham->getL() );
80 
81  }
82 
83 }

117 bool CheMPS2::Problem::checkConsistency() const{
118 
119  Irreps SymmInfo(gSy());
120  if ((gIrrep()<0) || (gIrrep()>=SymmInfo.getNumberOfIrreps())){
121  cout << "Problem::Problem() : Irrep out of bound : Irrep = " << gIrrep() << endl;
122  return false;
123  }
124  if (gTwoS()<0){
125  cout << "Problem::checkConsistency() : TwoS = " << gTwoS() << endl;
126  return false;
127  }
128  if (gN()<0){
129  cout << "Problem::checkConsistency() : N = " << gN() << endl;
130  return false;
131  }
132  if (gL()<0){
133  cout << "Problem::checkConsistency() : L = " << gL() << endl;
134  return false;
135  }
136  if (gN()>2*gL()){
137  cout << "Problem::checkConsistency() : N > 2*L ; N = " << gN() << " and L = " << gL() << endl;
138  return false;
139  }
140  if ( (gN()%2) != (gTwoS()%2) ){
141  cout << "Problem::checkConsistency() : N%2 != TwoS%2 ; N = " << gN() << " and TwoS = " << gTwoS() << endl;
142  return false;
143  }
144  if ( gTwoS() > gL() - abs(gN() - gL()) ){
145  cout << "Problem::checkConsistency() : TwoS > L - |N-L| ; N = " << gN() << " and TwoS = " << gTwoS() << " and L = " << gL() << endl;
146  return false;
147  }
148 
149  return true;
150 
151 }
